4#[verify::library_checker("jump_on_tree")]
5pub fn jump_on_tree(reader: impl Read, mut writer: impl Write) {
6    let s = read_all_unchecked(reader);
7    let mut scanner = Scanner::new(&s);
8    scan!(scanner, n, q, (g, _): @TreeGraphScanner::<usize>::new(n));
9    let la = g.level_ancestor(0);
10    let lca = g.lca(0);
11    for _ in 0..q {
12        scan!(scanner, s, t, i);
13        let l = lca.lca(s, t);
14        let ds = la.depth(s) - la.depth(l);
15        let dt = la.depth(t) - la.depth(l);
16        let ans = if i <= ds {
17            la.la(s, i)
18        } else if i <= ds + dt {
19            la.la(t, ds + dt - i)
20        } else {
21            None
22        };
23        writeln!(writer, "{}", ans.unwrap_or(!0) as isize).ok();
24    }
25}

27#[verify::library_checker("jump_on_tree")]
28pub fn jump_on_tree_batch(reader: impl Read, mut writer: impl Write) {
29    let s = read_all_unchecked(reader);
30    let mut scanner = Scanner::new(&s);
31    scan!(scanner, n, q, (g, _): @TreeGraphScanner::<usize>::new(n), queries: [(usize, usize, usize)]);
32    let lca = g.lca(0);
33    let depth = g.tree_depth(0);
34    let results = g.level_ancestor_batch(
35        0,
36        queries.take(q).map(|(s, t, i)| {
37            let l = lca.lca(s, t);
38            let ds = (depth[s] - depth[l]) as usize;
39            let dt = (depth[t] - depth[l]) as usize;
40            if i <= ds {
41                (s, i)
42            } else if i <= ds + dt {
43                (t, ds + dt - i)
44            } else {
45                (0, n)
46            }
47        }),
48    );
49    iter_print!(writer, @lf @it results.iter().map(|&v| v.unwrap_or(!0) as isize));
50}
