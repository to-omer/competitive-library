65    pub fn new(shape: (usize, usize), z: R::T) -> Self {
66        Self {
67            shape,
68            data: vec![vec![z; shape.1]; shape.0],
69            _marker: PhantomData,
70        }

72
73    pub fn from_vec(data: Vec<Vec<R::T>>) -> Self {
74        let shape = (data.len(), data.first().map(Vec::len).unwrap_or_default());
75        assert!(data.iter().all(|r| r.len() == shape.1));
76        Self {
77            shape,
78            data,
79            _marker: PhantomData,
80        }
81    }
82
83    pub fn new_with(shape: (usize, usize), mut f: impl FnMut(usize, usize) -> R::T) -> Self {
84        let data = (0..shape.0)
85            .map(|i| (0..shape.1).map(|j| f(i, j)).collect())
86            .collect();
87        Self {
88            shape,
89            data,
90            _marker: PhantomData,
91        }
92    }
93
94    pub fn zeros(shape: (usize, usize)) -> Self {
95        Self {
96            shape,
97            data: vec![vec![R::zero(); shape.1]; shape.0],
98            _marker: PhantomData,
99        }
100    }
101
102    pub fn eye(shape: (usize, usize)) -> Self {
103        let mut data = vec![vec![R::zero(); shape.1]; shape.0];
104        for (i, d) in data.iter_mut().enumerate().take(shape.1) {
105            d[i] = R::one();
106        }
107        Self {
108            shape,
109            data,
110            _marker: PhantomData,
111        }
112    }
113
114    pub fn map<S, F>(&self, mut f: F) -> Matrix<S>
115    where
116        S: SemiRing,
117        F: FnMut(&R::T) -> S::T,
118    {
119        Matrix::<S>::new_with(self.shape, |i, j| f(&self[i][j]))
120    }
121
122    pub fn add_row_with(&mut self, mut f: impl FnMut(usize, usize) -> R::T) {
123        self.data
124            .push((0..self.shape.1).map(|j| f(self.shape.0, j)).collect());
125        self.shape.0 += 1;
126    }
127
128    pub fn add_col_with(&mut self, mut f: impl FnMut(usize, usize) -> R::T) {
129        for i in 0..self.shape.0 {
130            self.data[i].push(f(i, self.shape.1));
131        }
132        self.shape.1 += 1;
133    }
134
135    pub fn pairwise_assign<F>(&mut self, other: &Self, mut f: F)
136    where
137        F: FnMut(&mut R::T, &R::T),
138    {
139        assert_eq!(self.shape, other.shape);
140        for i in 0..self.shape.0 {
141            for j in 0..self.shape.1 {
142                f(&mut self[i][j], &other[i][j]);
143            }
144        }
145    }

148impl<R> Matrix<R>
149where
150    R: Field,
151    R::Additive: Invertible,
152    R::Multiplicative: Invertible,
153    R::T: PartialEq,
154{
155    /// f: (row, pivot_row, col)
156    pub fn row_reduction_with<F>(&mut self, normalize: bool, mut f: F)
157    where
158        F: FnMut(usize, usize, usize),
159    {
160        let (n, m) = self.shape;
161        let mut c = 0;
162        let mut row_id: Vec<usize> = (0..n).collect();
163        for r in 0..n {
164            loop {
165                if c >= m {
166                    return;
167                }
168                if let Some(pivot) = (r..n).find(|&p| !R::is_zero(&self[p][c])) {
169                    f(r, pivot, c);
170                    self.data.swap(r, pivot);
171                    row_id.swap(r, pivot);
172                    break;
173                };
174                c += 1;
175            }
176            let d = R::inv(&self[r][c]);
177            if normalize {
178                for j in c..m {
179                    R::mul_assign(&mut self[r][j], &d);
180                }
181            }
182            for i in (0..n).filter(|&i| i != r) {
183                let mut e = self[i][c].clone();
184                if !normalize {
185                    R::mul_assign(&mut e, &d);
186                }
187                for j in c..m {
188                    let e = R::mul(&e, &self[r][j]);
189                    R::sub_assign(&mut self[i][j], &e);
190                }
191            }
192            c += 1;
193        }
194    }
195
196    pub fn row_reduction(&mut self, normalize: bool) {
197        self.row_reduction_with(normalize, |_, _, _| {});
198    }
199
200    pub fn rank(&mut self) -> usize {
201        let n = self.shape.0;
202        self.row_reduction(false);
203        (0..n)
204            .filter(|&i| !self.data[i].iter().all(|x| R::is_zero(x)))
205            .count()
206    }
207
208    pub fn determinant(&mut self) -> R::T {
209        assert_eq!(self.shape.0, self.shape.1);
210        let mut neg = false;
211        self.row_reduction_with(false, |r, p, _| neg ^= r != p);
212        let mut d = R::one();
213        if neg {
214            d = R::neg(&d);
215        }
216        for i in 0..self.shape.0 {
217            R::mul_assign(&mut d, &self[i][i]);
218        }
219        d
220    }
221
222    pub fn solve_system_of_linear_equations(&self, b: &[R::T]) -> Option<Vec<R::T>> {
223        assert_eq!(self.shape.0, b.len());
224        let (n, m) = self.shape;
225        let mut c = Matrix::<R>::zeros((n, m + 1));
226        for i in 0..n {
227            c[i][..m].clone_from_slice(&self[i]);
228            c[i][m] = b[i].clone();
229        }
230        c.row_reduction(true);
231        let mut x = vec![R::zero(); m];
232        for i in 0..n {
233            let mut j = 0usize;
234            while j <= m && R::is_zero(&c[i][j]) {
235                j += 1;
236            }
237            if j == m {
238                return None;
239            }
240            if j < m {
241                x[j] = c[i][m].clone();
242            }
243        }
244        Some(x)
245    }
246
247    pub fn inverse(&self) -> Option<Matrix<R>> {
248        assert_eq!(self.shape.0, self.shape.1);
249        let n = self.shape.0;
250        let mut c = Matrix::<R>::zeros((n, n * 2));
251        for i in 0..n {
252            c[i][..n].clone_from_slice(&self[i]);
253            c[i][n + i] = R::one();
254        }
255        c.row_reduction(true);
256        if (0..n).any(|i| R::is_zero(&c[i][i])) {
257            None
258        } else {
259            Some(Self::from_vec(
260                c.data.into_iter().map(|r| r[n..].to_vec()).collect(),
261            ))
262        }
263    }
264
265    pub fn characteristic_polynomial(&mut self) -> Vec<R::T> {
266        let n = self.shape.0;
267        if n == 0 {
268            return vec![R::one()];
269        }
270        assert!(self.data.iter().all(|a| a.len() == n));
271        for j in 0..(n - 1) {
272            if let Some(x) = ((j + 1)..n).find(|&x| !R::is_zero(&self[x][j])) {
273                self.data.swap(j + 1, x);
274                self.data.iter_mut().for_each(|a| a.swap(j + 1, x));
275                let inv = R::inv(&self[j + 1][j]);
276                let mut v = vec![];
277                let src = std::mem::take(&mut self[j + 1]);
278                for a in self.data[(j + 2)..].iter_mut() {
279                    let mul = R::mul(&a[j], &inv);
280                    for (a, src) in a[j..].iter_mut().zip(src[j..].iter()) {
281                        R::sub_assign(a, &R::mul(&mul, src));
282                    }
283                    v.push(mul);
284                }
285                self[j + 1] = src;
286                for a in self.data.iter_mut() {
287                    let v = a[(j + 2)..]
288                        .iter()
289                        .zip(v.iter())
290                        .fold(R::zero(), |s, a| R::add(&s, &R::mul(a.0, a.1)));
291                    R::add_assign(&mut a[j + 1], &v);
292                }
293            }
294        }
295        let mut dp = vec![vec![R::one()]];
296        for i in 0..n {
297            let mut next = vec![R::zero(); i + 2];
298            for (j, dp) in dp[i].iter().enumerate() {
299                R::sub_assign(&mut next[j], &R::mul(dp, &self[i][i]));
300                R::add_assign(&mut next[j + 1], dp);
301            }
302            let mut mul = R::one();
303            for j in (0..i).rev() {
304                mul = R::mul(&mul, &self[j + 1][j]);
305                let c = R::mul(&mul, &self[j][i]);
306                for (next, dp) in next.iter_mut().zip(dp[j].iter()) {
307                    R::sub_assign(next, &R::mul(&c, dp));
308                }
309            }
310            dp.push(next);
311        }
312        dp.pop().unwrap()
313    }

357        where

359            $($($clauses)*)?
360        {

363            }
364        }

366        where
367            R: SemiRing,
368            $($($clauses)*)?
369        {
370            fn $method_assign(&mut self, rhs: &Self) {
371                self.pairwise_assign(rhs, |a, b| R::$method_assign(a, b));
372            }
373        }
374        impl<R> $imp for Matrix<R>
375        where
376            R: SemiRing,
377            $($($clauses)*)?
378        {
379            type Output = Matrix<R>;
380            fn $method(mut self, rhs: Self) -> Self::Output {
381                self.$method_assign(rhs);
382                self
383            }
384        }
385        impl<R> $imp<&Matrix<R>> for Matrix<R>
386        where
387            R: SemiRing,
388            $($($clauses)*)?
389        {
390            type Output = Matrix<R>;
391            fn $method(mut self, rhs: &Self) -> Self::Output {
392                self.$method_assign(rhs);
393                self

395        }

398            R: SemiRing,

400        {
401            type Output = Matrix<R>;
402            fn $method(self, mut rhs: Matrix<R>) -> Self::Output {
403                rhs.pairwise_assign(self, |a, b| *a = R::$method(b, a));

406        }

453where

457    fn mul(self, rhs: &Matrix<R>) -> Self::Output {
458        assert_eq!(self.shape.1, rhs.shape.0);
459        let mut res = Matrix::zeros((self.shape.0, rhs.shape.1));
460        for i in 0..self.shape.0 {
461            for k in 0..self.shape.1 {
462                for j in 0..rhs.shape.1 {
463                    R::add_assign(&mut res[i][j], &R::mul(&self[i][k], &rhs[k][j]));
464                }
465            }
466        }
467        res
468    }

484impl<R> Matrix<R>
485where
486    R: SemiRing,
487{
488    pub fn pow(self, mut n: usize) -> Self {
489        assert_eq!(self.shape.0, self.shape.1);
490        let mut res = Matrix::eye(self.shape);
491        let mut x = self;
492        while n > 0 {
493            if n & 1 == 1 {
494                res = &res * &x;
495            }
496            x = &x * &x;
497            n >>= 1;
498        }
499        res
500    }
