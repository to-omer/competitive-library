3unsafe fn normalize_avx2<M>(a: &mut [u32])
4where
5    M: Montgomery32NttModulus,
6{
7    let mod_vec = _mm256_set1_epi32(M::MOD as i32);
8    let mut i = 0;
9    while i + 8 <= a.len() {
10        let x = _mm256_loadu_si256(a.as_ptr().add(i) as *const __m256i);
11        let y = _mm256_min_epu32(x, _mm256_sub_epi32(x, mod_vec));
12        _mm256_storeu_si256(a.as_mut_ptr().add(i) as *mut __m256i, y);
13        i += 8;
14    }
15    while i < a.len() {
16        let x = a[i];
17        a[i] = if x >= M::MOD { x - M::MOD } else { x };
18        i += 1;
19    }
20}

23    a: __m256i,
24    b: __m256i,
25    mod_vec: __m256i,
26    mod2_vec: __m256i,
27) -> __m256i

54unsafe fn mul_vec_avx2<M>(a: __m256i, b: __m256i, r_vec: __m256i, mod_vec: __m256i) -> __m256i
55where
56    M: Montgomery32NttModulus,
57{
58    if M::MOD < LAZY_THRESHOLD {
59        simd32::montgomery_mul_256(a, b, r_vec, mod_vec)

66pub(in super::super) unsafe fn pointwise_multiply_avx2<M>(f: &mut [MInt<M>], g: &[MInt<M>])
67where
68    M: Montgomery32NttModulus,
69{
70    let r_vec = _mm256_set1_epi32(M::R as i32);
71    let mod_vec = _mm256_set1_epi32(M::MOD as i32);
72    let mut i = 0;
73    while i + 8 <= f.len() {
74        let a = _mm256_loadu_si256(f.as_ptr().add(i) as *const __m256i);
75        let b = _mm256_loadu_si256(g.as_ptr().add(i) as *const __m256i);
76        let x = simd32::montgomery_mul_256_canon(a, b, r_vec, mod_vec);
77        _mm256_storeu_si256(f.as_mut_ptr().add(i) as *mut __m256i, x);
78        i += 8;
79    }
80    while i < f.len() {
81        f[i] *= g[i];
82        i += 1;
83    }
84}

86#[target_feature(enable = "avx2")]
87pub(in super::super) unsafe fn ntt_avx2<M>(a: &mut [MInt<M>])
88where
89    M: Montgomery32NttModulus,
90{
91    let n = a.len();
92    if n <= 1 {
93        return;
94    }
95    let ptr = a.as_mut_ptr() as *mut u32;
96    let a = std::slice::from_raw_parts_mut(ptr, n);
97    let mod_vec = _mm256_set1_epi32(M::MOD as i32);
98    let mod2_vec = _mm256_set1_epi32(M::MOD.wrapping_add(M::MOD) as i32);
99    let r_vec = _mm256_set1_epi32(M::R as i32);
100    let imag = M::INFO.root[2];
101    let imag_vec = _mm256_set1_epi32(imag as i32);
102
103    let mut v = n / 2;
104    if n.trailing_zeros() & 1 == 1 {
105        let mut i = 0;
106        while i + 8 <= v {
107            let x0 = _mm256_loadu_si256(a.as_ptr().add(i) as *const __m256i);
108            let x1 = _mm256_loadu_si256(a.as_ptr().add(v + i) as *const __m256i);
109            let y0 = add_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
110            let y1 = sub_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
111            _mm256_storeu_si256(a.as_mut_ptr().add(i) as *mut __m256i, y0);
112            _mm256_storeu_si256(a.as_mut_ptr().add(v + i) as *mut __m256i, y1);
113            i += 8;
114        }
115        while i < v {
116            let x0 = a[i];
117            let x1 = a[v + i];
118            a[i] = M::mod_add(x0, x1);
119            a[v + i] = M::mod_sub(x0, x1);
120            i += 1;
121        }
122        v >>= 1;
123    }
124    while v > 1 {
125        let half = v >> 1;
126        let mut w1 = M::N1;
127        for (s, block) in a.chunks_exact_mut(v << 1).enumerate() {
128            let base = block.as_mut_ptr();
129            let ll = base;
130            let lr = base.add(half);
131            let rl = base.add(v);
132            let rr = base.add(v + half);
133
134            let w2 = M::mod_mul(w1, w1);
135            let w3 = M::mod_mul(w2, w1);
136            let w1v = _mm256_set1_epi32(w1 as i32);
137            let w2v = _mm256_set1_epi32(w2 as i32);
138            let w3v = _mm256_set1_epi32(w3 as i32);
139
140            let mut i = 0;
141            while i + 8 <= half {
142                let x0 = _mm256_loadu_si256(ll.add(i) as *const __m256i);
143                let x1 = _mm256_loadu_si256(lr.add(i) as *const __m256i);
144                let x2 = _mm256_loadu_si256(rl.add(i) as *const __m256i);
145                let x3 = _mm256_loadu_si256(rr.add(i) as *const __m256i);
146
147                let (a1, a2, a3) = if s == 0 {
148                    (x1, x2, x3)
149                } else {
150                    (
151                        mul_vec_avx2::<M>(x1, w1v, r_vec, mod_vec),
152                        mul_vec_avx2::<M>(x2, w2v, r_vec, mod_vec),
153                        mul_vec_avx2::<M>(x3, w3v, r_vec, mod_vec),
154                    )
155                };
156
157                let a0pa2 = add_vec_avx2::<M>(x0, a2, mod_vec, mod2_vec);
158                let a0na2 = sub_vec_avx2::<M>(x0, a2, mod_vec, mod2_vec);
159                let a1pa3 = add_vec_avx2::<M>(a1, a3, mod_vec, mod2_vec);
160                let a1na3 = sub_vec_avx2::<M>(a1, a3, mod_vec, mod2_vec);
161                let a1na3imag = mul_vec_avx2::<M>(a1na3, imag_vec, r_vec, mod_vec);
162
163                let y0 = add_vec_avx2::<M>(a0pa2, a1pa3, mod_vec, mod2_vec);
164                let y1 = sub_vec_avx2::<M>(a0pa2, a1pa3, mod_vec, mod2_vec);
165                let y2 = add_vec_avx2::<M>(a0na2, a1na3imag, mod_vec, mod2_vec);
166                let y3 = sub_vec_avx2::<M>(a0na2, a1na3imag, mod_vec, mod2_vec);
167
168                _mm256_storeu_si256(ll.add(i) as *mut __m256i, y0);
169                _mm256_storeu_si256(lr.add(i) as *mut __m256i, y1);
170                _mm256_storeu_si256(rl.add(i) as *mut __m256i, y2);
171                _mm256_storeu_si256(rr.add(i) as *mut __m256i, y3);
172                i += 8;
173            }
174            while i < half {
175                let a0 = *ll.add(i);
176                let a1 = M::mod_mul(*lr.add(i), w1);
177                let a2 = M::mod_mul(*rl.add(i), w2);
178                let a3 = M::mod_mul(*rr.add(i), w3);
179                let a0pa2 = M::mod_add(a0, a2);
180                let a0na2 = M::mod_sub(a0, a2);
181                let a1pa3 = M::mod_add(a1, a3);
182                let a1na3 = M::mod_sub(a1, a3);
183                let a1na3imag = M::mod_mul(a1na3, imag);
184                *ll.add(i) = M::mod_add(a0pa2, a1pa3);
185                *lr.add(i) = M::mod_sub(a0pa2, a1pa3);
186                *rl.add(i) = M::mod_add(a0na2, a1na3imag);
187                *rr.add(i) = M::mod_sub(a0na2, a1na3imag);
188                i += 1;
189            }
190            w1 = M::mod_mul(w1, M::INFO.rate3[s.trailing_ones() as usize]);
191        }
192        v >>= 2;
193    }
194    normalize_avx2::<M>(a);
195}

197#[target_feature(enable = "avx2")]
198pub(in super::super) unsafe fn intt_avx2<M>(a: &mut [MInt<M>])
199where
200    M: Montgomery32NttModulus,
201{
202    let n = a.len();
203    if n <= 1 {
204        return;
205    }
206    let ptr = a.as_mut_ptr() as *mut u32;
207    let a = std::slice::from_raw_parts_mut(ptr, n);
208    let mod_vec = _mm256_set1_epi32(M::MOD as i32);
209    let mod2_vec = _mm256_set1_epi32(M::MOD.wrapping_add(M::MOD) as i32);
210    let r_vec = _mm256_set1_epi32(M::R as i32);
211    let iimag = M::INFO.inv_root[2];
212    let iimag_vec = _mm256_set1_epi32(iimag as i32);
213
214    let mut v = 1;
215    let limit = if n.trailing_zeros() & 1 == 1 {
216        n / 2
217    } else {
218        n
219    };
220    while v < limit {
221        let mut w1 = M::N1;
222        for (s, block) in a.chunks_exact_mut(v << 2).enumerate() {
223            let base = block.as_mut_ptr();
224            let ll = base;
225            let lr = base.add(v);
226            let rl = base.add(v << 1);
227            let rr = base.add(v * 3);
228
229            let w2 = M::mod_mul(w1, w1);
230            let w3 = M::mod_mul(w2, w1);
231            let w1v = _mm256_set1_epi32(w1 as i32);
232            let w2v = _mm256_set1_epi32(w2 as i32);
233            let w3v = _mm256_set1_epi32(w3 as i32);
234
235            let mut i = 0;
236            while i + 8 <= v {
237                let x0 = _mm256_loadu_si256(ll.add(i) as *const __m256i);
238                let x1 = _mm256_loadu_si256(lr.add(i) as *const __m256i);
239                let x2 = _mm256_loadu_si256(rl.add(i) as *const __m256i);
240                let x3 = _mm256_loadu_si256(rr.add(i) as *const __m256i);
241
242                let a0pa1 = add_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
243                let a0na1 = sub_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
244                let a2pa3 = add_vec_avx2::<M>(x2, x3, mod_vec, mod2_vec);
245                let a2na3 = sub_vec_avx2::<M>(x2, x3, mod_vec, mod2_vec);
246                let a2na3iimag = mul_vec_avx2::<M>(a2na3, iimag_vec, r_vec, mod_vec);
247
248                let y0 = add_vec_avx2::<M>(a0pa1, a2pa3, mod_vec, mod2_vec);
249                let y1 = add_vec_avx2::<M>(a0na1, a2na3iimag, mod_vec, mod2_vec);
250                let y2 = sub_vec_avx2::<M>(a0pa1, a2pa3, mod_vec, mod2_vec);
251                let y3 = sub_vec_avx2::<M>(a0na1, a2na3iimag, mod_vec, mod2_vec);
252
253                let (y1, y2, y3) = if s == 0 {
254                    (y1, y2, y3)
255                } else {
256                    (
257                        mul_vec_avx2::<M>(y1, w1v, r_vec, mod_vec),
258                        mul_vec_avx2::<M>(y2, w2v, r_vec, mod_vec),
259                        mul_vec_avx2::<M>(y3, w3v, r_vec, mod_vec),
260                    )
261                };
262
263                _mm256_storeu_si256(ll.add(i) as *mut __m256i, y0);
264                _mm256_storeu_si256(lr.add(i) as *mut __m256i, y1);
265                _mm256_storeu_si256(rl.add(i) as *mut __m256i, y2);
266                _mm256_storeu_si256(rr.add(i) as *mut __m256i, y3);
267                i += 8;
268            }
269            while i < v {
270                let a0 = *ll.add(i);
271                let a1 = *lr.add(i);
272                let a2 = *rl.add(i);
273                let a3 = *rr.add(i);
274                let a0pa1 = M::mod_add(a0, a1);
275                let a0na1 = M::mod_sub(a0, a1);
276                let a2pa3 = M::mod_add(a2, a3);
277                let a2na3iimag = M::mod_mul(M::mod_sub(a2, a3), iimag);
278                *ll.add(i) = M::mod_add(a0pa1, a2pa3);
279                *lr.add(i) = M::mod_mul(M::mod_add(a0na1, a2na3iimag), w1);
280                *rl.add(i) = M::mod_mul(M::mod_sub(a0pa1, a2pa3), w2);
281                *rr.add(i) = M::mod_mul(M::mod_sub(a0na1, a2na3iimag), w3);
282                i += 1;
283            }
284            w1 = M::mod_mul(w1, M::INFO.inv_rate3[s.trailing_ones() as usize]);
285        }
286        v <<= 2;
287    }
288    if n.trailing_zeros() & 1 == 1 {
289        let half = n >> 1;
290        let mut i = 0;
291        while i + 8 <= half {
292            let x0 = _mm256_loadu_si256(a.as_ptr().add(i) as *const __m256i);
293            let x1 = _mm256_loadu_si256(a.as_ptr().add(half + i) as *const __m256i);
294            let y0 = add_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
295            let y1 = sub_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
296            _mm256_storeu_si256(a.as_mut_ptr().add(i) as *mut __m256i, y0);
297            _mm256_storeu_si256(a.as_mut_ptr().add(half + i) as *mut __m256i, y1);
298            i += 8;
299        }
300        while i < half {
301            let x0 = a[i];
302            let x1 = a[half + i];
303            a[i] = M::mod_add(x0, x1);
304            a[half + i] = M::mod_sub(x0, x1);
305            i += 1;
306        }
307    }
308    let inv = M::mod_inv(<M as MIntConvert<u32>>::from(n as u32));
309    let inv_vec = _mm256_set1_epi32(inv as i32);
310    let mut i = 0;
311    while i + 8 <= n {
312        let x = _mm256_loadu_si256(a.as_ptr().add(i) as *const __m256i);
313        let y = simd32::montgomery_mul_256_canon(x, inv_vec, r_vec, mod_vec);
314        _mm256_storeu_si256(a.as_mut_ptr().add(i) as *mut __m256i, y);
315        i += 8;
316    }
317    while i < n {
318        a[i] = M::mod_mul(a[i], inv);
319        i += 1;
320    }
