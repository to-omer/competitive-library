4fn choose2(n: Wrapping<u64>) -> Wrapping<u64> {
5    if n.0 % 2 == 0 {
6        n / Wrapping(2) * (n - Wrapping(1))
7    } else {
8        (n - Wrapping(1)) / Wrapping(2) * n
9    }
10}

12/// Sum of Floor of Linear mod 2^64
13///
14/// $$\sum_{i=0}^{n-1}\left\lfloor\frac{a\times i+b}{m}\right\rfloor$$
15pub fn floor_sum(n: u64, a: u64, b: u64, m: u64) -> u64 {
16    let mut ans = Wrapping(0u64);
17    let (mut n, mut m, mut a, mut b) = (Wrapping(n), Wrapping(m), Wrapping(a), Wrapping(b));
18    loop {
19        let br = BarrettReduction::<u64>::new(m.0);
20        if a >= m {
21            let (q, r) = br.div_rem(a.0);
22            ans += choose2(n) * Wrapping(q);
23            a = Wrapping(r);
24        }
25        if b >= m {
26            let (q, r) = br.div_rem(b.0);
27            ans += n * Wrapping(q);
28            b = Wrapping(r);
29        }
30        let y_max = a * n + b;
31        if y_max < m {
32            break;
33        }
34        let (q, r) = br.div_rem(y_max.0);
35        n = Wrapping(q);
36        b = Wrapping(r);
37        swap(&mut m, &mut a);
38    }
39    ans.0
40}

42/// Sum of Floor of Linear mod 2^64
43///
44/// $$\sum_{i=l}^{r-1}\left\lfloor\frac{a\times i+b}{m}\right\rfloor$$
45pub fn floor_sum_i64(l: i64, r: i64, a: i64, b: i64, m: u64) -> i64 {
46    let mut ans = Wrapping(0i64);
47    let (n, m, a, b) = (
48        Wrapping((r - l) as u64),
49        Wrapping(m as i64),
50        Wrapping(a),
51        Wrapping(b) + Wrapping(l) * Wrapping(a),
52    );
53    let a = if a.0 < 0 {
54        let r = a.0.rem_euclid(m.0);
55        let nc2 = choose2(n);
56        ans -= Wrapping(nc2.0 as _) * ((Wrapping(r) - a) / m);
57        r
58    } else {
59        a.0
60    };
61    let b = if b.0 < 0 {
62        let r = b.0.rem_euclid(m.0);
63        ans -= Wrapping(n.0 as _) * ((Wrapping(r) - b) / m);
64        r
65    } else {
66        b.0
67    };
68    (ans + Wrapping(floor_sum(n.0, a as u64, b as u64, m.0 as u64) as i64)).0
69}

71pub fn floor_sum_range_freq(l: i64, r: i64, a: i64, b: i64, m: u64, range: Range<i64>) -> i64 {
72    if range.start >= range.end {
73        return 0;
74    }
75    assert!(0 <= range.start && range.end <= m as i64);
76    let x1 = floor_sum_i64(l, r, a, b - range.start, m);
77    let x2 = floor_sum_i64(l, r, a, b - range.end, m);
78    x1 - x2
79}

81#[test]
82fn test_floor_sum() {
83    use crate::tools::Xorshift;
84    const A: u64 = 1_000;
85    const B: i64 = 1_000;
86    const Q: usize = 1_000;
87    let mut rng = Xorshift::new();
88    for _ in 0..Q {
89        let (n, a, b, m) = rng.random((..A, ..A, ..A, 1..A));
90        let expected: u64 = (0..n).map(|i| (a * i + b) / m).sum();
91        let result = floor_sum(n, a, b, m);
92        assert_eq!(expected, result);
93
94        let (mut l, mut r, a, b) = rng.random((-B..B, -B..B, -B..B, -B..B));
95        if l > r {
96            swap(&mut l, &mut r);
97        }
98        let expected: i64 = (l..r).map(|i| (a * i + b).div_euclid(m as i64)).sum();
99        let result = floor_sum_i64(l, r, a, b, m);
100        assert_eq!(expected, result);
101
102        let (mut lv, mut rv) = rng.random((0..m as i64, 0..m as i64));
103        if lv > rv {
104            swap(&mut lv, &mut rv);
105        }
106        let range = lv..rv + 1;
107        let expected = (l..r)
108            .filter(|&i| range.contains(&(a * i + b).rem_euclid(m as i64)))
109            .count() as i64;
110        let result = floor_sum_range_freq(l, r, a, b, m, range);
111        assert_eq!(expected, result);
112    }
