9#[target_feature(enable = "avx2,fma")]
10#[inline]
11unsafe fn round4(value: &[f64; 4]) -> [i64; 4] {
12    let magic = _mm256_set1_pd((3i64 << 51) as f64);
13    let rounded = _mm256_sub_epi64(
14        _mm256_castpd_si256(_mm256_add_pd(_mm256_load_pd(value.as_ptr()), magic)),
15        _mm256_castpd_si256(magic),
16    );
17    let mut result = [0; 4];
18    _mm256_storeu_si256(result.as_mut_ptr().cast(), rounded);
19    result
20}

22#[target_feature(enable = "avx2,fma")]
23unsafe fn split_coefficients<M>(
24    values: Vec<MInt<M>>,
25    n: usize,
26    modulus: i64,
27    split: i64,
28) -> (Vec<Complex4>, Vec<Complex4>)
29where
30    M: MIntConvert + MIntConvert<u32>,
31{
32    let mut low = vec![Complex4::default(); n / 4];
33    let mut high = low.clone();
34    let divisor = _mm256_set1_pd(split as f64);
35    let split = _mm256_set1_pd(split as f64);
36    for i in (0..values.len()).step_by(4) {
37        let mut centered = [0i32; 4];
38        for lane in 0..4.min(values.len() - i) {
39            let mut value = <M as MIntConvert<u32>>::into(values[i + lane].inner()) as i64;
40            if value * 2 > modulus {
41                value -= modulus;
42            }
43            centered[lane] = value as i32;
44        }
45        let value = _mm256_cvtepi32_pd(_mm_loadu_si128(centered.as_ptr().cast()));
46        let upper = _mm256_round_pd::<{ _MM_FROUND_TO_NEAREST_INT | _MM_FROUND_NO_EXC }>(
47            _mm256_div_pd(value, divisor),
48        );
49        let lower = _mm256_fnmadd_pd(upper, split, value);
50        if i < n {
51            _mm256_store_pd(low[i >> 2].re.as_mut_ptr(), lower);
52            _mm256_store_pd(high[i >> 2].re.as_mut_ptr(), upper);
53        } else {
54            _mm256_store_pd(low[(i - n) >> 2].im.as_mut_ptr(), lower);
55            _mm256_store_pd(high[(i - n) >> 2].im.as_mut_ptr(), upper);
56        }
57    }
58    (low, high)
59}

61#[target_feature(enable = "avx2,fma")]
62unsafe fn dot_soa(a0: &mut [Complex4], a1: &mut [Complex4], b0: &mut [Complex4], b1: &[Complex4]) {
63    let n = a0.len() * 4;
64    RotateCache::ensure(n / 2);
65    RotateCache::with(|cache| {
66        for i in 0..a0.len() {
67            let (mut cr, mut ci) = load4(&b0[i]);
68            let (mut dr, mut di) = load4(&b1[i]);
69            let mut c0r = _mm256_setzero_pd();
70            let mut c0i = _mm256_setzero_pd();
71            let mut c1r = _mm256_setzero_pd();
72            let mut c1i = _mm256_setzero_pd();
73            let mut c2r = _mm256_setzero_pd();
74            let mut c2i = _mm256_setzero_pd();
75            let w = eval_twiddle(cache, 1, a0.len(), i);
76            let wr = _mm256_setr_pd(w.re, 1.0, 1.0, 1.0);
77            let wi = _mm256_setr_pd(w.im, 0.0, 0.0, 0.0);
78            for lane in 0..4 {
79                let ar = _mm256_set1_pd(a0[i].re[lane]);
80                let ai = _mm256_set1_pd(a0[i].im[lane]);
81                let br = _mm256_set1_pd(a1[i].re[lane]);
82                let bi = _mm256_set1_pd(a1[i].im[lane]);
83                multiply_accumulate4(&mut c0r, &mut c0i, ar, ai, cr, ci);
84                multiply_accumulate4(&mut c1r, &mut c1i, ar, ai, dr, di);
85                multiply_accumulate4(&mut c1r, &mut c1i, br, bi, cr, ci);
86                multiply_accumulate4(&mut c2r, &mut c2i, br, bi, dr, di);
87                if lane != 3 {
88                    cr = _mm256_permute4x64_pd::<0x93>(cr);
89                    ci = _mm256_permute4x64_pd::<0x93>(ci);
90                    dr = _mm256_permute4x64_pd::<0x93>(dr);
91                    di = _mm256_permute4x64_pd::<0x93>(di);
92                    (cr, ci) = mul4(cr, ci, wr, wi);
93                    (dr, di) = mul4(dr, di, wr, wi);
94                }
95            }
96            store4(&mut a0[i], c0r, c0i);
97            store4(&mut a1[i], c1r, c1i);
98            store4(&mut b0[i], c2r, c2i);
99        }
100    });
101}

103#[target_feature(enable = "avx2,fma")]
104unsafe fn split_u64_coefficients(values: &[u64], n: usize, factor: u64) -> [Vec<Complex4>; 4] {
105    let mut result = std::array::from_fn(|_| vec![Complex4::default(); n / 4]);
106    let mut multiplier = 1u64;
107    for (i, &value) in values.iter().enumerate() {
108        let mut value = value.wrapping_mul(multiplier);
109        let (i, imag) = if i < n { (i, false) } else { (i - n, true) };
110        for part in &mut result {
111            let digit = value as i16;
112            value = (value >> 16).wrapping_add(u64::from(digit < 0));
113            if imag {
114                part[i >> 2].im[i & 3] = digit as f64;
115            } else {
116                part[i >> 2].re[i & 3] = digit as f64;
117            }
118        }
119        multiplier = multiplier.wrapping_mul(factor);
120    }
121    result
122}

124#[target_feature(enable = "avx2,fma")]
125unsafe fn dot_u64_soa(a: &mut [Vec<Complex4>; 4], b: &[Vec<Complex4>; 4]) {
126    let n = a[0].len() * 4;
127    RotateCache::ensure(n / 2);
128    RotateCache::with(|cache| {
129        for block in 0..a[0].len() {
130            let mut br = [_mm256_setzero_pd(); 4];
131            let mut bi = br;
132            let mut rr = br;
133            let mut ri = br;
134            for part in 0..4 {
135                (br[part], bi[part]) = load4(&b[part][block]);
136            }
137            let w = eval_twiddle(cache, 1, a[0].len(), block);
138            let wr = _mm256_setr_pd(w.re, 1.0, 1.0, 1.0);
139            let wi = _mm256_setr_pd(w.im, 0.0, 0.0, 0.0);
140            for lane in 0..4 {
141                let ar: [__m256d; 4] =
142                    std::array::from_fn(|part| _mm256_set1_pd(a[part][block].re[lane]));
143                let ai: [__m256d; 4] =
144                    std::array::from_fn(|part| _mm256_set1_pd(a[part][block].im[lane]));
145                for part in 0..4 {
146                    for left in 0..=part {
147                        multiply_accumulate4(
148                            &mut rr[part],
149                            &mut ri[part],
150                            ar[left],
151                            ai[left],
152                            br[part - left],
153                            bi[part - left],
154                        );
155                    }
156                }
157                if lane != 3 {
158                    for part in 0..4 {
159                        br[part] = _mm256_permute4x64_pd::<0x93>(br[part]);
160                        bi[part] = _mm256_permute4x64_pd::<0x93>(bi[part]);
161                        (br[part], bi[part]) = mul4(br[part], bi[part], wr, wi);
162                    }
163                }
164            }
165            for part in 0..4 {
166                store4(&mut a[part][block], rr[part], ri[part]);
167            }
168        }
169    });
170}

172#[target_feature(enable = "avx2,fma")]
173pub unsafe fn convolve_u64_avx2(a: Vec<u64>, b: Vec<u64>, factor: u64, inverse: u64) -> Vec<u64> {
174    let len = a.len() + b.len() - 1;
175    let n = len.next_power_of_two() / 2;
176    let mut fa = split_u64_coefficients(&a, n, factor);
177    drop(a);
178    let mut fb = split_u64_coefficients(&b, n, factor);
179    drop(b);
180    for part in 0..4 {
181        fft_soa(&mut fa[part]);
182        fft_soa(&mut fb[part]);
183    }
184    dot_u64_soa(&mut fa, &fb);
185    drop(fb);
186    for part in &mut fa {
187        ifft_soa(part);
188    }
189    let mut result = vec![0; len];
190    let mut real_multiplier = 1u64;
191    let mut imag_multiplier = inverse.wrapping_pow(n as u32);
192    for (block, _) in fa[0].iter().enumerate() {
193        let real: [[i64; 4]; 4] = std::array::from_fn(|part| round4(&fa[part][block].re));
194        let imag: [[i64; 4]; 4] = std::array::from_fn(|part| round4(&fa[part][block].im));
195        for lane in 0..4 {
196            let i = block * 4 + lane;
197            if i < len {
198                result[i] = (real[0][lane] as u64)
199                    .wrapping_add((real[1][lane] as u64) << 16)
200                    .wrapping_add((real[2][lane] as u64) << 32)
201                    .wrapping_add((real[3][lane] as u64) << 48)
202                    .wrapping_mul(real_multiplier);
203            }
204            real_multiplier = real_multiplier.wrapping_mul(inverse);
205            if i + n < len {
206                result[i + n] = (imag[0][lane] as u64)
207                    .wrapping_add((imag[1][lane] as u64) << 16)
208                    .wrapping_add((imag[2][lane] as u64) << 32)
209                    .wrapping_add((imag[3][lane] as u64) << 48)
210                    .wrapping_mul(imag_multiplier);
211            }
212            imag_multiplier = imag_multiplier.wrapping_mul(inverse);
213        }
214    }
215    result
216}

218#[target_feature(enable = "avx2,fma")]
219pub unsafe fn convolve_mint_avx2<M>(a: Vec<MInt<M>>, b: Vec<MInt<M>>) -> Vec<MInt<M>>
220where
221    M: MIntConvert + MIntConvert<u32>,
222{
223    let len = a.len() + b.len() - 1;
224    let n = len.next_power_of_two() / 2;
225    let modulus = <M as MIntConvert<u32>>::mod_into() as i64;
226    let split = (modulus as f64).sqrt() as i64 + 1;
227    let (mut a0, mut a1) = split_coefficients(a, n, modulus, split);
228    let (mut b0, mut b1) = split_coefficients(b, n, modulus, split);
229    fft_soa(&mut a0);
230    fft_soa(&mut a1);
231    fft_soa(&mut b0);
232    fft_soa(&mut b1);
233    dot_soa(&mut a0, &mut a1, &mut b0, &b1);
234    drop(b1);
235    ifft_soa(&mut a0);
236    ifft_soa(&mut a1);
237    ifft_soa(&mut b0);
238    let modulus = modulus as u64;
239    let split = split as u64;
240    let split2 = split * split % modulus;
241    let mut result = vec![MInt::<M>::from(0u32); len];
242    for (block, ((a0, a1), b0)) in a0.iter().zip(&a1).zip(&b0).enumerate() {
243        let values = [
244            [round4(&a0.re), round4(&a1.re), round4(&b0.re)],
245            [round4(&a0.im), round4(&a1.im), round4(&b0.im)],
246        ];
247        for (part, values) in values.into_iter().enumerate() {
248            for (lane, value0) in values[0].into_iter().enumerate() {
249                let i = block * 4 + lane + part * n;
250                if i < len {
251                    let value = (value0.rem_euclid(modulus as i64) as u64
252                        + values[1][lane].rem_euclid(modulus as i64) as u64 * split
253                        + values[2][lane].rem_euclid(modulus as i64) as u64 * split2)
254                        % modulus;
255                    result[i] = MInt::<M>::from(value as u32);
256                }
257            }
258        }
259    }
260    result
