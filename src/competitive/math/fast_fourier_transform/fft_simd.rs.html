7struct Complex4 {

12impl Complex4 {
13    #[inline]
14    fn get(&self, lane: usize) -> Complex<f64> {
15        Complex::new(self.re[lane], self.im[lane])
16    }

18    #[inline]
19    fn set(&mut self, lane: usize, value: Complex<f64>) {

21        self.im[lane] = value.im;
22    }

41#[target_feature(enable = "avx2,fma")]
42#[inline]
43unsafe fn mul4(ar: __m256d, ai: __m256d, br: __m256d, bi: __m256d) -> (__m256d, __m256d) {
44    (
45        _mm256_fmsub_pd(ar, br, _mm256_mul_pd(ai, bi)),
46        _mm256_fmadd_pd(ai, br, _mm256_mul_pd(ar, bi)),
47    )

50#[target_feature(enable = "avx2,fma")]
51#[inline]
52unsafe fn round4(value: &[f64; 4], scale: __m256d) -> [i64; 4] {
53    let magic = _mm256_set1_pd((3i64 << 51) as f64);
54    let rounded = _mm256_sub_epi64(
55        _mm256_castpd_si256(_mm256_add_pd(
56            _mm256_mul_pd(_mm256_load_pd(value.as_ptr()), scale),
57            magic,
58        )),
59        _mm256_castpd_si256(magic),
60    );
61    let mut result = [0; 4];
62    _mm256_storeu_si256(result.as_mut_ptr().cast(), rounded);
63    result
64}

66#[target_feature(enable = "avx2,fma")]
67unsafe fn fft_soa(a: &mut [Complex4]) {
68    let n = a.len() * 4;
69    RotateCache::ensure(n / 2);
70    RotateCache::with(|cache| {
71        let mut v = n / 2;
72        while v >= 8 {
73            let l = v / 2;
74            let block_len = l / 4;
75            for (q, block) in a.chunks_exact_mut(block_len * 4).enumerate() {
76                let (a, rest) = block.split_at_mut(block_len);
77                let (b, rest) = rest.split_at_mut(block_len);
78                let (c, d) = rest.split_at_mut(block_len);
79                let w0 = cache[q];
80                let w1 = cache[q << 1];
81                let w2 = cache[q << 1 | 1];
82                let w0r = _mm256_set1_pd(w0.re);
83                let w0i = _mm256_set1_pd(w0.im);
84                let w1r = _mm256_set1_pd(w1.re);
85                let w1i = _mm256_set1_pd(w1.im);
86                let w2r = _mm256_set1_pd(w2.re);
87                let w2i = _mm256_set1_pd(w2.im);
88                for i in 0..block_len {
89                    let (ar, ai) = load4(&a[i]);
90                    let (br, bi) = load4(&b[i]);
91                    let (cr, ci) = load4(&c[i]);
92                    let (dr, di) = load4(&d[i]);
93                    let (cr, ci) = mul4(cr, ci, w0r, w0i);
94                    let (dr, di) = mul4(dr, di, w0r, w0i);
95                    let ac0r = _mm256_add_pd(ar, cr);
96                    let ac0i = _mm256_add_pd(ai, ci);
97                    let ac1r = _mm256_sub_pd(ar, cr);
98                    let ac1i = _mm256_sub_pd(ai, ci);
99                    let (bd0r, bd0i) = mul4(_mm256_add_pd(br, dr), _mm256_add_pd(bi, di), w1r, w1i);
100                    let (bd1r, bd1i) = mul4(_mm256_sub_pd(br, dr), _mm256_sub_pd(bi, di), w2r, w2i);
101                    store4(
102                        &mut a[i],
103                        _mm256_add_pd(ac0r, bd0r),
104                        _mm256_add_pd(ac0i, bd0i),
105                    );
106                    store4(
107                        &mut b[i],
108                        _mm256_sub_pd(ac0r, bd0r),
109                        _mm256_sub_pd(ac0i, bd0i),
110                    );
111                    store4(
112                        &mut c[i],
113                        _mm256_add_pd(ac1r, bd1r),
114                        _mm256_add_pd(ac1i, bd1i),
115                    );
116                    store4(
117                        &mut d[i],
118                        _mm256_sub_pd(ac1r, bd1r),
119                        _mm256_sub_pd(ac1i, bd1i),
120                    );
121                }
122            }
123            v >>= 2;
124        }
125        if v == 4 {
126            for (block, w) in a.chunks_exact_mut(2).zip(cache) {
127                let (ar, ai) = load4(&block[0]);
128                let (br, bi) = load4(&block[1]);
129                let (br, bi) = mul4(br, bi, _mm256_set1_pd(w.re), _mm256_set1_pd(w.im));
130                store4(&mut block[0], _mm256_add_pd(ar, br), _mm256_add_pd(ai, bi));
131                store4(&mut block[1], _mm256_sub_pd(ar, br), _mm256_sub_pd(ai, bi));
132            }
133            v = 2;
134        }
135        if v == 2 {
136            for (value, w) in a.iter_mut().zip(cache) {
137                let (re, im) = load4(value);
138                let yr = _mm256_permute2f128_pd::<0x01>(re, re);
139                let yi = _mm256_permute2f128_pd::<0x01>(im, im);
140                let (yr, yi) = mul4(yr, yi, _mm256_set1_pd(w.re), _mm256_set1_pd(w.im));
141                let dr =
142                    _mm256_permute2f128_pd::<0x01>(_mm256_sub_pd(re, yr), _mm256_sub_pd(re, yr));
143                let di =
144                    _mm256_permute2f128_pd::<0x01>(_mm256_sub_pd(im, yi), _mm256_sub_pd(im, yi));
145                store4(
146                    value,
147                    _mm256_blend_pd::<0b1100>(_mm256_add_pd(re, yr), dr),
148                    _mm256_blend_pd::<0b1100>(_mm256_add_pd(im, yi), di),
149                );
150            }
151        }
152        for (q, value) in a.iter_mut().enumerate() {
153            let (re, im) = load4(value);
154            let yr = _mm256_permute_pd::<0b0101>(re);
155            let yi = _mm256_permute_pd::<0b0101>(im);
156            let w0 = cache[q << 1];
157            let w1 = cache[q << 1 | 1];
158            let wr = _mm256_setr_pd(w0.re, w0.re, w1.re, w1.re);
159            let wi = _mm256_setr_pd(w0.im, w0.im, w1.im, w1.im);
160            let (yr, yi) = mul4(yr, yi, wr, wi);
161            let dr = _mm256_permute_pd::<0b0101>(_mm256_sub_pd(re, yr));
162            let di = _mm256_permute_pd::<0b0101>(_mm256_sub_pd(im, yi));
163            store4(
164                value,
165                _mm256_blend_pd::<0b1010>(_mm256_add_pd(re, yr), dr),
166                _mm256_blend_pd::<0b1010>(_mm256_add_pd(im, yi), di),
167            );
168        }
169    });
170}

172#[target_feature(enable = "avx2,fma")]
173unsafe fn ifft_soa(a: &mut [Complex4]) {
174    let n = a.len() * 4;
175    RotateCache::ensure(n / 2);
176    RotateCache::with(|cache| {
177        for (q, value) in a.iter_mut().enumerate() {
178            let (re, im) = load4(value);
179            let yr = _mm256_permute_pd::<0b0101>(re);
180            let yi = _mm256_permute_pd::<0b0101>(im);
181            let w0 = cache[q << 1].conjugate();
182            let w1 = cache[q << 1 | 1].conjugate();
183            let wr = _mm256_setr_pd(w0.re, w0.re, w1.re, w1.re);
184            let wi = _mm256_setr_pd(w0.im, w0.im, w1.im, w1.im);
185            let (dr, di) = mul4(_mm256_sub_pd(re, yr), _mm256_sub_pd(im, yi), wr, wi);
186            store4(
187                value,
188                _mm256_blend_pd::<0b1010>(_mm256_add_pd(re, yr), _mm256_permute_pd::<0b0101>(dr)),
189                _mm256_blend_pd::<0b1010>(_mm256_add_pd(im, yi), _mm256_permute_pd::<0b0101>(di)),
190            );
191        }
192        for (value, w) in a.iter_mut().zip(cache) {
193            let (re, im) = load4(value);
194            let yr = _mm256_permute2f128_pd::<0x01>(re, re);
195            let yi = _mm256_permute2f128_pd::<0x01>(im, im);
196            let w = w.conjugate();
197            let (dr, di) = mul4(
198                _mm256_sub_pd(re, yr),
199                _mm256_sub_pd(im, yi),
200                _mm256_set1_pd(w.re),
201                _mm256_set1_pd(w.im),
202            );
203            store4(
204                value,
205                _mm256_blend_pd::<0b1100>(
206                    _mm256_add_pd(re, yr),
207                    _mm256_permute2f128_pd::<0x01>(dr, dr),
208                ),
209                _mm256_blend_pd::<0b1100>(
210                    _mm256_add_pd(im, yi),
211                    _mm256_permute2f128_pd::<0x01>(di, di),
212                ),
213            );
214        }
215        let mut v = 4;
216        while v << 1 < n {
217            let block_len = v / 4;
218            for (q, block) in a.chunks_exact_mut(block_len * 4).enumerate() {
219                let (a, rest) = block.split_at_mut(block_len);
220                let (b, rest) = rest.split_at_mut(block_len);
221                let (c, d) = rest.split_at_mut(block_len);
222                let w0 = cache[q].conjugate();
223                let w1 = cache[q << 1].conjugate();
224                let w2 = cache[q << 1 | 1].conjugate();
225                let w0r = _mm256_set1_pd(w0.re);
226                let w0i = _mm256_set1_pd(w0.im);
227                let w1r = _mm256_set1_pd(w1.re);
228                let w1i = _mm256_set1_pd(w1.im);
229                let w2r = _mm256_set1_pd(w2.re);
230                let w2i = _mm256_set1_pd(w2.im);
231                for i in 0..block_len {
232                    let (ar, ai) = load4(&a[i]);
233                    let (br, bi) = load4(&b[i]);
234                    let (cr, ci) = load4(&c[i]);
235                    let (dr, di) = load4(&d[i]);
236                    let ab0r = _mm256_add_pd(ar, br);
237                    let ab0i = _mm256_add_pd(ai, bi);
238                    let (ab1r, ab1i) = mul4(_mm256_sub_pd(ar, br), _mm256_sub_pd(ai, bi), w1r, w1i);
239                    let cd0r = _mm256_add_pd(cr, dr);
240                    let cd0i = _mm256_add_pd(ci, di);
241                    let (cd1r, cd1i) = mul4(_mm256_sub_pd(cr, dr), _mm256_sub_pd(ci, di), w2r, w2i);
242                    store4(
243                        &mut a[i],
244                        _mm256_add_pd(ab0r, cd0r),
245                        _mm256_add_pd(ab0i, cd0i),
246                    );
247                    store4(
248                        &mut b[i],
249                        _mm256_add_pd(ab1r, cd1r),
250                        _mm256_add_pd(ab1i, cd1i),
251                    );
252                    let (cr, ci) = mul4(
253                        _mm256_sub_pd(ab0r, cd0r),
254                        _mm256_sub_pd(ab0i, cd0i),
255                        w0r,
256                        w0i,
257                    );
258                    store4(&mut c[i], cr, ci);
259                    let (dr, di) = mul4(
260                        _mm256_sub_pd(ab1r, cd1r),
261                        _mm256_sub_pd(ab1i, cd1i),
262                        w0r,
263                        w0i,
264                    );
265                    store4(&mut d[i], dr, di);
266                }
267            }
268            v <<= 2;
269        }
270        if v < n {
271            let block_len = v / 4;
272            for (block, w) in a.chunks_exact_mut(block_len * 2).zip(cache) {
273                let (l, r) = block.split_at_mut(block_len);
274                let w = w.conjugate();
275                let wr = _mm256_set1_pd(w.re);
276                let wi = _mm256_set1_pd(w.im);
277                for i in 0..block_len {
278                    let (lr, li) = load4(&l[i]);
279                    let (rr, ri) = load4(&r[i]);
280                    store4(&mut l[i], _mm256_add_pd(lr, rr), _mm256_add_pd(li, ri));
281                    let (rr, ri) = mul4(_mm256_sub_pd(lr, rr), _mm256_sub_pd(li, ri), wr, wi);
282                    store4(&mut r[i], rr, ri);
283                }
284            }
285        }
286    });
287}

289#[target_feature(enable = "avx2,fma")]
290unsafe fn transpose4(a: __m256d, b: __m256d, c: __m256d, d: __m256d) -> [__m256d; 4] {
291    let ab0 = _mm256_unpacklo_pd(a, b);
292    let ab1 = _mm256_unpackhi_pd(a, b);
293    let cd0 = _mm256_unpacklo_pd(c, d);
294    let cd1 = _mm256_unpackhi_pd(c, d);
295    [
296        _mm256_permute2f128_pd::<0x20>(ab0, cd0),
297        _mm256_permute2f128_pd::<0x20>(ab1, cd1),
298        _mm256_permute2f128_pd::<0x31>(ab0, cd0),
299        _mm256_permute2f128_pd::<0x31>(ab1, cd1),
300    ]
301}

303#[target_feature(enable = "avx2,fma")]
304unsafe fn bit_reverse_soa(a: &mut Vec<Complex4>) {
305    let quarter = a.len() / 4;
306    let shift = usize::BITS - quarter.trailing_zeros();
307    let mut result = vec![Complex4::default(); a.len()];
308    for i in 0..quarter {
309        let j = i.reverse_bits() >> shift;
310        let (r0, i0) = load4(&a[j]);
311        let (r1, i1) = load4(&a[j + quarter * 2]);
312        let (r2, i2) = load4(&a[j + quarter]);
313        let (r3, i3) = load4(&a[j + quarter * 3]);
314        let re = transpose4(r0, r1, r2, r3);
315        let im = transpose4(i0, i1, i2, i3);
316        store4(&mut result[i], re[0], im[0]);
317        store4(&mut result[i + quarter], re[2], im[2]);
318        store4(&mut result[i + quarter * 2], re[1], im[1]);
319        store4(&mut result[i + quarter * 3], re[3], im[3]);
320    }
321    *a = result;
322}

324#[target_feature(enable = "avx2,fma")]
325unsafe fn real_fft_soa(a: &mut Vec<Complex4>) {
326    let n = a.len() * 8;
327    fft_soa(a);
328    bit_reverse_soa(a);
329    let z = a[0].get(0);
330    a[0].set(0, Complex::new(z.re + z.im, z.re - z.im));
331    let middle = n / 4;
332    let z = a[middle >> 2].get(middle & 3).conjugate();
333    a[middle >> 2].set(middle & 3, z);
334    RotateCache::ensure(n / 2);
335    RotateCache::with(|cache| {
336        let shift = usize::BITS - (n / 2).trailing_zeros();
337        for k in 1..n / 4 {
338            let j = n / 2 - k;
339            let x = a[k >> 2].get(k & 3);
340            let y = a[j >> 2].get(j & 3);
341            let w = cache[k.reverse_bits() >> shift];
342            let c = w.conjugate().transpose() + 1.0;
343            let d = c * (x - y.conjugate()) * 0.5;
344            a[k >> 2].set(k & 3, x - d);
345            a[j >> 2].set(j & 3, y + d.conjugate());
346        }
347    });
348}

350#[target_feature(enable = "avx2,fma")]
351unsafe fn real_ifft_soa(a: &mut Vec<Complex4>) {
352    let n = a.len() * 8;
353    let z = a[0].get(0);
354    a[0].set(0, Complex::new((z.re + z.im) * 0.5, (z.re - z.im) * 0.5));
355    let middle = n / 4;
356    let z = a[middle >> 2].get(middle & 3).conjugate();
357    a[middle >> 2].set(middle & 3, z);
358    RotateCache::ensure(n / 2);
359    RotateCache::with(|cache| {
360        let shift = usize::BITS - (n / 2).trailing_zeros();
361        for k in 1..n / 4 {
362            let j = n / 2 - k;
363            let x = a[k >> 2].get(k & 3);
364            let y = a[j >> 2].get(j & 3);
365            let w = cache[k.reverse_bits() >> shift].conjugate();
366            let c = w.transpose().conjugate() + 1.0;
367            let d = c * (x - y.conjugate()) * 0.5;
368            a[k >> 2].set(k & 3, x - d);
369            a[j >> 2].set(j & 3, y + d.conjugate());
370        }
371    });
372    bit_reverse_soa(a);
373    ifft_soa(a);
374}

376fn split_coefficients<M>(
377    values: Vec<MInt<M>>,
378    blocks: usize,
379    modulus: i64,
380    split: i64,
381) -> (Vec<Complex4>, Vec<Complex4>)
382where
383    M: MIntConvert + MIntConvert<u32>,
384{
385    let mut low = vec![Complex4::default(); blocks];
386    let mut high = low.clone();
387    for (i, x) in values.into_iter().enumerate() {
388        let mut x = <M as MIntConvert<u32>>::into(x.inner()) as i64;
389        if x * 2 > modulus {
390            x -= modulus;
391        }
392        let upper = (x as f64 / split as f64).round() as i64;
393        let lane = (i >> 1) & 3;
394        let (low, high) = if i & 1 == 0 {
395            (&mut low[i >> 3].re, &mut high[i >> 3].re)
396        } else {
397            (&mut low[i >> 3].im, &mut high[i >> 3].im)
398        };
399        low[lane] = (x - upper * split) as f64;
400        high[lane] = upper as f64;
401    }
402    (low, high)
403}

405// Fixed lane indexes keep coefficient order and compile to straight-line recovery.
406#[target_feature(enable = "avx2,fma")]
407#[allow(clippy::needless_range_loop)]
408pub(super) unsafe fn convolve_mint_avx2<M>(a: Vec<MInt<M>>, b: Vec<MInt<M>>) -> Vec<MInt<M>>
409where
410    M: MIntConvert + MIntConvert<u32>,
411{
412    let len = a.len() + b.len() - 1;
413    let n = len.next_power_of_two();
414    let modulus = <M as MIntConvert<u32>>::mod_into() as i64;
415    let split = (modulus as f64).sqrt() as i64 + 1;
416    let (mut a0, mut a1) = split_coefficients(a, n / 8, modulus, split);
417    let (mut b0, mut b1) = split_coefficients(b, n / 8, modulus, split);
418    real_fft_soa(&mut a0);
419    real_fft_soa(&mut a1);
420    real_fft_soa(&mut b0);
421    real_fft_soa(&mut b1);
422
423    let a00 = a0[0].get(0);
424    let a10 = a1[0].get(0);
425    let b00 = b0[0].get(0);
426    let b10 = b1[0].get(0);
427    for i in 0..a0.len() {
428        let (a0r, a0i) = load4(&a0[i]);
429        let (a1r, a1i) = load4(&a1[i]);
430        let (b0r, b0i) = load4(&b0[i]);
431        let (b1r, b1i) = load4(&b1[i]);
432        let (c0r, c0i) = mul4(a0r, a0i, b0r, b0i);
433        let (c01r, c01i) = mul4(a0r, a0i, b1r, b1i);
434        let (c10r, c10i) = mul4(a1r, a1i, b0r, b0i);
435        let (c2r, c2i) = mul4(a1r, a1i, b1r, b1i);
436        store4(&mut a0[i], c0r, c0i);
437        store4(
438            &mut a1[i],
439            _mm256_add_pd(c01r, c10r),
440            _mm256_add_pd(c01i, c10i),
441        );
442        store4(&mut b0[i], c2r, c2i);
443    }
444    a0[0].set(0, Complex::new(a00.re * b00.re, a00.im * b00.im));
445    a1[0].set(
446        0,
447        Complex::new(
448            a00.re * b10.re + a10.re * b00.re,
449            a00.im * b10.im + a10.im * b00.im,
450        ),
451    );
452    b0[0].set(0, Complex::new(a10.re * b10.re, a10.im * b10.im));
453    real_ifft_soa(&mut a0);
454    real_ifft_soa(&mut a1);
455    real_ifft_soa(&mut b0);
456    let inv_n = 2.0 / n as f64;
457    let modulus = modulus as u64;
458    let split = split as u64;
459    let split2 = split * split % modulus;
460    let scale = _mm256_set1_pd(inv_n);
461    let mut result = Vec::with_capacity(len);
462    for block in 0..a0.len() {
463        let values = [
464            [round4(&a0[block].re, scale), round4(&a0[block].im, scale)],
465            [round4(&a1[block].re, scale), round4(&a1[block].im, scale)],
466            [round4(&b0[block].re, scale), round4(&b0[block].im, scale)],
467        ];
468        for lane in 0..4 {
469            for part in 0..2 {
470                if result.len() == len {
471                    break;
472                }
473                let value = (values[0][part][lane].rem_euclid(modulus as i64) as u64
474                    + values[1][part][lane].rem_euclid(modulus as i64) as u64 * split
475                    + values[2][part][lane].rem_euclid(modulus as i64) as u64 * split2)
476                    % modulus;
477                result.push(MInt::<M>::from(value as u32));
478            }
479        }
480    }
481    result
