6#[inline]
7pub(super) fn splay<Spec, Data, Seeker>(
8    root: BstRoot<Spec>,
9    mut seeker: Seeker,
10) -> (Ordering, BstRoot<Spec>)
11where
12    Spec: BstSpec<Data = Data, Parent = WithNoParent<Data>>,
13    Seeker: BstSeeker<Spec = Spec>,
14{
15    let mut root = root;
16    let mut left_subtree = None;
17    let mut right_subtree = None;
18    let mut left_entry = &mut left_subtree;
19    let mut right_entry = &mut right_subtree;
20    let mut inline_stack = [None; 64];
21    let mut inline_len = 0;
22    let mut overflow_stack = vec![];
23
24    macro_rules! push_node {
25        ($node:expr) => {
26            if inline_len < inline_stack.len() {
27                inline_stack[inline_len] = Some($node);
28                inline_len += 1;
29            } else {
30                overflow_stack.push($node);
31            }
32        };
33    }
34
35    macro_rules! add {
36        (@left $node:ident) => {
37            *left_entry = Some($node.node);
38            push_node!($node.node);
39            left_entry = unsafe { &mut $node.node.as_mut().child[1] };
40        };
41        (@right $node:ident) => {
42            *right_entry = Some($node.node);
43            push_node!($node.node);
44            right_entry = unsafe { &mut $node.node.as_mut().child[0] };
45        };
46    }
47
48    let root_ordering = loop {
49        Spec::top_down(root.borrow_datamut());
50        match seeker.bst_seek(root.reborrow()) {
51            Ordering::Greater => {
52                let Some(mut child) = (unsafe { root.borrow_mut().left_mut().take() }) else {
53                    break Ordering::Greater;
54                };
55                Spec::top_down(child.borrow_datamut());
56                match seeker.bst_seek(child.reborrow()) {
57                    Ordering::Greater => {
58                        let Some(mut grandchild) =
59                            (unsafe { child.borrow_mut().left_mut().take() })
60                        else {
61                            add!(@right root);
62                            root = child;
63                            break Ordering::Greater;
64                        };
65                        Spec::top_down(grandchild.borrow_datamut());
66                        let child_right = unsafe { child.borrow_mut().right_mut().take() };
67                        if let Some(child_right) = child_right {
68                            unsafe { root.borrow_mut().left_mut().set(child_right) };
69                        }
70                        Spec::bottom_up(root.borrow_datamut());
71                        unsafe { child.borrow_mut().right_mut().set(root) };
72                        add!(@right child);
73                        root = grandchild;
74                    }
75                    Ordering::Equal => {
76                        add!(@right root);
77                        root = child;
78                        break Ordering::Equal;
79                    }
80                    Ordering::Less => {
81                        let Some(mut grandchild) =
82                            (unsafe { child.borrow_mut().right_mut().take() })
83                        else {
84                            add!(@right root);
85                            root = child;
86                            break Ordering::Less;
87                        };
88                        Spec::top_down(grandchild.borrow_datamut());
89                        add!(@right root);
90                        add!(@left child);
91                        root = grandchild;
92                    }
93                }
94            }
95            Ordering::Equal => break Ordering::Equal,
96            Ordering::Less => {
97                let Some(mut child) = (unsafe { root.borrow_mut().right_mut().take() }) else {
98                    break Ordering::Less;
99                };
100                Spec::top_down(child.borrow_datamut());
101                match seeker.bst_seek(child.reborrow()) {
102                    Ordering::Greater => {
103                        let Some(mut grandchild) =
104                            (unsafe { child.borrow_mut().left_mut().take() })
105                        else {
106                            add!(@left root);
107                            root = child;
108                            break Ordering::Greater;
109                        };
110                        Spec::top_down(grandchild.borrow_datamut());
111                        add!(@left root);
112                        add!(@right child);
113                        root = grandchild;
114                    }
115                    Ordering::Equal => {
116                        add!(@left root);
117                        root = child;
118                        break Ordering::Equal;
119                    }
120                    Ordering::Less => {
121                        let Some(mut grandchild) =
122                            (unsafe { child.borrow_mut().right_mut().take() })
123                        else {
124                            add!(@left root);
125                            root = child;
126                            break Ordering::Less;
127                        };
128                        Spec::top_down(grandchild.borrow_datamut());
129                        let child_left = unsafe { child.borrow_mut().left_mut().take() };
130                        if let Some(child_left) = child_left {
131                            unsafe { root.borrow_mut().right_mut().set(child_left) };
132                        }
133                        Spec::bottom_up(root.borrow_datamut());
134                        unsafe { child.borrow_mut().left_mut().set(root) };
135                        add!(@left child);
136                        root = grandchild;
137                    }
138                }
139            }
140        }
141    };
142
143    *left_entry = unsafe { root.borrow_mut().left_mut().take() }.map(|node| node.node);
144    *right_entry = unsafe { root.borrow_mut().right_mut().take() }.map(|node| node.node);
145    unsafe {
146        root.node.as_mut().child[0] = left_subtree;
147        root.node.as_mut().child[1] = right_subtree;
148        while let Some(node) = overflow_stack.pop() {
149            Spec::bottom_up(BstRoot::new(node).borrow_datamut());
150        }
151        while inline_len > 0 {
152            inline_len -= 1;
153            let node = inline_stack[inline_len].unwrap_unchecked();
154            Spec::bottom_up(BstRoot::new(node).borrow_datamut());
155        }
156    }
157    Spec::bottom_up(root.borrow_datamut());
158    (root_ordering, root)
159}

161#[inline]
162pub(super) fn merge<Spec, Data>(
163    left: Option<BstRoot<Spec>>,
164    right: Option<BstRoot<Spec>>,
165) -> Option<BstRoot<Spec>>
166where
167    Spec: BstSpec<Data = Data, Parent = WithNoParent<Data>>,
168{
169    match (left, right) {
170        (None, None) => None,
171        (None, Some(root)) | (Some(root), None) => Some(root),
172        (Some(left), Some(mut right)) if right.reborrow().left().descend().is_err() => {
173            Spec::top_down(right.borrow_datamut());
174            unsafe { right.borrow_mut().left_mut().set(left) };
175            Spec::bottom_up(right.borrow_datamut());
176            Some(right)
177        }
178        (Some(left), Some(right)) => {
179            let (_, mut root) = splay(left, SeekRight::default());
180            unsafe { root.borrow_mut().right_mut().set(right) };
181            Spec::bottom_up(root.borrow_datamut());
182            Some(root)
183        }
184    }
185}

187#[inline]
188pub(super) fn split<Spec, Data, Seeker>(
189    root: Option<BstRoot<Spec>>,
190    seeker: Seeker,
191    eq_left: bool,
192) -> (Option<BstRoot<Spec>>, Option<BstRoot<Spec>>)
193where
194    Spec: BstSpec<Data = Data, Parent = WithNoParent<Data>>,
195    Seeker: BstSeeker<Spec = Spec>,
196{
197    let Some(root) = root else {
198        return (None, None);
199    };
200    let (ordering, mut root) = splay(root, seeker);
201    match ordering {
202        Ordering::Less => {
203            let right = unsafe { root.borrow_mut().right_mut().take() };
204            Spec::bottom_up(root.borrow_datamut());
205            (Some(root), right)
206        }
207        Ordering::Greater => {
208            let left = unsafe { root.borrow_mut().left_mut().take() };
209            Spec::bottom_up(root.borrow_datamut());
210            (left, Some(root))
211        }
212        Ordering::Equal if eq_left => {
213            let right = unsafe { root.borrow_mut().right_mut().take() };
214            Spec::bottom_up(root.borrow_datamut());
215            (Some(root), right)
216        }
217        Ordering::Equal => {
218            let left = unsafe { root.borrow_mut().left_mut().take() };
219            Spec::bottom_up(root.borrow_datamut());
220            (left, Some(root))
221        }
222    }
