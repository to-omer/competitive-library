109impl<M, L> BstSpec for TreapSpec<M, L>
110where
111    M: MonoidAct,
112    M::Key: Ord,
113    L: LazyMapMonoid,
114{
115    type Parent = WithParent<Self::Data>;
116    type Data = TreapData<M, L>;
117
118    fn top_down(mut node: BstDataMutRef<'_, Self>) {
119        MonoidActElement::<M>::top_down(node.reborrow_datamut());
120        LazyMapElement::<L>::top_down(node.reborrow_datamut());
121    }
122
123    fn bottom_up(mut node: BstDataMutRef<'_, Self>) {
124        LazyMapElement::<L>::bottom_up(node.reborrow_datamut());
125    }
126
127    fn merge(
128        left: Option<TreapRoot<M, L>>,
129        right: Option<TreapRoot<M, L>>,
130    ) -> Option<TreapRoot<M, L>> {
131        match (left, right) {
132            (None, None) => None,
133            (None, Some(node)) | (Some(node), None) => Some(node),
134            (Some(mut left), Some(mut right)) => unsafe {
135                if left.reborrow().into_data().priority > right.reborrow().into_data().priority {
136                    TreapSpec::top_down(left.borrow_datamut());
137                    let lr = left.borrow_mut().right().take();
138                    let lr = Self::merge(lr, Some(right)).unwrap_unchecked();
139                    left.borrow_mut().right().set(lr);
140                    TreapSpec::bottom_up(left.borrow_datamut());
141                    Some(left)
142                } else {
143                    TreapSpec::top_down(right.borrow_datamut());
144                    let rl = right.borrow_mut().left().take();
145                    let rl = Self::merge(Some(left), rl).unwrap_unchecked();
146                    right.borrow_mut().left().set(rl);
147                    TreapSpec::bottom_up(right.borrow_datamut());
148                    Some(right)
149                }
150            },
151        }
152    }
153
154    fn split<Seeker>(
155        node: Option<TreapRoot<M, L>>,
156        mut seeker: Seeker,
157        eq_left: bool,
158    ) -> (Option<TreapRoot<M, L>>, Option<TreapRoot<M, L>>)
159    where
160        Seeker: BstSeeker<Spec = Self>,
161    {
162        match node {
163            None => (None, None),
164            Some(mut node) => {
165                Self::top_down(node.borrow_datamut());
166                let seek_left = match seeker.bst_seek(node.reborrow()) {
167                    Ordering::Less => false,
168                    Ordering::Equal => !eq_left,
169                    Ordering::Greater => true,
170                };
171                if seek_left {
172                    unsafe {
173                        let left = node.borrow_mut().left().take();
174                        let (l, r) = Self::split(left, seeker, eq_left);
175                        if let Some(r) = r {
176                            node.borrow_mut().left().set(r);
177                        }
178                        Self::bottom_up(node.borrow_datamut());
179                        (l, Some(node))
180                    }
181                } else {
182                    unsafe {
183                        let right = node.borrow_mut().right().take();
184                        let (l, r) = Self::split(right, seeker, eq_left);
185                        if let Some(l) = l {
186                            node.borrow_mut().right().set(l);
187                        }
188                        Self::bottom_up(node.borrow_datamut());
189                        (Some(node), r)
190                    }
191                }
192            }
193        }
194    }
195}

197impl<M, L> TreapSpec<M, L>
198where
199    M: MonoidAct,
200    M::Key: Ord,
201    L: LazyMapMonoid,
202{
203    pub fn merge_ordered(
204        left: Option<TreapRoot<M, L>>,
205        right: Option<TreapRoot<M, L>>,
206    ) -> Option<TreapRoot<M, L>> {
207        match (left, right) {
208            (None, None) => None,
209            (None, Some(node)) | (Some(node), None) => Some(node),
210            (Some(mut left), Some(mut right)) => unsafe {
211                if left.reborrow().into_data().priority > right.reborrow().into_data().priority {
212                    Self::top_down(left.borrow_datamut());
213                    let key = &left.reborrow().into_data().key.key;
214                    let (rl, rr) = Self::split(Some(right), SeekByKey::new(key), false);
215                    let ll = left.borrow_mut().left().take();
216                    let lr = left.borrow_mut().right().take();
217                    if let Some(l) = Self::merge_ordered(ll, rl) {
218                        left.borrow_mut().left().set(l);
219                    }
220                    if let Some(r) = Self::merge_ordered(lr, rr) {
221                        left.borrow_mut().right().set(r);
222                    }
223                    Self::bottom_up(left.borrow_datamut());
224                    Some(left)
225                } else {
226                    Self::top_down(right.borrow_datamut());
227                    let key = &right.reborrow().into_data().key.key;
228                    let (ll, lr) = Self::split(Some(left), SeekByKey::new(key), false);
229                    let rl = right.borrow_mut().left().take();
230                    let rr = right.borrow_mut().right().take();
231                    if let Some(l) = Self::merge_ordered(ll, rl) {
232                        right.borrow_mut().left().set(l);
233                    }
234                    if let Some(r) = Self::merge_ordered(lr, rr) {
235                        right.borrow_mut().right().set(r);
236                    }
237                    Self::bottom_up(right.borrow_datamut());
238                    Some(right)
239                }
240            },
241        }
242    }
243}

307impl<M, L, A> Treap<M, L, A>
308where
309    M: MonoidAct,
310    M::Key: Ord,
311    L: LazyMapMonoid,
312    A: Allocator<TreapNode<M, L>>,
313{
314    pub fn len(&self) -> usize {
315        self.size
316    }
317
318    pub fn is_empty(&self) -> bool {
319        self.size == 0
320    }
321
322    pub fn clear(&mut self) {
323        unsafe {
324            if let Some(root) = self.root.take() {
325                root.into_dying().drop_all(self.allocator.deref_mut());
326            }
327            self.size = 0;
328            self.nodes.clear();
329        }
330    }
331
332    pub fn kth_inserted(&mut self, k: usize) -> Option<(&M::Key, &L::Key)> {
333        unsafe fn immut<M, L, A>(
334            _treap: &mut Treap<M, L, A>,
335            node: NonNull<TreapNode<M, L>>,
336        ) -> BstImmutRef<'_, TreapSpec<M, L>>
337        where
338            M: MonoidAct,
339            M::Key: Ord,
340            L: LazyMapMonoid,
341            A: Allocator<TreapNode<M, L>>,
342        {
343            unsafe { BstImmutRef::new_unchecked(node) }
344        }
345        let node = *self.nodes.get(k)?.as_ref()?;
346        unsafe {
347            WithParent::resolve_top_down::<TreapSpec<M, L>>(node);
348            let node = immut(self, node);
349            let data = node.into_data();
350            Some((&data.key.key, &data.value.key))
351        }
352    }
353
354    pub fn insert(&mut self, key: M::Key, value: L::Key) {
355        let (left, right) = TreapSpec::split(self.root.take(), SeekByKey::new(&key), false);
356        let data = TreapData {
357            priority: self.rng.rand64(),
358            key: MonoidActElement::from_key(key),
359            value: LazyMapElement::from_key(value),
360        };
361        let node = BstRoot::from_data(data, self.allocator.deref_mut());
362        self.nodes.push(Some(node.node));
363        self.size += 1;
364        self.root = TreapSpec::merge(TreapSpec::merge(left, Some(node)), right);
365    }
366
367    pub fn fold<Q, R>(&mut self, range: R) -> L::Agg
368    where
369        M::Key: Borrow<Q>,
370        Q: Ord + ?Sized,
371        R: RangeBounds<Q>,
372    {
373        if let Some(node) = Split3::seek_by_key(&mut self.root, range).mid() {
374            node.reborrow().into_data().value.agg.clone()
375        } else {
376            L::agg_unit()
377        }
378    }
379
380    pub fn update_key<Q, R>(&mut self, range: R, act: M::Act)
381    where
382        M::Key: Borrow<Q>,
383        Q: Ord + ?Sized,
384        R: RangeBounds<Q>,
385    {
386        let mut split = Split3::seek_by_key(&mut self.root, range);
387        if let Some(node) = split.mid_datamut() {
388            MonoidActElement::<M>::update_act(node, &act);
389        }
390        split.manually_merge(TreapSpec::merge_ordered);
391    }
392
393    pub fn update_value<Q, R>(&mut self, range: R, act: L::Act)
394    where
395        M::Key: Borrow<Q>,
396        Q: Ord + ?Sized,
397        R: RangeBounds<Q>,
398    {
399        if let Some(node) = Split3::seek_by_key(&mut self.root, range).mid_datamut() {
400            LazyMapElement::<L>::update_act(node, &act);
401        }
402    }

406mod tests {
407    use super::*;
408    use crate::algebra::{AdditiveOperation, FlattenAct, RangeSumRangeAdd};
409
410    #[test]
411    fn test_treap() {
412        const A: i64 = 100;
413        let mut rng = Xorshift::default();
414        let mut treap = Treap::<FlattenAct<AdditiveOperation<i64>>, RangeSumRangeAdd<i64>>::new();
415        let mut data = vec![];
416        for _ in 0..10000 {
417            let (l, r) = loop {
418                let l = rng.random(-A..=A);
419                let r = rng.random(-A..=A);
420                if l <= r {
421                    break (l, r);
422                }
423            };
424            assert_eq!(data.len(), treap.len());
425            assert_eq!(data.is_empty(), treap.is_empty());
426            match rng.random(0..5) {
427                0 => {
428                    let key = rng.random(-A..=A);
429                    let value = rng.random(-A..=A);
430                    treap.insert(key, value);
431                    data.push((key, value));
432                }
433                1 => {
434                    let expected: i64 = data
435                        .iter()
436                        .filter(|(k, _)| (l..r).contains(k))
437                        .map(|(_, v)| *v)
438                        .sum();
439                    let result = treap.fold(l..r).0;
440                    assert_eq!(expected, result);
441                }
442                2 => {
443                    let add = rng.random(-A..=A);
444                    for (k, v) in data.iter_mut() {
445                        if (l..r).contains(k) {
446                            *v += add;
447                        }
448                    }
449                    treap.update_value(l..r, add);
450                }
451                3 => {
452                    let add = rng.random(-A..=A);
453                    for (k, _) in data.iter_mut() {
454                        if (l..r).contains(k) {
455                            *k += add;
456                        }
457                    }
458                    treap.update_key(l..r, add);
459                }
460                _ => {
461                    if !data.is_empty() {
462                        let k = rng.random(0..data.len());
463                        let expected = data[k];
464                        let result = treap.kth_inserted(k).unwrap();
465                        assert_eq!(expected, (*result.0, *result.1));
466                    }
467                }
468            }
469        }
470    }
