115impl<Spec> BstSeeker for SeekBySize<Spec>
116where
117    Spec: BstSpec,
118    Spec::Data: BstDataAccess<data::marker::Size, Value = usize>,
119{
120    type Spec = Spec;
121
122    fn bst_seek(&mut self, node: BstImmutRef<'_, Self::Spec>) -> Ordering {
123        let lsize = node
124            .reborrow()
125            .left()
126            .descend()
127            .map(|l| *l.into_data().bst_data())
128            .unwrap_or_default();
129        let ord = lsize.cmp(&self.index);
130        if matches!(ord, Ordering::Less) {
131            self.index -= lsize + 1;
132        }
133        ord
134    }
135}

160impl<Spec, F, L> BstSeeker for SeekByAccCond<Spec, F, L>
161where
162    Spec: BstSpec,
163    Spec::Data: BstDataAccess<data::marker::LazyMap, Value = LazyMapElement<L>>,
164    L: LazyMapMonoid,
165    F: FnMut(&L::Agg) -> bool,
166{
167    type Spec = Spec;
168
169    fn bst_seek(&mut self, node: BstImmutRef<'_, Self::Spec>) -> Ordering {
170        if let Ok(left) = node.reborrow().left().descend() {
171            let left_agg = &left.into_data().bst_data().agg;
172            let nagg = L::agg_operate(&self.acc, left_agg);
173            if (self.f)(&nagg) {
174                return Ordering::Greater;
175            }
176            self.acc = nagg;
177        }
178        self.acc = L::agg_operate(&self.acc, &node.reborrow().into_data().bst_data().agg);
179        if (self.f)(&self.acc) {
180            Ordering::Equal
181        } else {
182            Ordering::Less
183        }
184    }
