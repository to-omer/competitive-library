81impl<T> ImplicitTreapSpec<T>
82where
83    T: LazyMapMonoid,
84{
85    fn update_act(mut node: BstDataMutRef<'_, Self>, act: &T::Act) {
86        if T::is_act_unit(act) {
87            return;
88        }
89        T::act_operate_assign(&mut node.data_mut().value.act, act);
90        node.data_mut().value.key = T::act_key(&node.reborrow().into_data().value.key, act);
91        if let Some(agg) = T::act_agg(&node.reborrow().into_data().value.agg, act) {
92            node.data_mut().value.agg = agg;
93        } else {
94            Self::top_down(node.reborrow_datamut());
95            Self::bottom_up(node);
96        }
97    }
98
99    fn reverse(mut node: BstDataMutRef<'_, Self>) {
100        unsafe {
101            node.node.as_mut().child.swap(0, 1);
102        }
103        let data = node.data_mut();
104        T::toggle(&mut data.value.agg);
105        data.rev ^= true;
106    }
107}

109impl<T> BstSpec for ImplicitTreapSpec<T>
110where
111    T: LazyMapMonoid,
112{
113    type Parent = WithNoParent<Self::Data>;
114    type Data = ImplicitTreapData<T>;
115
116    fn top_down(mut node: BstDataMutRef<'_, Self>) {
117        if !T::is_act_unit(&node.reborrow().into_data().value.act) {
118            let act = replace(&mut node.data_mut().value.act, T::act_unit());
119            if let Ok(left) = node.reborrow_datamut().left().descend() {
120                Self::update_act(left, &act);
121            }
122            if let Ok(right) = node.reborrow_datamut().right().descend() {
123                Self::update_act(right, &act);
124            }
125        }
126        if node.reborrow().into_data().rev {
127            node.data_mut().rev = false;
128            if let Ok(left) = node.reborrow_datamut().left().descend() {
129                Self::reverse(left);
130            }
131            if let Ok(right) = node.reborrow_datamut().right().descend() {
132                Self::reverse(right);
133            }
134        }
135    }
136
137    fn bottom_up(mut node: BstDataMutRef<'_, Self>) {
138        let mut agg = T::single_agg(&node.reborrow().into_data().value.key);
139        let mut size = 1;
140        if let Ok(left) = node.reborrow().left().descend() {
141            let data = left.into_data();
142            agg = T::agg_operate(&data.value.agg, &agg);
143            size += data.size;
144        }
145        if let Ok(right) = node.reborrow().right().descend() {
146            let data = right.into_data();
147            agg = T::agg_operate(&agg, &data.value.agg);
148            size += data.size;
149        }
150        let data = node.data_mut();
151        data.value.agg = agg;
152        data.size = size;
153    }
154
155    fn merge(
156        left: Option<ImplicitTreapRoot<T>>,
157        right: Option<ImplicitTreapRoot<T>>,
158    ) -> Option<ImplicitTreapRoot<T>> {
159        match (left, right) {
160            (None, None) => None,
161            (None, Some(node)) | (Some(node), None) => Some(node),
162            (Some(mut left), Some(mut right)) => unsafe {
163                if left.reborrow().into_data().priority > right.reborrow().into_data().priority {
164                    Self::top_down(left.borrow_datamut());
165                    let lr = left.borrow_mut().right().take();
166                    let lr = Self::merge(lr, Some(right)).unwrap_unchecked();
167                    left.borrow_mut().right().set(lr);
168                    Self::bottom_up(left.borrow_datamut());
169                    Some(left)
170                } else {
171                    Self::top_down(right.borrow_datamut());
172                    let rl = right.borrow_mut().left().take();
173                    let rl = Self::merge(Some(left), rl).unwrap_unchecked();
174                    right.borrow_mut().left().set(rl);
175                    Self::bottom_up(right.borrow_datamut());
176                    Some(right)
177                }
178            },
179        }
180    }
181
182    fn split<Seeker>(
183        node: Option<ImplicitTreapRoot<T>>,
184        mut seeker: Seeker,
185        eq_left: bool,
186    ) -> (Option<ImplicitTreapRoot<T>>, Option<ImplicitTreapRoot<T>>)
187    where
188        Seeker: BstSeeker<Spec = Self>,
189    {
190        match node {
191            None => (None, None),
192            Some(mut node) => {
193                Self::top_down(node.borrow_datamut());
194                let seek_left = match seeker.bst_seek(node.reborrow()) {
195                    Ordering::Less => false,
196                    Ordering::Equal => !eq_left,
197                    Ordering::Greater => true,
198                };
199                if seek_left {
200                    unsafe {
201                        let left = node.borrow_mut().left().take();
202                        let (l, r) = Self::split(left, seeker, eq_left);
203                        if let Some(r) = r {
204                            node.borrow_mut().left().set(r);
205                        }
206                        Self::bottom_up(node.borrow_datamut());
207                        (l, Some(node))
208                    }
209                } else {
210                    unsafe {
211                        let right = node.borrow_mut().right().take();
212                        let (l, r) = Self::split(right, seeker, eq_left);
213                        if let Some(l) = l {
214                            node.borrow_mut().right().set(l);
215                        }
216                        Self::bottom_up(node.borrow_datamut());
217                        (Some(node), r)
218                    }
219                }
220            }
221        }
222    }
223}

287impl<T, A> ImplicitTreap<T, A>
288where
289    T: LazyMapMonoid,
290    A: Allocator<ImplicitTreapNode<T>>,
291{
292    fn node(&mut self, key: T::Key) -> ImplicitTreapRoot<T> {
293        BstRoot::from_data(
294            ImplicitTreapData {
295                priority: self.rng.rand64(),
296                value: LazyMapElement::from_key(key),
297                size: 1,
298                rev: false,
299            },
300            self.allocator.deref_mut(),
301        )
302    }
303
304    fn build<I>(&mut self, iter: I) -> (Option<ImplicitTreapRoot<T>>, usize)
305    where
306        I: IntoIterator<Item = T::Key>,
307    {
308        let mut stack = vec![];
309        let mut len = 0;
310        for key in iter {
311            let mut cur = self.node(key).node;
312            let mut left = None;
313            unsafe {
314                while stack
315                    .last()
316                    .is_some_and(|node: &NonNull<ImplicitTreapNode<T>>| {
317                        node.as_ref().data.priority < cur.as_ref().data.priority
318                    })
319                {
320                    left = stack.pop();
321                }
322                cur.as_mut().child[0] = left;
323                if let Some(parent) = stack.last_mut() {
324                    parent.as_mut().child[1] = Some(cur);
325                }
326            }
327            stack.push(cur);
328            len += 1;
329        }
330        let root = stack.first().copied().map(BstRoot::new);
331        if let Some(mut root) = root {
332            Self::build_bottom_up(root.borrow_datamut());
333            (Some(root), len)
334        } else {
335            (None, len)
336        }
337    }
338
339    fn build_bottom_up(mut node: BstDataMutRef<'_, ImplicitTreapSpec<T>>) {
340        if let Ok(left) = node.reborrow_datamut().left().descend() {
341            Self::build_bottom_up(left);
342        }
343        if let Ok(right) = node.reborrow_datamut().right().descend() {
344            Self::build_bottom_up(right);
345        }
346        ImplicitTreapSpec::<T>::bottom_up(node);
347    }
348
349    pub fn len(&self) -> usize {
350        self.length
351    }
352
353    pub fn is_empty(&self) -> bool {
354        self.length == 0
355    }
356
357    pub fn update<R>(&mut self, range: R, x: T::Act)
358    where
359        R: RangeBounds<usize>,
360    {
361        let mut split = Split3::seek_by_size(&mut self.root, range);
362        if let Some(root) = split.mid_datamut() {
363            ImplicitTreapSpec::<T>::update_act(root, &x);
364        }
365    }
366
367    pub fn fold<R>(&mut self, range: R) -> T::Agg
368    where
369        R: RangeBounds<usize>,
370    {
371        let split = Split3::seek_by_size(&mut self.root, range);
372        split
373            .mid()
374            .map(|node| node.into_data().value.agg.clone())
375            .unwrap_or_else(T::agg_unit)
376    }
377
378    pub fn reverse<R>(&mut self, range: R)
379    where
380        R: RangeBounds<usize>,
381    {
382        let mut split = Split3::seek_by_size(&mut self.root, range);
383        if let Some(root) = split.mid_datamut() {
384            ImplicitTreapSpec::<T>::reverse(root);
385        }
386    }
387
388    pub fn get(&mut self, index: usize) -> Option<&T::Key> {
389        if index >= self.length {
390            return None;
391        }
392        let split = Split3::seek_by_size(&mut self.root, index..=index);
393        let node = split.mid()?.node;
394        drop(split);
395        Some(unsafe { &(*node.as_ptr()).data.value.key })
396    }
397
398    pub fn modify<F>(&mut self, index: usize, f: F)
399    where
400        F: FnOnce(&T::Key) -> T::Key,
401    {
402        assert!(index < self.length);
403        let mut split = Split3::seek_by_size(&mut self.root, index..=index);
404        let mut node = split.mid_datamut().unwrap();
405        ImplicitTreapSpec::<T>::top_down(node.reborrow_datamut());
406        {
407            let data = node.data_mut();
408            data.value.key = f(&data.value.key);
409            data.value.agg = T::single_agg(&data.value.key);
410        }
411        ImplicitTreapSpec::<T>::bottom_up(node);
412    }
413
414    pub fn insert(&mut self, index: usize, x: T::Key) {
415        assert!(index <= self.length);
416        let node = self.node(x);
417        if index == 0 {
418            self.root = ImplicitTreapSpec::<T>::merge(Some(node), self.root.take());
419        } else if index == self.length {
420            self.root = ImplicitTreapSpec::<T>::merge(self.root.take(), Some(node));
421        } else {
422            let mut node = Some(node);
423            let mut split = Split::new(&mut self.root, SeekBySize::new(index), false);
424            split.manually_merge(|left, right| {
425                ImplicitTreapSpec::<T>::merge(
426                    ImplicitTreapSpec::<T>::merge(left, node.take()),
427                    right,
428                )
429            });
430        }
431        self.length += 1;
432    }
433
434    pub fn remove(&mut self, index: usize) -> Option<T::Key> {
435        if index >= self.length {
436            return None;
437        }
438        let mid;
439        if index == 0 {
440            let (left, right) =
441                ImplicitTreapSpec::<T>::split(self.root.take(), SeekBySize::new(1), false);
442            mid = left;
443            self.root = right;
444        } else if index + 1 == self.length {
445            let (left, right) =
446                ImplicitTreapSpec::<T>::split(self.root.take(), SeekBySize::new(index), false);
447            mid = right;
448            self.root = left;
449        } else {
450            let (left, rest) =
451                ImplicitTreapSpec::<T>::split(self.root.take(), SeekBySize::new(index), false);
452            let (middle, right) = ImplicitTreapSpec::<T>::split(rest, SeekBySize::new(1), false);
453            mid = middle;
454            self.root = ImplicitTreapSpec::<T>::merge(left, right);
455        }
456        self.length -= 1;
457        let mut node = mid.unwrap();
458        ImplicitTreapSpec::<T>::top_down(node.borrow_datamut());
459        let data = unsafe { node.into_dying().into_data(self.allocator.deref_mut()) };
460        Some(data.value.key)
461    }
462
463    pub fn position_acc<R, F>(&mut self, range: R, f: F) -> Option<usize>
464    where
465        R: RangeBounds<usize>,
466        F: FnMut(&T::Agg) -> bool,
467    {
468        let mut split3 = Split3::seek_by_size(&mut self.root, range);
469        let front_size = split3
470            .left()
471            .map(|node| node.into_data().size)
472            .unwrap_or_default();
473        let split = split3.split_mid(SeekByAccCond::<ImplicitTreapSpec<T>, T, F>::new(f), false);
474        split.right()?;
475        let index = split
476            .left()
477            .map(|node| node.into_data().size)
478            .unwrap_or_default();
479        Some(front_size + index)
480    }
481
482    pub fn rposition_acc<R, F>(&mut self, range: R, f: F) -> Option<usize>
483    where
484        R: RangeBounds<usize>,
485        F: FnMut(&T::Agg) -> bool,
486    {
487        let mut split3 = Split3::seek_by_size(&mut self.root, range);
488        let front_size = split3
489            .left()
490            .map(|node| node.into_data().size)
491            .unwrap_or_default();
492        let split = split3.split_mid(SeekByRaccCond::<ImplicitTreapSpec<T>, T, F>::new(f), true);
493        let left_size = split.left()?.into_data().size;
494        Some(front_size + left_size - 1)
495    }
496
497    pub fn rotate_left(&mut self, mid: usize) {
498        assert!(mid <= self.length);
499        if mid == 0 || mid == self.length {
500            return;
501        }
502        let (left, right) =
503            ImplicitTreapSpec::<T>::split(self.root.take(), SeekBySize::new(mid), false);
504        self.root = ImplicitTreapSpec::<T>::merge(right, left);
505    }
506
507    pub fn rotate_right(&mut self, k: usize) {
508        assert!(k <= self.length);
509        self.rotate_left(self.length - k);
510    }
511}
