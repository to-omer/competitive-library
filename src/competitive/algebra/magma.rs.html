3/// binary operaion: $T \circ T \to T$
4pub trait Magma {
5    /// type of operands: $T$
6    type T: Clone;
7    /// binary operaion: $\circ$
8    fn operate(x: &Self::T, y: &Self::T) -> Self::T;
9    #[inline]
10    fn reverse_operate(x: &Self::T, y: &Self::T) -> Self::T {
11        Self::operate(y, x)
12    }
13    #[inline]
14    fn operate_assign(x: &mut Self::T, y: &Self::T) {
15        *x = Self::operate(x, y);
16    }
17}
18
19/// $\forall a,\forall b,\forall c \in T, (a \circ b) \circ c = a \circ (b \circ c)$
20pub trait Associative {}
21
22/// associative binary operation
23pub trait SemiGroup: Magma + Associative {}
24
25impl<S> SemiGroup for S where S: Magma + Associative {}

27/// $\exists e \in T, \forall a \in T, e \circ a = a \circ e = e$
28pub trait Unital: Magma {
29    /// identity element: $e$
30    fn unit() -> Self::T;
31    #[inline]
32    fn is_unit(x: &Self::T) -> bool
33    where
34        <Self as Magma>::T: PartialEq,
35    {
36        x == &Self::unit()
37    }
38    #[inline]
39    fn set_unit(x: &mut Self::T) {
40        *x = Self::unit();
41    }
42}
43
44/// associative binary operation and an identity element
45pub trait Monoid: SemiGroup + Unital {
46    /// binary exponentiation: $x^n = x\circ\ddots\circ x$
47    fn pow(mut x: Self::T, mut n: usize) -> Self::T {
48        let mut res = Self::unit();
49        while n > 0 {
50            if n & 1 == 1 {
51                res = Self::operate(&res, &x);
52            }
53            x = Self::operate(&x, &x);
54            n >>= 1;
55        }
56        res
57    }
58}
59
60impl<M> Monoid for M where M: SemiGroup + Unital {}

62/// $\exists e \in T, \forall a \in T, \exists b,c \in T, b \circ a = a \circ c = e$
63pub trait Invertible: Magma {
64    /// $a$ where $a \circ x = e$
65    fn inverse(x: &Self::T) -> Self::T;
66    #[inline]
67    fn rinv_operate(x: &Self::T, y: &Self::T) -> Self::T {
68        Self::operate(x, &Self::inverse(y))
69    }
70    #[inline]
71    fn rinv_operate_assign(x: &mut Self::T, y: &Self::T) {
72        *x = Self::rinv_operate(x, y);
73    }
74}

102#[macro_export]
103macro_rules! monoid_fold {
104    ($m:ty) => { <$m as Unital>::unit() };
105    ($m:ty,) => { <$m as Unital>::unit() };
106    ($m:ty, $f:expr) => { $f };
107    ($m:ty, $f:expr, $($ff:expr),*) => { <$m as Magma>::operate(&($f), &monoid_fold!($m, $($ff),*)) };
108}

110#[macro_export]
111macro_rules! define_monoid {
112    ($Name:ident, $t:ty, |$x:ident, $y:ident| $op:expr, $unit:expr) => {
113        struct $Name;
114        impl Magma for $Name {
115            type T = $t;
116            fn operate($x: &Self::T, $y: &Self::T) -> Self::T {
117                $op
118            }
119        }
120        impl Unital for $Name {
121            fn unit() -> Self::T {
122                $unit
123            }
124        }
125        impl Associative for $Name {}
126    };
127}
