4#[codesnip::entry("HeavyLightDecomposition", include("algebra", "SparseGraph"))]
5pub struct HeavyLightDecomposition {
6    pub par: Vec<usize>,
7    size: Vec<usize>,
8    head: Vec<usize>,
9    pub vidx: Vec<usize>,
10}
11#[codesnip::entry("HeavyLightDecomposition")]
12impl HeavyLightDecomposition {
13    pub fn new(root: usize, graph: &mut UndirectedSparseGraph) -> Self {
14        let mut self_ = Self {
15            par: vec![0; graph.vertices_size()],
16            size: vec![0; graph.vertices_size()],
17            head: vec![0; graph.vertices_size()],
18            vidx: vec![0; graph.vertices_size()],
19        };
20        self_.build(root, graph);
21        self_
22    }
23    fn dfs_size(&mut self, u: usize, p: usize, graph: &mut UndirectedSparseGraph) {
24        self.par[u] = p;
25        self.size[u] = 1;
26        let base = graph.start[u];
27        if graph.adjacencies(u).len() > 1 && graph.adjacencies(u).next().unwrap().to == p {
28            graph.elist.swap(base, base + 1);
29        }
30        for i in base..graph.start[u + 1] {
31            let a = graph.elist[i];
32            if a.to != p {
33                self.dfs_size(a.to, u, graph);
34                self.size[u] += self.size[a.to];
35                if self.size[graph.elist[base].to] < self.size[a.to] {
36                    graph.elist.swap(base, i);
37                }
38            }
39        }
40    }
41    fn dfs_hld(&mut self, u: usize, p: usize, t: &mut usize, graph: &UndirectedSparseGraph) {
42        self.vidx[u] = *t;
43        *t += 1;
44        let mut adjacencies = graph.adjacencies(u).filter(|a| a.to != p);
45        if let Some(a) = adjacencies.next() {
46            self.head[a.to] = self.head[u];
47            self.dfs_hld(a.to, u, t, graph);
48        }
49        for a in adjacencies {
50            self.head[a.to] = a.to;
51            self.dfs_hld(a.to, u, t, graph);
52        }
53    }
54    fn build(&mut self, root: usize, graph: &mut UndirectedSparseGraph) {
55        self.head[root] = root;
56        self.dfs_size(root, graph.vertices_size(), graph);
57        let mut t = 0;
58        self.dfs_hld(root, graph.vertices_size(), &mut t, graph);
59    }
60    pub fn lca(&self, mut u: usize, mut v: usize) -> usize {
61        loop {
62            if self.vidx[u] > self.vidx[v] {
63                std::mem::swap(&mut u, &mut v);
64            }
65            if self.head[u] == self.head[v] {
66                return u;
67            }
68            v = self.par[self.head[v]];
69        }
70    }
71    pub fn update<F: FnMut(usize, usize)>(
72        &self,
73        mut u: usize,
74        mut v: usize,
75        is_edge: bool,
76        mut f: F,
77    ) {
78        loop {
79            if self.vidx[u] > self.vidx[v] {
80                std::mem::swap(&mut u, &mut v);
81            }
82            if self.head[u] == self.head[v] {
83                break;
84            }
85            f(self.vidx[self.head[v]], self.vidx[v] + 1);
86            v = self.par[self.head[v]];
87        }
88        f(self.vidx[u] + is_edge as usize, self.vidx[v] + 1);
89    }
90    pub fn query<M: Monoid, F: FnMut(usize, usize) -> M::T>(
91        &self,
92        mut u: usize,
93        mut v: usize,
94        is_edge: bool,
95        mut f: F,
96    ) -> M::T {
97        let (mut l, mut r) = (M::unit(), M::unit());
98        loop {
99            if self.vidx[u] > self.vidx[v] {
100                std::mem::swap(&mut u, &mut v);
101                std::mem::swap(&mut l, &mut r);
102            }
103            if self.head[u] == self.head[v] {
104                break;
105            }
106            l = M::operate(&f(self.vidx[self.head[v]], self.vidx[v] + 1), &l);
107            v = self.par[self.head[v]];
108        }
109        M::operate(
110            &M::operate(&f(self.vidx[u] + is_edge as usize, self.vidx[v] + 1), &l),
111            &r,
112        )
113    }
114    pub fn query_noncom<
115        M: Monoid,
116        F1: FnMut(usize, usize) -> M::T,
117        F2: FnMut(usize, usize) -> M::T,
118    >(
119        &self,
120        mut u: usize,
121        mut v: usize,
122        is_edge: bool,
123        mut f1: F1,
124        mut f2: F2,
125    ) -> M::T {
126        let (mut l, mut r) = (M::unit(), M::unit());
127        while self.head[u] != self.head[v] {
128            if self.vidx[u] > self.vidx[v] {
129                l = M::operate(&l, &f2(self.vidx[self.head[u]], self.vidx[u] + 1));
130                u = self.par[self.head[u]];
131            } else {
132                r = M::operate(&f1(self.vidx[self.head[v]], self.vidx[v] + 1), &r);
133                v = self.par[self.head[v]];
134            }
135        }
136        M::operate(
137            &M::operate(
138                &l,
139                &if self.vidx[u] > self.vidx[v] {
140                    f2(self.vidx[v] + is_edge as usize, self.vidx[u] + 1)
141                } else {
142                    f1(self.vidx[u] + is_edge as usize, self.vidx[v] + 1)
143                },
144            ),
145            &r,
146        )
147    }
