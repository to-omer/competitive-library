11pub trait LinkCutTreeSpec: Sized {
12    type Value;
13    type Data;
14
15    /// Whether splay must propagate from the auxiliary root before rotations.
16    const ROOT_TO_NODE_TOP_DOWN: bool = true;
17
18    fn new(value: Self::Value) -> Self::Data;
19    fn value(data: &Self::Data) -> &Self::Value;
20    fn value_mut(data: &mut Self::Data) -> &mut Self::Value;
21    fn top_down(_data: &mut Self::Data, _children: [Option<&mut Self::Data>; 2]) {}
22    fn bottom_up(data: &mut Self::Data, children: [Option<&Self::Data>; 2]);
23    fn reverse(data: &mut Self::Data);
24    fn attach_virtual(_parent: &mut Self::Data, _child: &mut Self::Data) {}
25    fn detach_virtual(_parent: &mut Self::Data, _child: &mut Self::Data) {}
26    /// Moves state associated with the same path-parent edge after a splay.
27    fn transfer_path_parent(_old_root: &mut Self::Data, _new_root: &mut Self::Data) {}

66impl<S> LinkCutBstSpec<S>
67where
68    S: LinkCutTreeSpec,
69{
70    #[inline]
71    unsafe fn toggle(mut node: LinkCutPtr<S>) {
72        unsafe { node.as_mut().child.swap(0, 1) };
73        let data = unsafe { &mut node.as_mut().data };
74        data.reverse ^= true;
75        S::reverse(&mut data.inner);
76    }
77
78    #[inline]
79    unsafe fn with_two_inner_mut<R>(
80        mut left: LinkCutPtr<S>,
81        mut right: LinkCutPtr<S>,
82        f: impl FnOnce(&mut S::Data, &mut S::Data) -> R,
83    ) -> R {
84        let left = unsafe { &mut left.as_mut().data.inner };
85        let right = unsafe { &mut right.as_mut().data.inner };
86        f(left, right)
87    }
88}

90impl<S> BstSpec for LinkCutBstSpec<S>
91where
92    S: LinkCutTreeSpec,
93{
94    type Parent = WithParent<Self::Data>;
95    type Data = LinkCutData<S>;

97    #[inline]
98    fn top_down(mut node: BstDataMutRef<'_, Self>) {
99        let pointer = node.node;
100        if node.reborrow().into_data().reverse {
101            node.data_mut().reverse = false;
102            let children = unsafe { pointer.as_ref().child };
103            for child in children.into_iter().flatten() {
104                unsafe { Self::toggle(child) };
105            }
106        }
107        let children = unsafe { pointer.as_ref().child };
108        let data = unsafe { &mut (*pointer.as_ptr()).data.inner };
109        let children =
110            children.map(|child| child.map(|child| unsafe { &mut (*child.as_ptr()).data.inner }));
111        S::top_down(data, children);
112    }
113
114    #[inline]
115    fn bottom_up(node: BstDataMutRef<'_, Self>) {
116        let pointer = node.node;
117        let children = unsafe { pointer.as_ref().child };
118        let data = unsafe { &mut (*pointer.as_ptr()).data.inner };
119        let children =
120            children.map(|child| child.map(|child| unsafe { &(*child.as_ptr()).data.inner }));
121        S::bottom_up(data, children);
122    }
123
124    fn merge(_left: Option<BstRoot<Self>>, _right: Option<BstRoot<Self>>) -> Option<BstRoot<Self>> {
125        unreachable!("link-cut trees do not merge auxiliary trees through BstSpec")
126    }
127
128    fn split<Seeker>(
129        _node: Option<BstRoot<Self>>,
130        _seeker: Seeker,
131        _equal_side: EqualSide,
132    ) -> (Option<BstRoot<Self>>, Option<BstRoot<Self>>)
133    where
134        Seeker: BstSeeker<Spec = Self>,
135    {
136        unreachable!("link-cut trees do not split auxiliary trees through BstSpec")
137    }

140/// A link-cut forest with stable insertion-order node identifiers.
141///
142/// Its dynamic-tree operations take amortized `O(log n)` time when the spec
143/// hooks take constant time.
144pub struct LinkCutTree<S>
145where
146    S: LinkCutTreeSpec,
147{
148    nodes: Vec<LinkCutPtr<S>>,
149    allocator: MemoryPool<LinkCutNode<S>>,
150}
151
152impl<S> LinkCutTree<S>
153where
154    S: LinkCutTreeSpec,
155{
156    pub fn with_capacity(capacity: usize) -> Self {
157        Self {
158            nodes: Vec::with_capacity(capacity),
159            allocator: MemoryPool::with_capacity(capacity),
160        }
161    }
162
163    /// `edges` must form a tree over the values in iteration order.
164    pub fn from_edges<T>(values: T, edges: &[(usize, usize)]) -> Self
165    where
166        T: IntoIterator<Item = S::Value>,
167    {
168        let tree: Self = values.into_iter().collect();
169        for (child, parent, preferred) in
170            splay_operations::rooted_heavy_order(tree.nodes.len(), edges)
171                .into_iter()
172                .rev()
173        {
174            let child = tree.node(child);
175            let mut parent = tree.node(parent);
176            unsafe {
177                (*child.as_ptr()).parent.parent = Some(parent);
178                if preferred {
179                    parent.as_mut().child[1] = Some(child);
180                } else {
181                    LinkCutBstSpec::<S>::with_two_inner_mut(parent, child, S::attach_virtual);
182                }
183                Self::pull(parent);
184            }
185        }
186        tree
187    }
188
189    pub fn add_node(&mut self, value: S::Value) -> usize {
190        let index = self.nodes.len();
191        let node = self.allocator.allocate(BstNode::new(LinkCutData {
192            inner: S::new(value),
193            index,
194            reverse: false,
195        }));
196        self.nodes.push(node);
197        index
198    }
199
200    #[inline]
201    fn node(&self, index: usize) -> LinkCutPtr<S> {
202        self.nodes[index]
203    }
204
205    #[inline]
206    unsafe fn pull(node: LinkCutPtr<S>) {
207        unsafe {
208            LinkCutBstSpec::<S>::bottom_up(BstDataMutRef::new_unchecked(node));
209        }
210    }
211
212    #[inline]
213    unsafe fn splay(node: LinkCutPtr<S>) {
214        let root = if S::ROOT_TO_NODE_TOP_DOWN {
215            unsafe {
216                splay_operations::with_parent::splay::<LinkCutBstSpec<S>, LinkCutData<S>>(node)
217            }
218        } else {
219            unsafe {
220                splay_operations::with_parent::splay_with_local_top_down::<
221                    LinkCutBstSpec<S>,
222                    LinkCutData<S>,
223                >(node)
224            }
225        };
226        if root != node {
227            unsafe {
228                LinkCutBstSpec::<S>::with_two_inner_mut(root, node, S::transfer_path_parent);
229            }
230        }
231    }
232
233    fn access_node(mut node: LinkCutPtr<S>) {
234        unsafe {
235            Self::splay(node);
236            if let Some(right) = node.as_mut().child[1].take() {
237                LinkCutBstSpec::<S>::with_two_inner_mut(node, right, S::attach_virtual);
238            }
239            Self::pull(node);
240            while let Some(mut parent) = node.as_ref().parent.parent {
241                Self::splay(parent);
242                if let Some(right) = parent.as_mut().child[1].take() {
243                    LinkCutBstSpec::<S>::with_two_inner_mut(parent, right, S::attach_virtual);
244                }
245                LinkCutBstSpec::<S>::with_two_inner_mut(parent, node, S::detach_virtual);
246                parent.as_mut().child[1] = Some(node);
247                node.as_mut().parent.parent = Some(parent);
248                Self::pull(parent);
249                Self::splay(node);
250            }
251        }
252    }
253
254    pub fn get(&mut self, node: usize) -> &S::Value {
255        let node = self.node(node);
256        Self::access_node(node);
257        unsafe { S::value(&node.as_ref().data.inner) }
258    }
259
260    pub fn set(&mut self, node: usize, value: S::Value) {
261        let node = self.node(node);
262        Self::access_node(node);
263        unsafe {
264            *S::value_mut(&mut (*node.as_ptr()).data.inner) = value;
265            Self::pull(node);
266        }
267    }
268
269    pub fn modify<F>(&mut self, node: usize, f: F)
270    where
271        F: FnOnce(&S::Value) -> S::Value,
272    {
273        let node = self.node(node);
274        Self::access_node(node);
275        unsafe {
276            let data = &mut (*node.as_ptr()).data.inner;
277            *S::value_mut(data) = f(S::value(data));
278            Self::pull(node);
279        }
280    }
281
282    pub fn reroot(&mut self, node: usize) {
283        let node = self.node(node);
284        Self::access_node(node);
285        unsafe { LinkCutBstSpec::<S>::toggle(node) };
286    }
287
288    /// `child` and `parent` must belong to different trees.
289    pub fn link(&mut self, child: usize, parent: usize) {
290        assert_ne!(child, parent);
291        self.reroot(child);
292        let child = self.node(child);
293        let parent = self.node(parent);
294        Self::access_node(parent);
295        unsafe {
296            (*child.as_ptr()).parent.parent = Some(parent);
297            LinkCutBstSpec::<S>::with_two_inner_mut(parent, child, S::attach_virtual);
298            Self::pull(parent);
299        }
300    }
301
302    /// `(u, v)` must be an edge.
303    pub fn cut(&mut self, u: usize, v: usize) {
304        assert_ne!(u, v);
305        self.reroot(u);
306        let mut v = self.node(v);
307        Self::access_node(v);
308        unsafe {
309            let mut left = v.as_mut().child[0]
310                .take()
311                .expect("the specified edge must exist");
312            left.as_mut().parent.parent = None;
313            Self::pull(v);
314        }
315    }
316
317    pub fn root(&mut self, node: usize) -> usize {
318        let mut root = self.node(node);
319        Self::access_node(root);
320        unsafe {
321            loop {
322                LinkCutBstSpec::<S>::top_down(BstDataMutRef::new_unchecked(root));
323                match root.as_ref().child[0] {
324                    Some(left) => root = left,
325                    None => break,
326                }
327            }
328            Self::splay(root);
329            root.as_ref().data.index
330        }
331    }
332
333    pub fn is_connected(&mut self, u: usize, v: usize) -> bool {
334        self.root(u) == self.root(v)
335    }
336
337    fn detach_left<R>(node: LinkCutPtr<S>, f: impl FnOnce(&mut S::Data) -> R) -> R {
338        unsafe {
339            let left = (*node.as_ptr()).child[0].take();
340            if let Some(mut left) = left {
341                left.as_mut().parent.parent = None;
342            }
343            Self::pull(node);
344            let result = f(&mut (*node.as_ptr()).data.inner);
345            LinkCutBstSpec::<S>::top_down(BstDataMutRef::new_unchecked(node));
346            (*node.as_ptr()).child[0] = left;
347            if let Some(mut left) = left {
348                left.as_mut().parent.parent = Some(node);
349            }
350            Self::pull(node);
351            result
352        }
353    }
354}

432impl<L> PathLinkCutTreeSpec<L>
433where
434    L: LazyMapMonoid,
435{
436    #[inline]
437    fn apply_non_unit(data: &mut PathLinkCutTreeData<L>, action: &L::Act) {
438        L::act_operate_assign(&mut data.value.act, action);
439        data.value.key = L::act_key(&data.value.key, action);
440        data.value.agg = L::act_agg(&data.value.agg, action)
441            .expect("a path link-cut tree action must update aggregates lazily");
442    }

445impl<L> LinkCutTreeSpec for PathLinkCutTreeSpec<L>
446where
447    L: LazyMapMonoid,
448{
449    type Value = L::Key;
450    type Data = PathLinkCutTreeData<L>;
451
452    const ROOT_TO_NODE_TOP_DOWN: bool = false;
453
454    fn new(value: Self::Value) -> Self::Data {
455        Self::Data {
456            value: LazyMapElement::from_key(value),
457        }
458    }
459
460    fn value(data: &Self::Data) -> &Self::Value {
461        &data.value.key
462    }
463
464    fn value_mut(data: &mut Self::Data) -> &mut Self::Value {
465        &mut data.value.key
466    }
467
468    fn top_down(data: &mut Self::Data, children: [Option<&mut Self::Data>; 2]) {
469        if L::is_act_unit(&data.value.act) {
470            return;
471        }
472        let action = replace(&mut data.value.act, L::act_unit());
473        for child in children.into_iter().flatten() {
474            Self::apply_non_unit(child, &action);
475        }
476    }
477
478    fn bottom_up(data: &mut Self::Data, children: [Option<&Self::Data>; 2]) {
479        let mut aggregate = L::single_agg(&data.value.key);
480        if let Some(left) = children[0] {
481            aggregate = L::agg_operate(&left.value.agg, &aggregate);
482        }
483        if let Some(right) = children[1] {
484            aggregate = L::agg_operate(&aggregate, &right.value.agg);
485        }
486        data.value.agg = aggregate;
487    }
488
489    fn reverse(data: &mut Self::Data) {
490        L::toggle(&mut data.value.agg);
491    }
492}

539        let mut parent = vec![usize::MAX; adjacency.len()];
540        let mut stack = vec![start];
541        while let Some(u) = stack.pop() {
542            for &v in &adjacency[u] {

546                }

557        }

761    fn run_path_case(graph: &UndirectedSparseGraph, rounds: usize, rng: &mut Xorshift) {
762        let n = graph.vertices_size();
763        let mut adjacency = adjacency(graph);
764        let mut edges = graph.edges.clone();
765        let mut values = (0..n).map(|_| rng.random(-20i64..=20)).collect::<Vec<_>>();
766        let mut tree =
767            PathLinkCutTree::<RangeSumRangeLinear<i64>>::from_edges(values.iter().copied(), &edges);
768        let root = rng.random(0..n);
769        tree.reroot(root);
770        for u in 0..n {
771            assert_eq!(tree.root(u), root);
772        }
773
774        for _ in 0..rounds {
775            match rng.random(0..if edges.is_empty() { 2 } else { 3 }) {
776                0 => {
777                    let u = rng.random(0..n);
778                    if rng.random(0..2) == 0 {
779                        values[u] = rng.random(-20i64..=20);
780                        tree.set(u, values[u]);
781                    } else {
782                        let action = rng.random(-20i64..=20);
783                        values[u] += action;
784                        tree.modify(u, |value| *value + action);
785                    }
786                }
787                1 => {
788                    let u = rng.random(0..n);
789                    let v = rng.random(0..n);
790                    let action = (rng.random(0i64..=1), rng.random(-20i64..=20));
791                    let path = naive_path(&adjacency, u, v);
792                    for &x in &path {
793                        values[x] = action.0 * values[x] + action.1;
794                    }
795                    tree.update_path(u, v, &action);
796                }
797                _ => {
798                    let (u, v, a, b) = rewire(&mut adjacency, &mut edges, rng);
799                    tree.cut(u, v);
800                    assert!(!tree.is_connected(u, v));
801                    tree.link(a, b);
802                    assert!(tree.is_connected(u, v));
803                }
804            }
805
806            let u = rng.random(0..n);
807            let v = rng.random(0..n);
808            let path = naive_path(&adjacency, u, v);
809            assert_eq!(
810                tree.fold_path(u, v),
811                (path.iter().map(|&x| values[x]).sum(), path.len() as i64)
812            );
813            let u = rng.random(0..n);
814            assert_eq!(*tree.get(u), values[u]);
815        }
816    }

818    fn run_ordered_path_case(graph: &UndirectedSparseGraph, rounds: usize, rng: &mut Xorshift) {
819        let n = graph.vertices_size();
820        let mut adjacency = adjacency(graph);
821        let mut edges = graph.edges.clone();
822        let mut values = (0..n)
823            .map(|_| rng.random(b'a'..=b'z') as char)
824            .collect::<Vec<_>>();
825        let mut tree = PathLinkCutTree::<StringPath>::from_edges(values.iter().copied(), &edges);
826
827        for _ in 0..rounds {
828            match rng.random(0..if edges.is_empty() { 2 } else { 3 }) {
829                0 => {
830                    let u = rng.random(0..n);
831                    values[u] = rng.random(b'a'..=b'z') as char;
832                    if rng.random(0..2) == 0 {
833                        tree.set(u, values[u]);
834                    } else {
835                        tree.modify(u, |_| values[u]);
836                    }
837                }
838                1 => {
839                    let u = rng.random(0..n);
840                    tree.reroot(u);
841                    tree.reroot(u);
842                }
843                _ => {
844                    let (u, v, a, b) = rewire(&mut adjacency, &mut edges, rng);
845                    tree.cut(u, v);
846                    tree.link(a, b);
847                }
848            }
849
850            let u = rng.random(0..n);
851            let v = rng.random(0..n);
852            assert_eq!(
853                tree.fold_path(u, v).0,
854                naive_path(&adjacency, u, v)
855                    .into_iter()
856                    .map(|u| values[u])
857                    .collect::<String>()
858            );
859        }
860    }

862    fn run_subtree_case(graph: &UndirectedSparseGraph, rounds: usize, rng: &mut Xorshift) {
863        let n = graph.vertices_size();
864        let mut adjacency = adjacency(graph);
865        let mut edges = graph.edges.clone();
866        let mut values = (0..n).map(|_| rng.random(-20i64..=20)).collect::<Vec<_>>();
867        let mut tree = LinkCutTree::<SubtreeSum>::from_edges(values.iter().copied(), &edges);
868
869        for _ in 0..rounds {
870            match rng.random(0..3) {
871                0 => {
872                    let (u, v, a, b) = rewire(&mut adjacency, &mut edges, rng);
873                    tree.cut(u, v);
874                    tree.link(a, b);
875                }
876                1 => {
877                    let u = rng.random(0..n);
878                    if rng.random(0..2) == 0 {
879                        values[u] = rng.random(-20i64..=20);
880                        tree.set(u, values[u]);
881                    } else {
882                        let action = rng.random(-20i64..=20);
883                        values[u] += action;
884                        tree.modify(u, |value| *value + action);
885                    }
886                }
887                _ => {
888                    let &(u, v) = &edges[rng.random(0..edges.len())];
889                    let (node, parent) = if rng.random(0..2) == 0 {
890                        (u, v)
891                    } else {
892                        (v, u)
893                    };
894                    let subtree = naive_subtree(&adjacency, node, parent);
895                    let action = rng.random(-20i64..=20);
896                    for &x in &subtree {
897                        values[x] += action;
898                    }
899                    tree.update_subtree(node, parent, &action);
900                }
901            }
902
903            let &(u, v) = &edges[rng.random(0..edges.len())];
904            for (node, parent) in [(u, v), (v, u)] {
905                let subtree = naive_subtree(&adjacency, node, parent);
906                assert_eq!(
907                    tree.fold_subtree(node, parent),
908                    subtree.iter().map(|&x| values[x]).sum()
909                );
910            }
911            let u = rng.random(0..n);
912            assert_eq!(*tree.get(u), values[u]);
913        }
914    }
