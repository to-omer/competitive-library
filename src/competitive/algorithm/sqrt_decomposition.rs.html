3pub trait SqrtDecomposition: Sized {
4    type M: Monoid<T = Self::T>;
5    type T: Clone;
6    type B;
7    fn bucket(bsize: usize) -> Self::B;
8    fn update_bucket(bucket: &mut Self::B, x: &Self::T);
9    fn update_cell(bucket: &mut Self::B, cell: &mut Self::T, x: &Self::T);
10    fn fold_bucket(bucket: &Self::B) -> Self::T;
11    fn fold_cell(bucket: &Self::B, cell: &Self::T) -> Self::T;
12    fn sqrt_decomposition(n: usize, bucket_size: usize) -> SqrtDecompositionBuckets<Self> {
13        let mut buckets = vec![];
14        for l in (0..n).step_by(bucket_size) {
15            let bsize = (l + bucket_size).min(n) - l;
16            let x = Self::bucket(bsize);
17            buckets.push((vec![Self::M::unit(); bsize], x));
18        }
19        SqrtDecompositionBuckets {
20            bucket_size,
21            buckets,
22            _marker: std::marker::PhantomData,
23        }
24    }
25}
26
27pub struct SqrtDecompositionBuckets<S>
28where
29    S: SqrtDecomposition,
30{
31    bucket_size: usize,
32    buckets: Vec<(Vec<S::T>, S::B)>,
33    _marker: std::marker::PhantomData<fn() -> S>,
34}
35impl<S> SqrtDecompositionBuckets<S>
36where
37    S: SqrtDecomposition,
38{
39    pub fn update(&mut self, l: usize, r: usize, x: S::T) {
40        for (i, (cells, bucket)) in self.buckets.iter_mut().enumerate() {
41            let s = i * self.bucket_size;
42            let t = s + cells.len();
43            if t <= l || r <= s {
44            } else if l <= s && t <= r {
45                S::update_bucket(bucket, &x);
46            } else {
47                for cell in &mut cells[l.max(s) - s..r.min(t) - s] {
48                    S::update_cell(bucket, cell, &x);
49                }
50            }
51        }
52    }
53    pub fn fold(&self, l: usize, r: usize) -> S::T {
54        let mut res = S::M::unit();
55        for (i, (cells, bucket)) in self.buckets.iter().enumerate() {
56            let s = i * self.bucket_size;
57            let t = s + cells.len();
58            if t <= l || r <= s {
59            } else if l <= s && t <= r {
60                <S::M as Magma>::operate_assign(&mut res, &S::fold_bucket(bucket));
61            } else {
62                for cell in &cells[l.max(s) - s..r.min(t) - s] {
63                    <S::M as Magma>::operate_assign(&mut res, &S::fold_cell(bucket, cell));
64                }
65            }
66        }
67        res
68    }
