2pub fn slide_minimum<T: Clone + Ord>(v: &[T], k: usize) -> Vec<usize> {
3    let mut deq = std::collections::VecDeque::new();
4    let mut res = vec![];
5    for i in 0..v.len() {
6        while deq.back().map(|&j| v[j] >= v[i]).unwrap_or(false) {
7            deq.pop_back();
8        }
9        deq.push_back(i);
10        if i + 1 >= k {
11            let f = *deq.front().unwrap();
12            res.push(f);
13            if f == i + 1 - k {
14                deq.pop_front();
15            }
16        }
17    }
18    res
19}

22pub struct SlideMinimum<'a> {
23    deq: std::collections::VecDeque<usize>,
24    width: usize,
25    left: usize,
26    right: usize,
27    seq: &'a [i64],
28}
29#[codesnip::entry("SlideMinimum")]
30impl<'a> SlideMinimum<'a> {
31    pub fn new(width: usize, seq: &'a [i64]) -> Self {
32        let mut self_ = Self {
33            deq: std::collections::VecDeque::new(),
34            width,
35            left: 0,
36            right: 0,
37            seq,
38        };
39        self_.build();
40        self_
41    }
42    fn build(&mut self) {
43        while self.right + 1 < self.width {
44            self.rsucc();
45        }
46    }
47    fn rsucc(&mut self) {
48        while self
49            .deq
50            .back()
51            .is_some_and(|&v| self.seq[v] >= self.seq[self.right])
52        {
53            self.deq.pop_back();
54        }
55        self.deq.push_back(self.right);
56        self.right += 1;
57    }
58    fn lsucc(&mut self) {
59        if *self.deq.front().unwrap() < self.left {
60            self.deq.pop_front();
61        }
62        self.left += 1;
63    }
64    pub fn next_minimum(&mut self) -> i64 {
65        self.rsucc();
66        self.lsucc();
67        self.seq[*self.deq.front().unwrap()]
68    }
