1use super::Monoid;

3pub struct Doubling<M>
4where
5    M: Monoid,
6{
7    size: usize,
8    table: Vec<(usize, M::T)>,
9}
10
11impl<M> Doubling<M>
12where
13    M: Monoid,
14{
15    pub fn new(size: usize, f: impl Fn(usize) -> (usize, M::T)) -> Self {
16        let mut table = Vec::with_capacity(size);
17        for i in 0..size {
18            table.push(f(i));
19        }
20        Self { size, table }
21    }
22
23    pub fn double(&mut self) {
24        let base = self.table.len() - self.size;
25        for i in 0..self.size {
26            let &(to, ref val) = &self.table[base + i];
27            if to != !0 {
28                let &(to2, ref val2) = &self.table[base + to];
29                self.table.push((to2, M::operate(val, val2)));
30            } else {
31                self.table.push((!0, M::unit()));
32            }
33        }
34    }
35
36    pub fn kth(&mut self, mut pos: usize, mut k: usize) -> (usize, M::T) {
37        let mut x = M::unit();
38        for chunk in self.table.chunks_exact(self.size) {
39            if k & 1 == 1 {
40                let &(to, ref val) = &chunk[pos];
41                if to == !0 {
42                    return (!0, M::unit());
43                }
44                x = M::operate(&x, val);
45                pos = to;
46            }
47            k >>= 1;
48            if k == 0 {
49                break;
50            }
51        }
52        while k > 0 {
53            self.double();
54            if k & 1 == 1 {
55                let base = self.table.len() - self.size;
56                let &(to, ref val) = &self.table[base + pos];
57                if to == !0 {
58                    return (!0, M::unit());
59                }
60                x = M::operate(&x, val);
61                pos = to;
62            }
63            k >>= 1;
64        }
65        (pos, x)
66    }
67
68    /// queries: (pos, k)
69    /// Return: (pos, acc)
70    pub fn kth_multiple(
71        &self,
72        queries: impl IntoIterator<Item = (usize, usize)>,
73    ) -> Vec<(usize, M::T)> {
74        let (mut ks, mut results): (Vec<usize>, Vec<(usize, M::T)>) = queries
75            .into_iter()
76            .map(|(start, k)| (k, (start, M::unit())))
77            .unzip();
78        for chunk in self.table.chunks_exact(self.size) {
79            for (i, k) in ks.iter_mut().enumerate() {
80                if *k & 1 == 1 {
81                    let &(to, ref val) = &chunk[results[i].0];
82                    if to == !0 {
83                        results[i] = (!0, M::unit());
84                        *k = 0;
85                    } else {
86                        results[i].1 = M::operate(&results[i].1, val);
87                        results[i].0 = to;
88                    }
89                }
90                *k >>= 1;
91            }
92        }
93        if ks.iter().any(|&k| k > 0) {
94            let mut dp = self.table[self.table.len() - self.size..].to_vec();
95            while ks.iter().any(|&k| k > 0) {
96                let mut ndp = Vec::with_capacity(dp.len());
97                for i in 0..self.size {
98                    let &(to, ref val) = &dp[i];
99                    if to != !0 {
100                        let &(to2, ref val2) = &dp[to];
101                        ndp.push((to2, M::operate(val, val2)));
102                    } else {
103                        ndp.push((!0, M::unit()));
104                    }
105                }
106                dp = ndp;
107                for (i, k) in ks.iter_mut().enumerate() {
108                    if *k & 1 == 1 {
109                        let &(to, ref val) = &dp[results[i].0];
110                        if to == !0 {
111                            results[i] = (!0, M::unit());
112                            *k = 0;
113                        } else {
114                            results[i].1 = M::operate(&results[i].1, val);
115                            results[i].0 = to;
116                        }
117                    }
118                    *k >>= 1;
119                }
120            }
121        }
122        results
123    }

125    /// Return: (k, (pos, acc))
126    pub fn find_last(
127        &self,
128        mut pos: usize,
129        mut pred: impl FnMut(usize, &M::T) -> bool,
130    ) -> (usize, (usize, M::T)) {
131        let mut k = 0usize;
132        let mut x = M::unit();
133        assert!(pred(pos, &x));
134        for (i, chunk) in self.table.chunks_exact(self.size).enumerate().rev() {
135            let &(to, ref val) = &chunk[pos];
136            let nx = M::operate(&x, val);
137            if pred(to, &nx) {
138                x = nx;
139                pos = to;
140                k |= 1 << i;
141            }
142        }
143        (k, (pos, x))
144    }
145
146    /// Return: (k, (pos, acc))
147    pub fn find_first(
148        &self,
149        pos: usize,
150        mut pred: impl FnMut(usize, &M::T) -> bool,
151    ) -> Option<(usize, (usize, M::T))> {
152        let (mut k, (mut pos, mut x)) = self.find_last(pos, |k, x| !pred(k, x));
153        k += 1;
154        M::operate_assign(&mut x, &self.table[pos].1);
155        pos = self.table[pos].0;
156        if pred(pos, &x) {
157            Some((k, (pos, x)))
158        } else {
159            None
160        }
161    }

173    #[test]
174    fn test_kth() {
175        let mut rng = Xorshift::default();
176        for _ in 0..200 {
177            let n = rng.random(1usize..100);
178            let to: Vec<_> = rng
179                .random_iter(0..=n)
180                .take(n)
181                .map(|x| x.wrapping_sub(1))
182                .collect();
183            let w: Vec<MInt998244353> = rng.random_iter(..).take(n).collect();
184            let mut doubling = Doubling::<AdditiveOperation<_>>::new(n, |i| (to[i], w[i]));
185            let mut queries = vec![];
186            let mut results = vec![];
187            for s in 0..n {
188                let mut pos = s;
189                let mut x = MInt998244353::zero();
190                for k in 0..100 {
191                    if pos == !0 {
192                        assert_eq!(doubling.kth(s, k), (pos, MInt998244353::zero()));
193                        queries.push((s, k));
194                        results.push((pos, MInt998244353::zero()));
195                    } else {
196                        assert_eq!(doubling.kth(s, k), (pos, x));
197                        x += w[pos];
198                        pos = to[pos];
199                    }
200                }
201            }
202            let doubling = Doubling::<AdditiveOperation<_>>::new(n, |i| (to[i], w[i]));
203            assert_eq!(doubling.kth_multiple(queries), results);
204        }
205    }

207    #[test]
208    fn test_find() {
209        let mut rng = Xorshift::default();
210        for _ in 0..200 {
211            let n = rng.random(1usize..100);
212            let to: Vec<_> = rng.random_iter(0..n).take(n).collect();
213            let w: Vec<u64> = rng.random_iter(1..100).take(n).collect();
214            let mut doubling = Doubling::<AdditiveOperation<_>>::new(n, |i| (to[i], w[i]));
215            for _ in 0..10 {
216                doubling.double();
217            }
218            for s in 0..n {
219                let mut k = 0usize;
220                let mut pos = s;
221                let mut acc = 0u64;
222                for x in 0u64..200 {
223                    while acc + w[pos] <= x {
224                        acc += w[pos];
225                        pos = to[pos];
226                        k += 1;
227                    }
228                    assert_eq!(doubling.find_last(s, |_, &v| v <= x), (k, (pos, acc)));
229                    assert_eq!(
230                        doubling.find_first(s, |_, &v| v > x),
231                        Some((k + 1, (to[pos], acc + w[pos])))
232                    );
233                }
234                assert_eq!(doubling.find_first(s, |_, &v| v > 1_000_000), None);
235            }
236        }
