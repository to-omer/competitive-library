53impl<R, Input, Class, FC, FF> EsperEstimator<R, Input, Class, FC, FF>
54where
55    R: Field,
56    R::Additive: Invertible,
57    R::Multiplicative: Invertible,
58    Class: Eq + Hash,
59    FC: Fn(&Input) -> Class,
60    FF: Fn(&Input) -> Vec<R::T>,
61{
62    pub fn new(class: FC, feature: FF) -> Self {
63        Self {
64            class,
65            feature,
66            data: Default::default(),
67            _marker: PhantomData,
68        }
69    }
70
71    pub fn push(&mut self, input: Input, output: R::T) {
72        let class = (self.class)(&input);
73        let feature = (self.feature)(&input);
74        let entry = self.data.entry(class).or_default();
75        entry.a.push(feature);
76        entry.b.push(output);
77    }
78}

80impl<R, Input, Class, FC, FF> EsperEstimator<R, Input, Class, FC, FF>
81where
82    R: Field,
83    R::Additive: Invertible,
84    R::Multiplicative: Invertible,
85    R::T: PartialEq,
86    Class: Eq + Hash,
87    FC: Fn(&Input) -> Class,
88    FF: Fn(&Input) -> Vec<R::T>,
89{
90    pub fn solve(self) -> EsperSolver<R, Input, Class, FC, FF> {
91        let data: HashMap<_, _> = self
92            .data
93            .into_iter()
94            .map(|(key, SystemOfLinearEquation { a, b })| {
95                (
96                    key,
97                    Matrix::<R>::from_vec(a)
98                        .solve_system_of_linear_equations(&b)
99                        .map(|sol| sol.particular),
100                )
101            })
102            .collect();
103        EsperSolver {
104            class: self.class,
105            feature: self.feature,
106            data,
107            _marker: PhantomData,
108        }
109    }
110
111    pub fn solve_checked(self) -> EsperSolver<R, Input, Class, FC, FF>
112    where
113        Class: Debug,
114        R::T: Debug,
115    {
116        let data: HashMap<_, _> = self
117            .data
118            .into_iter()
119            .map(|(key, SystemOfLinearEquation { a, b })| {
120                let mat = Matrix::<R>::from_vec(a);
121                let coeff = mat
122                    .solve_system_of_linear_equations(&b)
123                    .map(|sol| sol.particular);
124                if coeff.is_none() {
125                    eprintln!(
126                        "failed to solve linear equations: key={:?} A={:?} b={:?}",
127                        key, &mat.data, &b
128                    );
129                }
130                (key, coeff)
131            })
132            .collect();
133        EsperSolver {
134            class: self.class,
135            feature: self.feature,
136            data,
137            _marker: PhantomData,
138        }
139    }
140}

142impl<R, Input, Class, FC, FF> EsperSolver<R, Input, Class, FC, FF>
143where
144    R: Field,
145    R::Additive: Invertible,
146    R::Multiplicative: Invertible,
147    Class: Eq + Hash,
148    FC: Fn(&Input) -> Class,
149    FF: Fn(&Input) -> Vec<R::T>,
150{
151    pub fn solve(&self, input: Input) -> R::T {
152        let coeff = self
153            .data
154            .get(&(self.class)(&input))
155            .expect("unrecognized class")
156            .as_ref()
157            .expect("failed to solve");
158        let feature = (self.feature)(&input);
159        feature
160            .into_iter()
161            .zip(coeff)
162            .map(|(x, y)| R::mul(&x, y))
163            .fold(R::zero(), |x, y| R::add(&x, &y))
164    }
