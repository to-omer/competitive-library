40impl<R, Input, Class, FC, FF> EsperEstimator<R, Input, Class, FC, FF>
41where
42    R: Field,
43    R::Additive: Invertible,
44    R::Multiplicative: Invertible,
45    Class: Eq + Hash,
46    FC: Fn(&Input) -> Class,
47    FF: Fn(&Input) -> Vec<R::T>,
48{
49    pub fn new(class: FC, feature: FF) -> Self {
50        Self {
51            class,
52            feature,
53            data: Default::default(),
54            _marker: PhantomData,
55        }
56    }
57
58    pub fn push(&mut self, input: Input, output: R::T) {
59        let class = (self.class)(&input);
60        let feature = (self.feature)(&input);
61        let entry = self.data.entry(class).or_default();
62        entry.0.push(feature);
63        entry.1.push(output);
64    }
65}

67impl<R, Input, Class, FC, FF> EsperEstimator<R, Input, Class, FC, FF>
68where
69    R: Field,
70    R::Additive: Invertible,
71    R::Multiplicative: Invertible,
72    R::T: PartialEq,
73    Class: Eq + Hash,
74    FC: Fn(&Input) -> Class,
75    FF: Fn(&Input) -> Vec<R::T>,
76{
77    pub fn solve(self) -> EsperSolver<R, Input, Class, FC, FF> {
78        let data: HashMap<_, _> = self
79            .data
80            .into_iter()
81            .map(|(key, (a, b))| {
82                (
83                    key,
84                    Matrix::<R>::from_vec(a).solve_system_of_linear_equations(&b),
85                )
86            })
87            .collect();
88        EsperSolver {
89            class: self.class,
90            feature: self.feature,
91            data,
92            _marker: PhantomData,
93        }
94    }
95
96    pub fn solve_checked(self) -> EsperSolver<R, Input, Class, FC, FF>
97    where
98        Class: Debug,
99        R::T: Debug,
100    {
101        let data: HashMap<_, _> = self
102            .data
103            .into_iter()
104            .map(|(key, (a, b))| {
105                let mat = Matrix::<R>::from_vec(a);
106                let coeff = mat.solve_system_of_linear_equations(&b);
107                if coeff.is_none() {
108                    eprintln!(
109                        "failed to solve linear equations: key={:?} A={:?} b={:?}",
110                        key, &mat.data, &b
111                    );
112                }
113                (key, coeff)
114            })
115            .collect();
116        EsperSolver {
117            class: self.class,
118            feature: self.feature,
119            data,
120            _marker: PhantomData,
121        }
122    }
123}

125impl<R, Input, Class, FC, FF> EsperSolver<R, Input, Class, FC, FF>
126where
127    R: Field,
128    R::Additive: Invertible,
129    R::Multiplicative: Invertible,
130    Class: Eq + Hash,
131    FC: Fn(&Input) -> Class,
132    FF: Fn(&Input) -> Vec<R::T>,
133{
134    pub fn solve(&self, input: Input) -> R::T {
135        let coeff = self
136            .data
137            .get(&(self.class)(&input))
138            .expect("unrecognized class")
139            .as_ref()
140            .expect("failed to solve");
141        let feature = (self.feature)(&input);
142        feature
143            .into_iter()
144            .zip(coeff)
145            .map(|(x, y)| R::mul(&x, y))
146            .fold(R::zero(), |x, y| R::add(&x, &y))
147    }
