353    fn new(search: &'a MultipleStringSearch<T>) -> Self {
354        let n = search.search.text.len();
355        let mut prefix = Vec::with_capacity(n);
356        prefix.push(0);
357        let mut total = 0u64;
358        for i in 1..=n {
359            let len = search.suffix_len(i);
360            let prev_len = search.suffix_len(i - 1);
361            let lcp_prev = search.search.lcp_array[i - 1].min(len).min(prev_len);
362            total += (len - lcp_prev) as u64;
363            prefix.push(total);
364        }

366    }

368    pub fn kth_distinct_substring(&self, k: u64) -> Option<(usize, Range<usize>)> {
369        let idx = self.prefix.partition_point(|&x| x <= k);
370        if idx == self.prefix.len() {
371            return None;
372        }
373        let (text_idx, pos) = self.search.position_map[self.search.search.suffix_array[idx]];
374        let len = self.search.suffix_len(idx) - (self.prefix[idx] - k) as usize + 1;
