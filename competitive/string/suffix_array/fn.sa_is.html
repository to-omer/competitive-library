34fn induced_sort(s: &[usize], upper: usize, is_s: &[bool], lms: &[usize]) -> Vec<usize> {
35    let n = s.len();
36    let mut sa = vec![!0usize; n];
37
38    let mut bucket_size = vec![0usize; upper + 1];
39    for &c in s {
40        bucket_size[c] += 1;
41    }
42
43    let mut bucket_tail = vec![0usize; upper + 1];
44    {
45        let mut sum = 0usize;
46        for i in 0..=upper {
47            sum += bucket_size[i];
48            bucket_tail[i] = sum;
49        }
50    }
51    for &pos in lms.iter().rev() {
52        let c = s[pos];
53        bucket_tail[c] -= 1;
54        sa[bucket_tail[c]] = pos;
55    }
56
57    let mut bucket_head = vec![0usize; upper + 1];
58    {
59        let mut sum = 0usize;
60        for i in 0..=upper {
61            bucket_head[i] = sum;
62            sum += bucket_size[i];
63        }
64    }
65    for i in 0..n {
66        let v = sa[i];
67        if v == !0 || v == 0 {
68            continue;
69        }
70        let p = v - 1;
71        if !is_s[p] {
72            let c = s[p];
73            sa[bucket_head[c]] = p;
74            bucket_head[c] += 1;
75        }
76    }
77
78    let mut bucket_tail = vec![0usize; upper + 1];
79    {
80        let mut sum = 0usize;
81        for i in 0..=upper {
82            sum += bucket_size[i];
83            bucket_tail[i] = sum;
84        }
85    }
86    for i in (0..n).rev() {
87        let v = sa[i];
88        if v == !0 || v == 0 {
89            continue;
90        }
91        let p = v - 1;
92        if is_s[p] {
93            let c = s[p];
94            bucket_tail[c] -= 1;
95            sa[bucket_tail[c]] = p;
96        }
97    }
98
99    sa
100}

102fn sa_is(s: &[usize], upper: usize) -> Vec<usize> {
103    let n = s.len();
104    match n {
105        0 => return vec![],
106        1 => return vec![0],
107        _ => {}
108    }
109
110    let mut is_s = vec![false; n];
111    is_s[n - 1] = true;
112    for i in (0..n - 1).rev() {
113        is_s[i] = if s[i] < s[i + 1] {
114            true
115        } else if s[i] > s[i + 1] {
116            false
117        } else {
118            is_s[i + 1]
119        };
120    }
121
122    let mut lms_map = vec![!0; n];
123    let mut lms = vec![];
124    for i in 1..n {
125        if is_s[i] && !is_s[i - 1] {
126            lms_map[i] = lms.len();
127            lms.push(i);
128        }
129    }
130
131    let mut sa = induced_sort(s, upper, &is_s, &lms);
132
133    if lms.len() > 1 {
134        let mut rec_s = vec![0usize; lms.len()];
135        let mut rec_upper = 0usize;
136        let mut prev = !0usize;
137        for &pos in &sa {
138            if pos != !0 && lms_map[pos] != !0 {
139                if prev != !0 {
140                    let mut offset = 0usize;
141                    loop {
142                        if s[prev + offset] != s[pos + offset]
143                            || is_s[prev + offset] != is_s[pos + offset]
144                        {
145                            rec_upper += 1;
146                            break;
147                        }
148                        if offset != 0 {
149                            let prev_is_lms = lms_map[prev + offset] != !0;
150                            let pos_is_lms = lms_map[pos + offset] != !0;
151                            if prev_is_lms || pos_is_lms {
152                                rec_upper += (prev_is_lms != pos_is_lms) as usize;
153                                break;
154                            }
155                        }
156                        offset += 1;
157                    }
158                }
159                rec_s[lms_map[pos]] = rec_upper;
160                prev = pos;
161            }
162        }
163
164        let rec_sa = if rec_upper + 1 == lms.len() {
165            let mut rec_sa = vec![0usize; lms.len()];
166            for i in 0..lms.len() {
167                rec_sa[rec_s[i]] = i;
168            }
169            rec_sa
170        } else {
171            sa_is(&rec_s, rec_upper)
172        };
173
174        let mut ordered_lms = Vec::with_capacity(lms.len());
175        for &i in &rec_sa {
176            ordered_lms.push(lms[i]);
177        }
178        sa = induced_sort(s, upper, &is_s, &ordered_lms);
179    }
180
181    sa
