421        .chunks(block_len)

423        .collect();

425        None
426    } else {

565fn reconstruct_mint_crt<M, N1, N2, N3>(f: (MVec<N1>, MVec<N2>, MVec<N3>)) -> MVec<M>
566where
567    M: MIntConvert + MIntConvert<u32>,
568    N1: Montgomery32NttModulus,
569    N2: Montgomery32NttModulus,
570    N3: Montgomery32NttModulus,
571{
572    let t1 = MInt::<N2>::new(N1::get_mod()).inv();
573    let m1_3 = MInt::<N3>::new(N1::get_mod());
574    let t2 = (m1_3 * MInt::<N3>::new(N2::get_mod())).inv();
575    let modulus = <M as MIntConvert<u32>>::mod_into() as u64;
576    let m1 = N1::get_mod() as u64;
577    let m2 = m1 * N2::get_mod() as u64 % modulus;
578    let fits_u64 = (N1::get_mod() - 1) as u128
579        + (N2::get_mod() - 1) as u128 * m1 as u128
580        + (N3::get_mod() - 1) as u128 * m2 as u128
581        <= u64::MAX as u128;
582    f.0.into_iter()
583        .zip(f.1)
584        .zip(f.2)
585        .map(|((c1, c2), c3)| {
586            let d1 = c1.inner();
587            let d2 = ((c2 - MInt::<N2>::from(d1)) * t1).inner();
588            let x = MInt::<N3>::new(d1) + MInt::<N3>::new(d2) * m1_3;
589            let d3 = ((c3 - x) * t2).inner();
590            let value = if fits_u64 {
591                (d1 as u64 + d2 as u64 * m1 + d3 as u64 * m2) % modulus
592            } else {
593                ((d1 as u128 + d2 as u128 * m1 as u128 + d3 as u128 * m2 as u128) % modulus as u128)
594                    as u64
595            };
596            MInt::<M>::from(value as u32)
597        })
598        .collect()
599}

743    fn split(values: &[u64], factor: u64) -> [Vec<i64>; 4] {
744        let mut result = std::array::from_fn(|_| Vec::with_capacity(values.len()));
745        let mut multiplier = 1u64;
746        for &value in values {
747            let mut value = value.wrapping_mul(multiplier);
748            for part in &mut result {
749                let digit = value as i16;
750                part.push(digit as i64);

752            }
753            multiplier = multiplier.wrapping_mul(factor);

840    fn ntt_doubling(mut f: Self::F) -> Self::F {
841        let n = f.len();
842        let k = n.trailing_zeros() as usize;
843        let mut a = Self::inverse_transform_ntt(f.clone(), n);
844        let mut rot = MInt::<M>::one();
845        let zeta = MInt::<M>::new_unchecked(M::INFO.root[k + 1]);
846        for a in a.iter_mut() {

854    fn even_mul_normal_neg(f: &Self::F, g: &Self::F) -> Self::F {
855        assert_eq!(f.len(), g.len());
856        assert!(f.len().is_power_of_two());
857        assert!(f.len() >= 2);
858        let inv2 = MInt::<M>::from(2).inv();
859        let n = f.len() / 2;
860        (0..n)
861            .map(|i| (f[i << 1] * g[i << 1 | 1] + f[i << 1 | 1] * g[i << 1]) * inv2)
862            .collect()
863    }

866        assert_eq!(f.len(), g.len());
867        assert!(f.len().is_power_of_two());
868        assert!(f.len() >= 2);
869        let mut inv2 = MInt::<M>::from(2).inv();
870        let n = f.len() / 2;
871        let k = f.len().trailing_zeros() as usize;
872        let mut h = vec![MInt::<M>::zero(); n];
873        let w = MInt::<M>::new_unchecked(M::INFO.inv_root[k]);
874        BIT_REVERSE.with(|br| {
875            let br = unsafe { &mut *br.get() };
876            if br.len() < k {
877                br.resize_with(k, Default::default);
878            }
879            let k = k - 1;
880            if br[k].is_empty() {
881                let mut v = vec![0; 1 << k];
882                for i in 0..1 << k {
883                    v[i] = (v[i >> 1] >> 1) | ((i & 1) << k.saturating_sub(1));
884                }
885                br[k] = v;
886            }
887            for &i in &br[k] {
888                h[i] = (f[i << 1] * g[i << 1 | 1] - f[i << 1 | 1] * g[i << 1]) * inv2;
889                inv2 *= w;
890            }
891        });
892        h
893    }

669            .saturating_add(exp_q.saturating_mul(sq as _)) as usize;
670        if shift >= deg {
671            return Self::zeros(deg);
672        }
673        p.truncate(deg - shift);
674        q.truncate(deg - shift);
675
676        let mut f = Self::solve_sparse_differential2(
677            &p,
678            &q,
679            &p,
680            T::from(exp_p),
681            T::from(exp_q),
682            deg - shift,
683        );
684        f *= cp.signed_pow(exp_p) * cq.signed_pow(exp_q);
685        if shift > 0 {
686            f <<= shift;
687        }
688        f.prefix(deg)
689    }

692    pub fn exp_of_div_sparse(&self, q: &Self, deg: usize) -> Self {
693        if deg == 0 {
694            return Self::zero();
695        }
696        let shift_q = q
697            .iter()
698            .position(|value| !value.is_zero())
699            .expect("Zero denominator");
700        let shift_p = self.iter().position(|value| !value.is_zero()).unwrap_or(!0);
701        assert!(shift_p > shift_q);
702
703        let mut p = self >> shift_q;
704        let mut q = q >> shift_q;
705        assert!(!q.coeff(0).is_zero());
706
707        let c = q[0].clone();
708        p /= c.clone();
709        q /= c;
710
711        Self::solve_sparse_differential2(&p, &q, &q, T::one(), -T::one(), deg)
712    }

720    pub fn sqrt(&self, deg: usize) -> Option<Self> {
721        if self[0].is_zero() {
722            if let Some(k) = self.iter().position(|x| !x.is_zero()) {
723                if k % 2 != 0 {
724                    return None;
725                } else if deg > k / 2 {
726                    return Some((self >> k).sqrt(deg - k / 2)? << (k / 2));
727                }
728            }
729        } else {
730            let s = self[0].sqrt_coefficient()?;
731            if self.data.iter().filter(|x| !x.is_zero()).count()
732                <= deg.next_power_of_two().trailing_zeros() as usize * 4
733            {
734                let t = self[0].clone();
735                let mut f = self / t;
736                f = f.pow_sparse1(T::one() / T::from(2usize), deg);
737                f *= s;
738                return Some(f);
739            }
740
741            let mut f = Self::from(s);
742            let inv2 = T::one() / (T::one() + T::one());
743            let mut i = 1;
744            while i < deg {
745                f = (&f + &(self.prefix_ref(i * 2) * f.inv(i * 2))).prefix(i * 2) * &inv2;
746                i *= 2;
747            }
748            f.truncate(deg);
749            return Some(f);
750        }
751        Some(Self::zeros(deg))
752    }

798    /// [x^n] P(x) / Q(x)
799    pub fn bostan_mori(mut self, mut rhs: Self, mut n: usize) -> T
800    where
801        C: NttReuse<T = Vec<T>>,
802    {
803        let mut res = T::zero();
804        rhs.trim_tail_zeros();
805        if self.length() >= rhs.length() {
806            let r = &self / &rhs;
807            if n < r.length() {
808                res = r[n].clone();
809            }
810            self -= r * &rhs;
811            self.trim_tail_zeros();
812        }
813        let k = rhs.length().next_power_of_two();
814        let mut p = C::transform_ntt(self.data, k * 2);
815        let mut q = C::transform_ntt(rhs.data, k * 2);
816        while n > 0 {
817            let t = C::even_mul_normal_neg(&q, &q);
818            p = if n.is_multiple_of(2) {
819                C::even_mul_normal_neg(&p, &q)
820            } else {
821                C::odd_mul_normal_neg(&p, &q)
822            };
823            q = t;
824            n /= 2;
825            if n != 0 {
826                if C::MULTIPLE {
827                    p = C::transform_ntt(C::inverse_transform_ntt(p, k), k * 2);
828                    q = C::transform_ntt(C::inverse_transform_ntt(q, k), k * 2);
829                } else {
830                    p = C::ntt_doubling(p);
831                    q = C::ntt_doubling(q);
832                }
833            }
834        }
835        let p = C::inverse_transform_ntt(p, k);
836        let q = C::inverse_transform_ntt(q, k);
837        res + p[0].clone() / q[0].clone()
838    }
