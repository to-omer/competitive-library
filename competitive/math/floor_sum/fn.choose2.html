22pub fn floor_sum(n: u64, a: u64, b: u64, m: u64) -> u64 {
23    let mut ans = Wrapping(0u64);
24    let (mut n, mut m, mut a, mut b) = (Wrapping(n), m, a, b);

27        if a >= m {
28            let (q, r) = br.div_rem(a);

31        }

36        }

40        }
41        let (q, r) = br.div_rem(y_max);

43        b = r;
44        swap(&mut m, &mut a);
45    }
