291        }
292        let p0_inv = p[0].inv();
293        let mut x = vec![MInt::zero(); n];
294        for p in p.into_iter().skip(1) {
295            let p = -p * p0_inv;
296            for i in 0..n {

298            }
299            b = self.apply(&b);
300        }
301        Some(x)
