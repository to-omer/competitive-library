500    pub fn pow(&self, rhs: usize, deg: usize) -> Self
501    where
502        C: NttReuse<T = Vec<T>>,
503        C::F: Clone,
504    {
505        if rhs == 0 {
506            return Self::from_vec(
507                once(T::one())
508                    .chain(repeat_with(T::zero))
509                    .take(deg)
510                    .collect(),
511            );
512        }
513        if let Some(k) = self.iter().position(|x| !x.is_zero()) {
514            if k >= deg.div_ceil(rhs) {
515                Self::zeros(deg)
516            } else {
517                let deg = deg - k * rhs;
518                let x0 = self[k].clone();
519                let mut f = (self >> k) / &x0;
520                if f.data.iter().filter(|x| !x.is_zero()).count()
521                    <= deg.next_power_of_two().trailing_zeros() as usize * 12
522                {
523                    f = f.pow_sparse1(T::from(rhs), deg);
524                } else {
525                    f = (f.log(deg) * &T::from(rhs)).exp(deg);
526                }
527                f *= x0.pow(rhs);
528                f <<= k * rhs;
529                f
530            }
531        } else {
532            Self::zeros(deg)
533        }

535    fn pow_sparse1(&self, rhs: T, deg: usize) -> Self {
536        debug_assert!(!self[0].is_zero());
537        let pos: Vec<_> = self
538            .data
539            .iter()
540            .enumerate()
541            .skip(1)
542            .filter_map(|(i, x)| if x.is_zero() { None } else { Some(i) })
543            .collect();
544        let mf = T::memorized_factorial(deg);
545        let mut f = Self::zeros(deg);
546        f[0] = T::one();

549            for &j in &pos {

552                }

567    /// solve: $X(QF)'=\alpha P'(QF)+\beta P(Q'F)$ in $O(deg * max(nz(P), nz(Q), nz(X)))$
568    pub fn solve_sparse_differential2(
569        p: &Self,
570        q: &Self,
571        x: &Self,
572        alpha: T,
573        beta: T,
574        deg: usize,
575    ) -> Self {
576        if deg == 0 {
577            return Self::zero();
578        }
579        let collect_sparse = |p: &Self| -> Vec<(usize, T)> {
580            p.iter()
581                .enumerate()
582                .filter(|&(_, x)| !x.is_zero())
583                .map(|(i, x)| (i, x.clone()))
584                .collect()
585        };
586        assert!(q.coeff(0).is_one());
587        assert!(x.coeff(0).is_one());
588        let p = collect_sparse(p);
589        let q = collect_sparse(q);
590        let x = collect_sparse(x);
591        let diff = |p: &[(usize, T)]| -> Vec<(usize, T)> {
592            p.iter()
593                .filter(|&&(i, _)| i > 0)
594                .map(|&(i, ref x)| (i - 1, x.clone() * T::from(i)))
595                .collect()
596        };
597        let dp = diff(&p);
598        let dq = diff(&q);
599
600        let mf = T::memorized_factorial(deg);
601        let mut f = Self::zeros(deg);
602        let mut qf = Self::zeros(deg);
603        let mut dq_f = Self::zeros(deg);
604        let mut d_qf = Self::zeros(deg);
605        f[0] = T::one();
606        for i in 0..deg - 1 {
607            qf[i] = f.sparse_fold(q.iter().cloned(), i);
608            dq_f[i] = f.sparse_fold(dq.iter().cloned(), i);
609            let dp_qf_i = qf.sparse_fold(dp.iter().cloned(), i);
610            let p_dq_f_i = dq_f.sparse_fold(p.iter().cloned(), i);
611            let x_d_qf_i = d_qf.sparse_fold(
612                x.iter()
613                    .map(|&(i, ref x)| (i, x.clone() - T::from((i == 0) as usize))),
614                i,
615            );
616            d_qf[i] = alpha.clone() * dp_qf_i + beta.clone() * p_dq_f_i - x_d_qf_i;
617
618            let mut f_ip1 = d_qf[i].clone();
619            for &(j, ref q) in q.iter().take_while(|&&(j, _)| j <= i) {
620                if j > 0 {
621                    f_ip1 -= q.clone() * &f[i - (j - 1)] * T::from(i - (j - 1));
622                }
623            }
624            f[i + 1] = f_ip1 * T::memorized_inv(&mf, i + 1);
625        }
626        f
627    }

720    pub fn sqrt(&self, deg: usize) -> Option<Self> {
721        if self[0].is_zero() {
722            if let Some(k) = self.iter().position(|x| !x.is_zero()) {
723                if k % 2 != 0 {
724                    return None;
725                } else if deg > k / 2 {
726                    return Some((self >> k).sqrt(deg - k / 2)? << (k / 2));
727                }
728            }
729        } else {
730            let s = self[0].sqrt_coefficient()?;
731            if self.data.iter().filter(|x| !x.is_zero()).count()
732                <= deg.next_power_of_two().trailing_zeros() as usize * 4
733            {
734                let t = self[0].clone();
735                let mut f = self / t;
736                f = f.pow_sparse1(T::one() / T::from(2usize), deg);
737                f *= s;
738                return Some(f);
739            }
740
741            let mut f = Self::from(s);
742            let inv2 = T::one() / (T::one() + T::one());
743            let mut i = 1;
744            while i < deg {
745                f = (&f + &(self.prefix_ref(i * 2) * f.inv(i * 2))).prefix(i * 2) * &inv2;
746                i *= 2;
747            }
748            f.truncate(deg);
749            return Some(f);
750        }
751        Some(Self::zeros(deg))
752    }

782    where
783        F: FnMut(usize) -> T,

798    /// [x^n] P(x) / Q(x)
799    pub fn bostan_mori(mut self, mut rhs: Self, mut n: usize) -> T
800    where
801        C: NttReuse<T = Vec<T>>,
802    {
803        let mut res = T::zero();
804        rhs.trim_tail_zeros();
805        if self.length() >= rhs.length() {
806            let r = &self / &rhs;
807            if n < r.length() {
808                res = r[n].clone();
809            }
810            self -= r * &rhs;
811            self.trim_tail_zeros();
812        }
813        let k = rhs.length().next_power_of_two();
814        let mut p = C::transform_ntt(self.data, k * 2);
815        let mut q = C::transform_ntt(rhs.data, k * 2);
816        while n > 0 {
817            let t = C::even_mul_normal_neg(&q, &q);
818            p = if n.is_multiple_of(2) {
819                C::even_mul_normal_neg(&p, &q)
820            } else {
821                C::odd_mul_normal_neg(&p, &q)
822            };
823            q = t;
824            n /= 2;
825            if n != 0 {
826                if C::MULTIPLE {
827                    p = C::transform_ntt(C::inverse_transform_ntt(p, k), k * 2);
828                    q = C::transform_ntt(C::inverse_transform_ntt(q, k), k * 2);
829                } else {
830                    p = C::ntt_doubling(p);
831                    q = C::ntt_doubling(q);
832                }
833            }
834        }
835        let p = C::inverse_transform_ntt(p, k);
836        let q = C::inverse_transform_ntt(q, k);
837        res + p[0].clone() / q[0].clone()
838    }
839    /// return F(x) where [x^n] P(x) / Q(x) = [x^d-1] P(x) F(x)
840    pub fn bostan_mori_msb(self, n: usize) -> Self {
841        let d = self.length() - 1;
842        if n == 0 {
843            return (Self::one() << (d - 1)) / self[0].clone();
844        }
845        let q = self;
846        let mq = q.clone().parity_inversion();
847        let w = (q * &mq).even().bostan_mori_msb(n / 2);
848        let mut s = Self::zeros(w.length() * 2 - (n % 2));
849        for (i, x) in w.iter().enumerate() {
850            s[i * 2 + (1 - n % 2)] = x.clone();
851        }
852        let len = 2 * d + 1;
853        let ts = C::transform(s.prefix(len).data, len);
854        mq.reversed().middle_product(&ts, len).prefix(d + 1)
855    }

871    pub fn multipoint_evaluation(self, points: &[T]) -> Vec<T> {
872        let n = points.len();
873        if n <= 32 {
874            return points.iter().map(|p| self.eval(p.clone())).collect();
875        }
876        let mut subproduct_tree = Vec::with_capacity(n * 2);
877        subproduct_tree.resize_with(n, Zero::zero);
878        for x in points {
879            subproduct_tree.push(Self::from_vec(vec![-x.clone(), T::one()]));
880        }
881        for i in (1..n).rev() {
882            subproduct_tree[i] = &subproduct_tree[i * 2] * &subproduct_tree[i * 2 + 1];
883        }
884        let mut uptree_t = Vec::with_capacity(n * 2);
885        uptree_t.resize_with(1, Zero::zero);
886        subproduct_tree.reverse();
887        subproduct_tree.pop();
888        let m = self.length();
889        let v = subproduct_tree.pop().unwrap().reversed().resized(m);
890        let s = C::transform(self.data, m * 2);
891        uptree_t.push(v.inv(m).middle_product(&s, m * 2).resized(n).reversed());
892        for i in 1..n {
893            let subl = subproduct_tree.pop().unwrap();
894            let subr = subproduct_tree.pop().unwrap();
895            let (dl, dr) = (subl.length(), subr.length());
896            let len = dl.max(dr) + uptree_t[i].length();
897            let s = C::transform(uptree_t[i].data.to_vec(), len);
898            uptree_t.push(subr.middle_product(&s, len).prefix(dl));
899            uptree_t.push(subl.middle_product(&s, len).prefix(dr));
900        }
901        uptree_t[n..]
902            .iter()
903            .map(|u| u.data.first().cloned().unwrap_or_else(Zero::zero))
904            .collect()
905    }
906    pub fn product_all<I>(iter: I, deg: usize) -> Self
907    where
908        I: IntoIterator<Item = Self>,
909    {
910        let mut heap: BinaryHeap<_> = iter
911            .into_iter()
912            .map(|f| PartialIgnoredOrd(Reverse(f.length()), f))
913            .collect();
914        while let Some(PartialIgnoredOrd(_, x)) = heap.pop() {
915            if let Some(PartialIgnoredOrd(_, y)) = heap.pop() {
916                let z = (x * y).prefix(deg);
917                heap.push(PartialIgnoredOrd(Reverse(z.length()), z));
918            } else {
919                return x;
920            }
921        }
922        Self::one()
923    }
924    pub fn sum_all_rational<I>(iter: I, deg: usize) -> (Self, Self)
925    where
926        I: IntoIterator<Item = (Self, Self)>,
927    {
928        let mut heap: BinaryHeap<_> = iter
929            .into_iter()
930            .map(|(f, g)| PartialIgnoredOrd(Reverse(f.length().max(g.length())), (f, g)))
931            .collect();
932        while let Some(PartialIgnoredOrd(_, (xa, xb))) = heap.pop() {
933            if let Some(PartialIgnoredOrd(_, (ya, yb))) = heap.pop() {
934                let zb = (&xb * &yb).prefix(deg);
935                let za = (xa * yb + ya * xb).prefix(deg);
936                heap.push(PartialIgnoredOrd(
937                    Reverse(za.length().max(zb.length())),
938                    (za, zb),
939                ));
940            } else {
941                return (xa, xb);
942            }
943        }
944        (Self::zero(), Self::one())
945    }

998    pub fn power_projection(&self, w: &[T], m: usize) -> Self
999    where
1000        C: NttReuse<T = Vec<T>>,
1001    {
1002        if w.is_empty() {
1003            return Self::zeros(m);
1004        }
1005        if m <= 1 {
1006            return Self::from_vec(vec![w[0].clone(); m]);
1007        }
1008
1009        let n0 = w.len();
1010        let mut n = n0.next_power_of_two();
1011        let mut f = self.prefix_ref(n);
1012        f.resize(n);
1013
1014        let base = n * 2;
1015        let mut p_flat = vec![T::zero(); base];
1016        for (i, wi) in w.iter().enumerate() {
1017            p_flat[n - 1 - i] = wi.clone();
1018        }
1019        let mut q_flat = vec![T::zero(); base * 2];
1020        q_flat[0] = T::one();
1021        let q_offset = base;
1022        for (i, fi) in f.iter().enumerate() {
1023            q_flat[q_offset + i] = -fi.clone();
1024        }
1025        let mut py = 1usize;
1026        let mut qy = 2usize;
1027
1028        let y_limit = m;
1029        while n > 1 {
1030            let base = n * 2;
1031            let len_p = base * py;
1032            let len_q = base * qy;
1033            let len = (len_p + len_q - 1).max(len_q + len_q - 1);
1034            let len_pot = len.max(1).next_power_of_two();
1035            let half = len_pot / 2;
1036
1037            let p_fft = C::transform_ntt(p_flat, len);
1038            let q_fft = C::transform_ntt(q_flat, len);
1039            let pr_fft = C::odd_mul_normal_neg(&p_fft, &q_fft);
1040            let qr_fft = C::even_mul_normal_neg(&q_fft, &q_fft);
1041            let mut pr = C::inverse_transform_ntt(pr_fft, half);
1042            let mut qr = C::inverse_transform_ntt(qr_fft, half);
1043
1044            let new_py = (py + qy - 1).min(y_limit);
1045            let new_qy = (qy + qy - 1).min(y_limit);
1046            let new_base = n;
1047            let need_p = new_base * new_py;
1048            if pr.len() < need_p {
1049                pr.resize_with(need_p, T::zero);
1050            } else if pr.len() > need_p {
1051                pr.truncate(need_p);
1052            }
1053            let need_q = new_base * new_qy;
1054            if qr.len() < need_q {
1055                qr.resize_with(need_q, T::zero);
1056            } else if qr.len() > need_q {
1057                qr.truncate(need_q);
1058            }
1059
1060            let n2 = n / 2;
1061            if n2 < new_base {
1062                for y in 0..new_py {
1063                    let start = y * new_base + n2;
1064                    let end = y * new_base + new_base;
1065                    for v in pr[start..end].iter_mut() {
1066                        *v = T::zero();
1067                    }
1068                }
1069                for y in 0..new_qy {
1070                    let start = y * new_base + n2;
1071                    let end = y * new_base + new_base;
1072                    for v in qr[start..end].iter_mut() {
1073                        *v = T::zero();
1074                    }
1075                }
1076            }
1077
1078            p_flat = pr;
1079            q_flat = qr;
1080            py = new_py;
1081            qy = new_qy;
1082            n = n2;
1083        }
1084
1085        let base = 2;
1086        let mut p_y = Vec::with_capacity(py);
1087        for y in 0..py {
1088            p_y.push(p_flat[base * y].clone());
1089        }
1090        let mut q_y = Vec::with_capacity(qy);
1091        for y in 0..qy {
1092            q_y.push(q_flat[base * y].clone());
1093        }
1094        (Self::from_vec(p_y) * Self::from_vec(q_y).inv(m)).prefix(m)
1095    }

1098    where
1099        C: NttReuse<T = Vec<T>>,
1100        C::F: Clone,
1101    {
1102        if deg == 0 {
1103            return Self::zero();
1104        }
1105        if deg == 1 {
1106            return Self::from_vec(vec![T::zero()]);
1107        }
1108        debug_assert!(self[0].is_zero());
1109        debug_assert!(!self[1].is_zero());
1110
1111        let mut f = self.prefix_ref(deg);
1112        f.resize(deg);
1113        let c = f[1].clone();
1114        f /= c.clone();
1115

1117        w[deg - 1] = T::one();
1118        let s = f.power_projection(&w, deg);
1119
1120        let n = deg - 1;
1121        let n_t = T::from(n);
1122        let mut h = vec![T::zero(); n];
1123        for i in 1..=n {
1124            h[n - i] = s[i].clone() * &n_t / T::from(i);
1125        }
1126
1127        let h_fps = Self::from_vec(h);
1128        let inv_n = T::one() / n_t;
1129        let mut t = h_fps.log(n);
1130        t *= -inv_n;
1131        let g_over_x = t.exp(n);
1132        let mut g = (g_over_x << 1).prefix(deg);
1133
1134        let inv_c = T::one() / c;
1135        let mut pow = T::one();
1136        for coef in g.iter_mut() {
1137            *coef *= pow.clone();
1138            pow *= inv_c.clone();
1139        }
1140        g
