325unsafe fn real_fft_soa(a: &mut Vec<Complex4>) {
326    let n = a.len() * 8;
327    fft_soa(a);
328    bit_reverse_soa(a);
329    let z = a[0].get(0);
330    a[0].set(0, Complex::new(z.re + z.im, z.re - z.im));
331    let middle = n / 4;
332    let z = a[middle >> 2].get(middle & 3).conjugate();
333    a[middle >> 2].set(middle & 3, z);
334    RotateCache::ensure(n / 2);
335    RotateCache::with(|cache| {
336        let shift = usize::BITS - (n / 2).trailing_zeros();
337        for k in 1..n / 4 {
338            let j = n / 2 - k;
339            let x = a[k >> 2].get(k & 3);
340            let y = a[j >> 2].get(j & 3);
341            let w = cache[k.reverse_bits() >> shift];
342            let c = w.conjugate().transpose() + 1.0;
343            let d = c * (x - y.conjugate()) * 0.5;
344            a[k >> 2].set(k & 3, x - d);
345            a[j >> 2].set(j & 3, y + d.conjugate());
346        }
347    });
348}

376fn split_coefficients<M>(
377    values: Vec<MInt<M>>,
378    blocks: usize,
379    modulus: i64,
380    split: i64,
381) -> (Vec<Complex4>, Vec<Complex4>)
382where
383    M: MIntConvert + MIntConvert<u32>,
384{
385    let mut low = vec![Complex4::default(); blocks];
386    let mut high = low.clone();
387    for (i, x) in values.into_iter().enumerate() {
388        let mut x = <M as MIntConvert<u32>>::into(x.inner()) as i64;
389        if x * 2 > modulus {
390            x -= modulus;
391        }
392        let upper = (x as f64 / split as f64).round() as i64;
393        let lane = (i >> 1) & 3;
394        let (low, high) = if i & 1 == 0 {
395            (&mut low[i >> 3].re, &mut high[i >> 3].re)
396        } else {
397            (&mut low[i >> 3].im, &mut high[i >> 3].im)
398        };
399        low[lane] = (x - upper * split) as f64;
400        high[lane] = upper as f64;
401    }
402    (low, high)
403}

408unsafe fn convolve_mint_avx2<M>(a: Vec<MInt<M>>, b: Vec<MInt<M>>) -> Vec<MInt<M>>
409where
410    M: MIntConvert + MIntConvert<u32>,
411{
412    let len = a.len() + b.len() - 1;
413    let n = len.next_power_of_two();
414    let modulus = <M as MIntConvert<u32>>::mod_into() as i64;
415    let split = (modulus as f64).sqrt() as i64 + 1;
416    let (mut a0, mut a1) = split_coefficients(a, n / 8, modulus, split);
417    let (mut b0, mut b1) = split_coefficients(b, n / 8, modulus, split);
418    real_fft_soa(&mut a0);
419    real_fft_soa(&mut a1);
420    real_fft_soa(&mut b0);
421    real_fft_soa(&mut b1);
422
423    let a00 = a0[0].get(0);
424    let a10 = a1[0].get(0);
425    let b00 = b0[0].get(0);
426    let b10 = b1[0].get(0);
427    for i in 0..a0.len() {
428        let (a0r, a0i) = load4(&a0[i]);
429        let (a1r, a1i) = load4(&a1[i]);
430        let (b0r, b0i) = load4(&b0[i]);
431        let (b1r, b1i) = load4(&b1[i]);
432        let (c0r, c0i) = mul4(a0r, a0i, b0r, b0i);
433        let (c01r, c01i) = mul4(a0r, a0i, b1r, b1i);
434        let (c10r, c10i) = mul4(a1r, a1i, b0r, b0i);
435        let (c2r, c2i) = mul4(a1r, a1i, b1r, b1i);
436        store4(&mut a0[i], c0r, c0i);
437        store4(
438            &mut a1[i],
439            _mm256_add_pd(c01r, c10r),
440            _mm256_add_pd(c01i, c10i),
441        );
442        store4(&mut b0[i], c2r, c2i);
443    }
444    a0[0].set(0, Complex::new(a00.re * b00.re, a00.im * b00.im));
445    a1[0].set(
446        0,
447        Complex::new(
448            a00.re * b10.re + a10.re * b00.re,
449            a00.im * b10.im + a10.im * b00.im,
450        ),
451    );
452    b0[0].set(0, Complex::new(a10.re * b10.re, a10.im * b10.im));
453    real_ifft_soa(&mut a0);
454    real_ifft_soa(&mut a1);
455    real_ifft_soa(&mut b0);
456    let inv_n = 2.0 / n as f64;
457    let modulus = modulus as u64;
458    let split = split as u64;
459    let split2 = split * split % modulus;
460    let scale = _mm256_set1_pd(inv_n);
461    let mut result = Vec::with_capacity(len);
462    for block in 0..a0.len() {
463        let values = [
464            [round4(&a0[block].re, scale), round4(&a0[block].im, scale)],
465            [round4(&a1[block].re, scale), round4(&a1[block].im, scale)],
466            [round4(&b0[block].re, scale), round4(&b0[block].im, scale)],
467        ];
468        for lane in 0..4 {
469            for part in 0..2 {
470                if result.len() == len {
471                    break;
472                }
473                let value = (values[0][part][lane].rem_euclid(modulus as i64) as u64
474                    + values[1][part][lane].rem_euclid(modulus as i64) as u64 * split
475                    + values[2][part][lane].rem_euclid(modulus as i64) as u64 * split2)
476                    % modulus;
477                result.push(MInt::<M>::from(value as u32));
478            }
479        }
480    }
481    result
482}
