398unsafe fn split_u64_coefficients(values: &[u64], n: usize, factor: u64) -> [Vec<Complex4>; 4] {
399    let mut result = std::array::from_fn(|_| vec![Complex4::default(); n / 4]);
400    let mut multiplier = 1u64;
401    for (i, &value) in values.iter().enumerate() {
402        let mut value = value.wrapping_mul(multiplier);
403        let (i, imag) = if i < n { (i, false) } else { (i - n, true) };
404        for part in &mut result {
405            let digit = value as i16;
406            value = (value >> 16).wrapping_add(u64::from(digit < 0));
407            if imag {
408                part[i >> 2].im[i & 3] = digit as f64;
409            } else {
410                part[i >> 2].re[i & 3] = digit as f64;
411            }
412        }
413        multiplier = multiplier.wrapping_mul(factor);
414    }
415    result
416}

419unsafe fn dot_u64_soa(a: &mut [Vec<Complex4>; 4], b: &[Vec<Complex4>; 4]) {
420    let n = a[0].len() * 4;
421    RotateCache::ensure(n / 2);
422    RotateCache::with(|cache| {
423        for block in 0..a[0].len() {
424            let mut br = [_mm256_setzero_pd(); 4];
425            let mut bi = br;
426            let mut rr = br;
427            let mut ri = br;
428            for part in 0..4 {
429                (br[part], bi[part]) = load4(&b[part][block]);
430            }
431            let w = eval_twiddle(cache, 1, a[0].len(), block);
432            let wr = _mm256_setr_pd(w.re, 1.0, 1.0, 1.0);
433            let wi = _mm256_setr_pd(w.im, 0.0, 0.0, 0.0);
434            for lane in 0..4 {
435                let ar: [__m256d; 4] =
436                    std::array::from_fn(|part| _mm256_set1_pd(a[part][block].re[lane]));
437                let ai: [__m256d; 4] =
438                    std::array::from_fn(|part| _mm256_set1_pd(a[part][block].im[lane]));
439                for part in 0..4 {
440                    for left in 0..=part {
441                        multiply_accumulate4(
442                            &mut rr[part],
443                            &mut ri[part],
444                            ar[left],
445                            ai[left],
446                            br[part - left],
447                            bi[part - left],
448                        );
449                    }
450                }
451                if lane != 3 {
452                    for part in 0..4 {
453                        br[part] = _mm256_permute4x64_pd::<0x93>(br[part]);
454                        bi[part] = _mm256_permute4x64_pd::<0x93>(bi[part]);
455                        (br[part], bi[part]) = mul4(br[part], bi[part], wr, wi);
456                    }
457                }
458            }
459            for part in 0..4 {
460                store4(&mut a[part][block], rr[part], ri[part]);
461            }
462        }
463    });
464}

466#[target_feature(enable = "avx2,fma")]
467pub unsafe fn convolve_u64_avx2(a: Vec<u64>, b: Vec<u64>, factor: u64, inverse: u64) -> Vec<u64> {
468    let len = a.len() + b.len() - 1;
469    let n = len.next_power_of_two() / 2;
470    let mut fa = split_u64_coefficients(&a, n, factor);
471    drop(a);
472    let mut fb = split_u64_coefficients(&b, n, factor);
473    drop(b);
474    for part in 0..4 {
475        fft_soa(&mut fa[part]);
476        fft_soa(&mut fb[part]);
477    }
478    dot_u64_soa(&mut fa, &fb);
479    drop(fb);
480    for part in &mut fa {
481        ifft_soa(part);
482    }
483    let mut result = vec![0; len];
484    let mut real_multiplier = 1u64;
485    let mut imag_multiplier = inverse.wrapping_pow(n as u32);
486    for (block, _) in fa[0].iter().enumerate() {
487        let real: [[i64; 4]; 4] = std::array::from_fn(|part| round4(&fa[part][block].re));
488        let imag: [[i64; 4]; 4] = std::array::from_fn(|part| round4(&fa[part][block].im));
489        for lane in 0..4 {
490            let i = block * 4 + lane;
491            if i < len {
492                result[i] = (real[0][lane] as u64)
493                    .wrapping_add((real[1][lane] as u64) << 16)
494                    .wrapping_add((real[2][lane] as u64) << 32)
495                    .wrapping_add((real[3][lane] as u64) << 48)
496                    .wrapping_mul(real_multiplier);
497            }
498            real_multiplier = real_multiplier.wrapping_mul(inverse);
499            if i + n < len {
500                result[i + n] = (imag[0][lane] as u64)
501                    .wrapping_add((imag[1][lane] as u64) << 16)
502                    .wrapping_add((imag[2][lane] as u64) << 32)
503                    .wrapping_add((imag[3][lane] as u64) << 48)
504                    .wrapping_mul(imag_multiplier);
505            }
506            imag_multiplier = imag_multiplier.wrapping_mul(inverse);
507        }
508    }
509    result
510}

512#[target_feature(enable = "avx2,fma")]
513pub unsafe fn convolve_mint_avx2<M>(a: Vec<MInt<M>>, b: Vec<MInt<M>>) -> Vec<MInt<M>>
514where
515    M: MIntConvert + MIntConvert<u32>,
516{
517    let len = a.len() + b.len() - 1;
518    let n = len.next_power_of_two() / 2;
519    let modulus = <M as MIntConvert<u32>>::mod_into() as i64;
520    let split = (modulus as f64).sqrt() as i64 + 1;
521    let (mut a0, mut a1) = split_coefficients(a, n, modulus, split);
522    let (mut b0, mut b1) = split_coefficients(b, n, modulus, split);
523    fft_soa(&mut a0);
524    fft_soa(&mut a1);
525    fft_soa(&mut b0);
526    fft_soa(&mut b1);
527    dot_soa(&mut a0, &mut a1, &mut b0, &b1);
528    drop(b1);
529    ifft_soa(&mut a0);
530    ifft_soa(&mut a1);
531    ifft_soa(&mut b0);
532    let modulus = modulus as u64;
533    let split = split as u64;
534    let split2 = split * split % modulus;
535    let mut result = vec![MInt::<M>::from(0u32); len];
536    for (block, ((a0, a1), b0)) in a0.iter().zip(&a1).zip(&b0).enumerate() {
537        let values = [
538            [round4(&a0.re), round4(&a1.re), round4(&b0.re)],
539            [round4(&a0.im), round4(&a1.im), round4(&b0.im)],
540        ];
541        for (part, values) in values.into_iter().enumerate() {
542            for (lane, value0) in values[0].into_iter().enumerate() {
543                let i = block * 4 + lane + part * n;
544                if i < len {
545                    let value = (value0.rem_euclid(modulus as i64) as u64
546                        + values[1][lane].rem_euclid(modulus as i64) as u64 * split
547                        + values[2][lane].rem_euclid(modulus as i64) as u64 * split2)
548                        % modulus;
549                    result[i] = MInt::<M>::from(value as u32);
550                }
551            }
552        }
553    }
554    result
