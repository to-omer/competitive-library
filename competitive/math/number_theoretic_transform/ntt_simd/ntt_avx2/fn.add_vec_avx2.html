295            let y1 = sub_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
296            _mm256_storeu_si256(a.as_mut_ptr().add(i) as *mut __m256i, y0);
297            _mm256_storeu_si256(a.as_mut_ptr().add(half + i) as *mut __m256i, y1);
298            i += 8;
299        }
300        while i < half {
301            let x0 = a[i];
302            let x1 = a[half + i];
303            a[i] = M::mod_add(x0, x1);
304            a[half + i] = M::mod_sub(x0, x1);
305            i += 1;
306        }
307    }
308    let inv = M::mod_inv(<M as MIntConvert<u32>>::from(n as u32));
309    let inv_vec = _mm256_set1_epi32(inv as i32);
310    let mut i = 0;
311    while i + 8 <= n {
312        let x = _mm256_loadu_si256(a.as_ptr().add(i) as *const __m256i);
313        let y = simd32::montgomery_mul_256_canon(x, inv_vec, r_vec, mod_vec);
314        _mm256_storeu_si256(a.as_mut_ptr().add(i) as *mut __m256i, y);
315        i += 8;
316    }
317    while i < n {
318        a[i] = M::mod_mul(a[i], inv);
319        i += 1;
320    }
321}
