363fn convolve_karatsuba<T>(a: &[T], b: &[T]) -> Vec<T>
364where
365    T: Copy + Zero + AddAssign<T> + SubAssign<T> + Mul<Output = T>,
366{
367    if a.len().min(b.len()) <= 30 {
368        return convolve_naive(a, b);
369    }
370    let m = a.len().max(b.len()).div_ceil(2);
371    let (a0, a1) = if a.len() <= m {
372        (a, &[][..])
373    } else {
374        a.split_at(m)
375    };
376    let (b0, b1) = if b.len() <= m {
377        (b, &[][..])
378    } else {
379        b.split_at(m)
380    };
381    let f00 = convolve_karatsuba(a0, b0);
382    let f11 = convolve_karatsuba(a1, b1);
383    let mut a0a1 = a0.to_vec();
384    for (a0a1, &a1) in a0a1.iter_mut().zip(a1) {
385        *a0a1 += a1;
386    }
387    let mut b0b1 = b0.to_vec();
388    for (b0b1, &b1) in b0b1.iter_mut().zip(b1) {
389        *b0b1 += b1;
390    }
391    let mut f01 = convolve_karatsuba(&a0a1, &b0b1);
392    for (f01, &f00) in f01.iter_mut().zip(&f00) {
393        *f01 -= f00;
394    }
395    for (f01, &f11) in f01.iter_mut().zip(&f11) {
396        *f01 -= f11;
397    }
398    let mut c = vec![T::zero(); a.len() + b.len() - 1];
399    for (c, &f00) in c.iter_mut().zip(&f00) {
400        *c += f00;
401    }
402    for (c, &f01) in c[m..].iter_mut().zip(&f01) {
403        *c += f01;
404    }
405    for (c, &f11) in c[m << 1..].iter_mut().zip(&f11) {
406        *c += f11;
407    }
408    c
409}

412fn convolve_large_ntt<M>(a: Vec<MInt<M>>, b: Vec<MInt<M>>) -> Vec<MInt<M>>
413where
414    M: Montgomery32NttModulus,
415{
416    let len = a.len() + b.len() - 1;
417    let ntt_len = 1usize << M::RANK;
418    let block_len = ntt_len / 2;
419    let same = a == b;
420    let fa: Vec<_> = a
421        .chunks(block_len)
422        .map(|a| Convolve::<M>::transform_ntt(a.to_vec(), ntt_len))
423        .collect();
424    let fb: Option<Vec<_>> = if same {
425        None
426    } else {
427        Some(
428            b.chunks(block_len)
429                .map(|b| Convolve::<M>::transform_ntt(b.to_vec(), ntt_len))
430                .collect(),
431        )
432    };
433    let b_blocks = fb.as_ref().map_or(fa.len(), Vec::len);
434    let mut result = vec![MInt::<M>::zero(); len];
435    for diagonal in 0..fa.len() + b_blocks - 1 {
436        let mut spectrum = vec![MInt::<M>::zero(); ntt_len];
437        let start = diagonal.saturating_sub(b_blocks - 1);
438        for i in start..=diagonal.min(fa.len() - 1) {
439            let j = diagonal - i;
440            let g = if let Some(fb) = &fb { &fb[j] } else { &fa[j] };
441            pointwise_multiply_add(&mut spectrum, &fa[i], g);
442        }
443        spectrum = Convolve::<M>::inverse_transform_ntt(spectrum, ntt_len);
444        let offset = diagonal * block_len;
445        for (result, value) in result[offset..].iter_mut().zip(spectrum) {
446            *result += value;
447        }
448    }
449    result
450}

452impl<M> ConvolveSteps for Convolve<M>
453where
454    M: Montgomery32NttModulus,
455{
456    type T = Vec<MInt<M>>;
457    type F = Vec<MInt<M>>;
458    fn length(t: &Self::T) -> usize {
459        t.len()
460    }
461    fn transform(mut t: Self::T, len: usize) -> Self::F {
462        t.resize_with(len.max(1).next_power_of_two(), Zero::zero);
463        #[cfg(target_arch = "x86_64")]
464        if use_block_ntt::<M>(t.len()) {
465            unsafe { ntt_simd::transform_blocks_avx2(&mut t) };
466            return t;
467        }
468        ntt(&mut t);
469        t
470    }
471    fn inverse_transform(mut f: Self::F, len: usize) -> Self::T {
472        #[cfg(target_arch = "x86_64")]
473        if use_block_ntt::<M>(f.len()) {
474            unsafe { ntt_simd::inverse_transform_blocks_avx2(&mut f) };
475            f.truncate(len);
476            return f;
477        }
478        intt(&mut f);
479        f.truncate(len);
480        f
481    }
482    fn multiply(f: &mut Self::F, g: &Self::F) {
483        assert_eq!(f.len(), g.len());
484        #[cfg(target_arch = "x86_64")]
485        if use_block_ntt::<M>(f.len()) {
486            unsafe { ntt_simd::multiply_blocks_avx2(f, g) };
487            return;
488        }
489        pointwise_multiply(f, g);
490    }
491    fn convolve(mut a: Self::T, mut b: Self::T) -> Self::T {
492        if Self::length(&a).max(Self::length(&b)) <= 100 {
493            return convolve_karatsuba(&a, &b);
494        }
495        if Self::length(&a).min(Self::length(&b)) <= 60 {
496            return convolve_naive(&a, &b);
497        }
498        let len = (Self::length(&a) + Self::length(&b)).saturating_sub(1);
499        let size = len.max(1).next_power_of_two();
500        if size > 1usize << M::RANK {
501            return convolve_large_ntt(a, b);
502        }
503        if len <= size / 2 + 2 {
504            let xa = a.pop().unwrap();
505            let xb = b.pop().unwrap();
506            let mut c = vec![MInt::<M>::zero(); len];
507            *c.last_mut().unwrap() = xa * xb;
508            for (a, c) in a.iter().zip(&mut c[b.len()..]) {
509                *c += *a * xb;
510            }
511            for (b, c) in b.iter().zip(&mut c[a.len()..]) {
512                *c += *b * xa;
513            }
514            let d = Self::convolve(a, b);
515            for (d, c) in d.into_iter().zip(&mut c) {
516                *c += d;
517            }
518            return c;
519        }
520        let same = a == b;
521        #[cfg(target_arch = "x86_64")]
522        if use_block_ntt::<M>(size) {
523            a.resize_with(size, Zero::zero);
524            b.resize_with(size, Zero::zero);
525            unsafe { ntt_simd::convolve_blocks_avx2(&mut a, &mut b, same) };
526            a.truncate(len);
527            return a;
528        }
529        let mut a = Self::transform(a, len);
530        if same {
531            for a in a.iter_mut() {
532                *a *= *a;
533            }
534        } else {
535            let b = Self::transform(b, len);
536            Self::multiply(&mut a, &b);
537        }
538        Self::inverse_transform(a, len)
539    }
540}

544fn convert_crt_input<M, N1, N2, N3>(t: MVec<M>, capacity: usize) -> (MVec<N1>, MVec<N2>, MVec<N3>)

553        MVec::<N2>::with_capacity(capacity),

565fn reconstruct_mint_crt<M, N1, N2, N3>(f: (MVec<N1>, MVec<N2>, MVec<N3>)) -> MVec<M>
566where
567    M: MIntConvert + MIntConvert<u32>,
568    N1: Montgomery32NttModulus,
569    N2: Montgomery32NttModulus,
570    N3: Montgomery32NttModulus,
571{
572    let t1 = MInt::<N2>::new(N1::get_mod()).inv();
573    let m1_3 = MInt::<N3>::new(N1::get_mod());
574    let t2 = (m1_3 * MInt::<N3>::new(N2::get_mod())).inv();
575    let modulus = <M as MIntConvert<u32>>::mod_into() as u64;
576    let m1 = N1::get_mod() as u64;
577    let m2 = m1 * N2::get_mod() as u64 % modulus;
578    let fits_u64 = (N1::get_mod() - 1) as u128
579        + (N2::get_mod() - 1) as u128 * m1 as u128
580        + (N3::get_mod() - 1) as u128 * m2 as u128
581        <= u64::MAX as u128;
582    f.0.into_iter()
583        .zip(f.1)
584        .zip(f.2)
585        .map(|((c1, c2), c3)| {
586            let d1 = c1.inner();
587            let d2 = ((c2 - MInt::<N2>::from(d1)) * t1).inner();
588            let x = MInt::<N3>::new(d1) + MInt::<N3>::new(d2) * m1_3;
589            let d3 = ((c3 - x) * t2).inner();
590            let value = if fits_u64 {
591                (d1 as u64 + d2 as u64 * m1 + d3 as u64 * m2) % modulus
592            } else {
593                ((d1 as u128 + d2 as u128 * m1 as u128 + d3 as u128 * m2 as u128) % modulus as u128)
594                    as u64
595            };
596            MInt::<M>::from(value as u32)
597        })
598        .collect()
599}

702                let d2 = ((c2 - MInt::<N2>::from(d1)) * t1).inner();
703                let x = MInt::<N3>::new(d1) + MInt::<N3>::new(d2) * m1_3;
704                let d3 = ((c3 - x) * t2).inner();
705                d1 as u64 + d2 as u64 * m1 + d3 as u64 * m2
706            })
707            .collect()
708    }
