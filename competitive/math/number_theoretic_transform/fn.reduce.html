37const fn mod_pow(mut x: u32, mut y: u32, p: u32, r: u32, mut z: u32) -> u32 {
38    while y > 0 {

49    const PRIMITIVE_ROOT: u32 = {
50        let mut g = 3u32;
51        loop {
52            let mut ok = true;
53            let mut d = 1u32;
54            while d * d < Self::MOD {
55                if (Self::MOD - 1) % d == 0 {
56                    let ds = [d, (Self::MOD - 1) / d];
