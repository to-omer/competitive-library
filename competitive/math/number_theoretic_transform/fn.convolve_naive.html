366    type F = Vec<MInt<M>>;
367    fn length(t: &Self::T) -> usize {
368        t.len()
369    }
370    fn transform(mut t: Self::T, len: usize) -> Self::F {
371        t.resize_with(len.max(1).next_power_of_two(), Zero::zero);
372        ntt(&mut t);
373        t
374    }

380    fn multiply(f: &mut Self::F, g: &Self::F) {
381        assert_eq!(f.len(), g.len());
382        #[cfg(target_arch = "x86_64")]
383        match simd_backend() {
384            SimdBackend::Avx512 => unsafe { ntt_simd::pointwise_multiply_avx512(f, g) },
385            SimdBackend::Avx2 => unsafe { ntt_simd::pointwise_multiply_avx2(f, g) },
386            SimdBackend::Scalar => {
387                for (f, g) in f.iter_mut().zip(g.iter()) {
388                    *f *= *g;
389                }
390            }
391        }
392        #[cfg(not(target_arch = "x86_64"))]
393        for (f, g) in f.iter_mut().zip(g.iter()) {
394            *f *= *g;
395        }
396    }
397    fn convolve(mut a: Self::T, mut b: Self::T) -> Self::T {
398        if Self::length(&a).max(Self::length(&b)) <= 100 {
399            return convolve_karatsuba(&a, &b);
400        }
401        if Self::length(&a).min(Self::length(&b)) <= 60 {
402            return convolve_naive(&a, &b);
403        }
404        let len = (Self::length(&a) + Self::length(&b)).saturating_sub(1);
405        let size = len.max(1).next_power_of_two();
406        if len <= size / 2 + 2 {
407            let xa = a.pop().unwrap();
408            let xb = b.pop().unwrap();
409            let mut c = vec![MInt::<M>::zero(); len];
410            *c.last_mut().unwrap() = xa * xb;
411            for (a, c) in a.iter().zip(&mut c[b.len()..]) {
412                *c += *a * xb;
413            }
414            for (b, c) in b.iter().zip(&mut c[a.len()..]) {
415                *c += *b * xa;
416            }
417            let d = Self::convolve(a, b);
418            for (d, c) in d.into_iter().zip(&mut c) {
419                *c += d;
420            }
421            return c;
422        }
423        let same = a == b;
424        #[cfg(target_arch = "x86_64")]
425        if !same && M::MOD < 1 << 30 && simd_backend() == SimdBackend::Avx2 {
426            a.resize_with(size, Zero::zero);
427            b.resize_with(size, Zero::zero);
428            unsafe { ntt_simd::convolve_blocks_avx2(&mut a, &mut b) };
429            a.truncate(len);
430            return a;
431        }
432        let mut a = Self::transform(a, len);
433        if same {
434            for a in a.iter_mut() {
435                *a *= *a;
436            }
437        } else {
438            let b = Self::transform(b, len);
439            Self::multiply(&mut a, &b);
440        }
441        Self::inverse_transform(a, len)
442    }

468fn reconstruct_mint_crt<M, N1, N2, N3>(f: (MVec<N1>, MVec<N2>, MVec<N3>)) -> MVec<M>
469where
470    M: MIntConvert + MIntConvert<u32>,
471    N1: Montgomery32NttModulus,
472    N2: Montgomery32NttModulus,
473    N3: Montgomery32NttModulus,
474{
475    let t1 = MInt::<N2>::new(N1::get_mod()).inv();
476    let m1_3 = MInt::<N3>::new(N1::get_mod());
477    let t2 = (m1_3 * MInt::<N3>::new(N2::get_mod())).inv();
478    let modulus = <M as MIntConvert<u32>>::mod_into() as u64;
479    let m1 = N1::get_mod() as u64;
480    let m2 = m1 * N2::get_mod() as u64 % modulus;
481    let fits_u64 = (N1::get_mod() - 1) as u128
482        + (N2::get_mod() - 1) as u128 * m1 as u128
483        + (N3::get_mod() - 1) as u128 * m2 as u128
484        <= u64::MAX as u128;
485    f.0.into_iter()
486        .zip(f.1)
487        .zip(f.2)
488        .map(|((c1, c2), c3)| {
489            let d1 = c1.inner();
490            let d2 = ((c2 - MInt::<N2>::from(d1)) * t1).inner();
491            let x = MInt::<N3>::new(d1) + MInt::<N3>::new(d2) * m1_3;
492            let d3 = ((c3 - x) * t2).inner();
493            let value = if fits_u64 {
494                (d1 as u64 + d2 as u64 * m1 + d3 as u64 * m2) % modulus
495            } else {
496                ((d1 as u128 + d2 as u128 * m1 as u128 + d3 as u128 * m2 as u128) % modulus as u128)
497                    as u64
498            };
499            MInt::<M>::from(value as u32)
500        })
501        .collect()
502}

620    }
