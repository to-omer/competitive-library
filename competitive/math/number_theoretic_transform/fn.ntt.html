565fn reconstruct_mint_crt<M, N1, N2, N3>(f: (MVec<N1>, MVec<N2>, MVec<N3>)) -> MVec<M>
566where
567    M: MIntConvert + MIntConvert<u32>,
568    N1: Montgomery32NttModulus,
569    N2: Montgomery32NttModulus,
570    N3: Montgomery32NttModulus,
571{
572    let t1 = MInt::<N2>::new(N1::get_mod()).inv();
573    let m1_3 = MInt::<N3>::new(N1::get_mod());
574    let t2 = (m1_3 * MInt::<N3>::new(N2::get_mod())).inv();
575    let modulus = <M as MIntConvert<u32>>::mod_into() as u64;
576    let m1 = N1::get_mod() as u64;
577    let m2 = m1 * N2::get_mod() as u64 % modulus;
578    let fits_u64 = (N1::get_mod() - 1) as u128
579        + (N2::get_mod() - 1) as u128 * m1 as u128
580        + (N3::get_mod() - 1) as u128 * m2 as u128
581        <= u64::MAX as u128;
582    f.0.into_iter()
583        .zip(f.1)
584        .zip(f.2)
585        .map(|((c1, c2), c3)| {
586            let d1 = c1.inner();
587            let d2 = ((c2 - MInt::<N2>::from(d1)) * t1).inner();
588            let x = MInt::<N3>::new(d1) + MInt::<N3>::new(d2) * m1_3;
589            let d3 = ((c3 - x) * t2).inner();
590            let value = if fits_u64 {
591                (d1 as u64 + d2 as u64 * m1 + d3 as u64 * m2) % modulus
592            } else {
593                ((d1 as u128 + d2 as u128 * m1 as u128 + d3 as u128 * m2 as u128) % modulus as u128)
594                    as u64
595            };
596            MInt::<M>::from(value as u32)
597        })
598        .collect()
599}

742fn convolve_u64_fft_scalar(a: Vec<u64>, b: Vec<u64>, factor: u64, inverse: u64) -> Vec<u64> {
743    fn split(values: &[u64], factor: u64) -> [Vec<i64>; 4] {
744        let mut result = std::array::from_fn(|_| Vec::with_capacity(values.len()));
745        let mut multiplier = 1u64;
746        for &value in values {
747            let mut value = value.wrapping_mul(multiplier);
748            for part in &mut result {
749                let digit = value as i16;
750                part.push(digit as i64);
751                value = (value >> 16).wrapping_add(u64::from(digit < 0));
752            }
753            multiplier = multiplier.wrapping_mul(factor);
754        }
755        result
756    }
757
758    let len = a.len() + b.len() - 1;
759    let fa = split(&a, factor).map(|a| ConvolveRealFft::transform(a, len));
760    drop(a);
761    let fb = split(&b, factor).map(|b| ConvolveRealFft::transform(b, len));
762    drop(b);
763    let values: [Vec<i64>; 4] = std::array::from_fn(|part| {
764        let mut sum = fa[0].clone();
765        ConvolveRealFft::multiply(&mut sum, &fb[part]);
766        for left in 1..=part {
767            let mut product = fa[left].clone();
768            ConvolveRealFft::multiply(&mut product, &fb[part - left]);
769            for (sum, product) in sum.iter_mut().zip(product) {
770                *sum += product;
771            }
772        }
773        ConvolveRealFft::inverse_transform(sum, len)
774    });
775    let mut multiplier = 1u64;
776    (0..len)
777        .map(|i| {
778            let value = (values[0][i] as u64)
779                .wrapping_add((values[1][i] as u64) << 16)
780                .wrapping_add((values[2][i] as u64) << 32)
781                .wrapping_add((values[3][i] as u64) << 48)
782                .wrapping_mul(multiplier);
783            multiplier = multiplier.wrapping_mul(inverse);
784            value
785        })
786        .collect()
787}
