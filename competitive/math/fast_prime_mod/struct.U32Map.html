290                    }
291                }
292                if x as usize >= k {
293                    continue;
294                }
295                while (i as u32) < x && lpf[x as usize] < i as u32 {
296                    let q = lpf[x as usize];
297                    x /= q;
298                    ans = add_mod(ans, log[k + q as usize], ord);
299                }
300                if 1 < x && x < i as u32 {
301                    ans = add_mod(ans, log[k + x as usize], ord);
302                    x = 1;
303                }
304                if x == 1 {
305                    log[k + i] = ans;
306                    break;
307                }
308            }
309        }
310    }
311    for i in 1..=k {
312        log[k - i] = add_mod(log[k + i], ord / 2, ord);
313    }
314    log
315}

381    fn insert(&mut self, key: u32, value: u32) {
382        debug_assert_ne!(key, 0);
383        let mut i = self.index(key);
384        while self.keys[i] != 0 && self.keys[i] != key {
385            i = (i + 1) & self.mask;
386        }
387        self.keys[i] = key;
388        self.values[i] = value;
389    }
390
391    fn get(&self, key: u32) -> Option<u32> {
392        debug_assert_ne!(key, 0);
393        let mut i = self.index(key);
394        while self.keys[i] != 0 {
395            if self.keys[i] == key {
396                return Some(self.values[i]);
397            }
398            i = (i + 1) & self.mask;
399        }
400        None
401    }
