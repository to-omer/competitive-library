212fn build_inv<const P: u32>() -> Box<[u32]> {
213    let k = table_k::<P>();
214    let mut inv = vec![0; table_len::<P>()].into_boxed_slice();
215    inv[k + 1] = 1;
216    for i in 2..=k {
217        let q = P.div_ceil(i as u32);

237        }
238        for &p in primes.iter() {
239            let p = p as usize;
240            if p > lpf[i] as usize || p > k / i {
241                break;
242            }

244        }
245    }

260    for i in 2..=k {
261        let p = lpf[i] as usize;
262        if p < i {

264        } else if i < 100 {

281        } else {

303                }

309        }
