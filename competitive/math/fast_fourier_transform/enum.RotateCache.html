130pub fn fft(a: &mut [Complex<f64>]) {
131    let n = a.len();
132    RotateCache::ensure(n / 2);
133    RotateCache::with(|cache| {
134        let mut v = n / 2;
135        while v >= 2 {
136            let l = v / 2;
137            for (q, block) in a.chunks_exact_mut(l * 4).enumerate() {
138                let (a, rest) = block.split_at_mut(l);
139                let (b, rest) = rest.split_at_mut(l);
140                let (c, d) = rest.split_at_mut(l);
141                let w0 = cache[q];
142                let w1 = cache[q << 1];
143                let w2 = cache[q << 1 | 1];
144                for i in 0..l {
145                    let cv = c[i] * w0;
146                    let dv = d[i] * w0;
147                    let ac0 = a[i] + cv;
148                    let ac1 = a[i] - cv;
149                    let bd0 = (b[i] + dv) * w1;
150                    let bd1 = (b[i] - dv) * w2;
151                    a[i] = ac0 + bd0;
152                    b[i] = ac0 - bd0;
153                    c[i] = ac1 + bd1;
154                    d[i] = ac1 - bd1;
155                }
156            }
157            v >>= 2;
158        }
159        if v == 1 {
160            for (a, w) in a.as_chunks_mut::<2>().0.iter_mut().zip(cache) {
161                let y = a[1] * *w;
162                a[1] = a[0] - y;
163                a[0] += y;
164            }
165        }
166    });
167}

169pub fn ifft(a: &mut [Complex<f64>]) {
170    let n = a.len();
171    RotateCache::ensure(n / 2);
172    RotateCache::with(|cache| {
173        let mut v = 1;
174        while v < n {
175            for (a, wj) in a.chunks_exact_mut(v << 1).zip(cache) {
176                let (l, r) = a.split_at_mut(v);
177                let wj = wj.conjugate();
178                for (x, y) in l.iter_mut().zip(r) {
179                    let ajv = *x - *y;
180                    *x += *y;
181                    *y = wj * ajv;
182                }
183            }
184            v <<= 1;
185        }
186    });

83unsafe fn fft_soa(a: &mut [Complex4]) {
84    let n = a.len() * 4;
85    RotateCache::ensure(n / 2);
86    RotateCache::with(|cache| {
87        let parity = n.trailing_zeros() & 1;
88        for leaf in (0..n).step_by(16) {
89            let mut level = (n + leaf).trailing_zeros();
90            level -= u32::from(level & 1 != parity);
91            while level >= 4 {
92                let len = 1usize << level;
93                let q = leaf >> level;
94                let width = len / 16;
95                let start = q * width * 4;
96                let (a, rest) = a[start..start + width * 4].split_at_mut(width);
97                let (b, rest) = rest.split_at_mut(width);
98                let (c, d) = rest.split_at_mut(width);
99                let w1 = eval_twiddle(cache, 4, n >> level, q);
100                let w2 = w1 * w1;
101                let w3 = w1 * w2;
102                let (w1r, w1i) = (_mm256_set1_pd(w1.re), _mm256_set1_pd(w1.im));
103                let (w2r, w2i) = (_mm256_set1_pd(w2.re), _mm256_set1_pd(w2.im));
104                let (w3r, w3i) = (_mm256_set1_pd(w3.re), _mm256_set1_pd(w3.im));
105                for i in 0..width {
106                    let (ar, ai) = load4(&a[i]);
107                    let (br, bi) = load4(&b[i]);
108                    let (cr, ci) = load4(&c[i]);
109                    let (dr, di) = load4(&d[i]);
110                    let (br, bi) = mul4(br, bi, w1r, w1i);
111                    let (cr, ci) = mul4(cr, ci, w2r, w2i);
112                    let (dr, di) = mul4(dr, di, w3r, w3i);
113                    let acr = _mm256_add_pd(ar, cr);

120                    let bdd_i = _mm256_sub_pd(bi, di);
121                    store4(&mut a[i], _mm256_add_pd(acr, bdr), _mm256_add_pd(aci, bdi));
122                    store4(&mut b[i], _mm256_sub_pd(acr, bdr), _mm256_sub_pd(aci, bdi));
123                    store4(
124                        &mut c[i],

128                    store4(

131                        _mm256_sub_pd(acd_i, bdd_r),
132                    );
133                }
134                level -= 2;
135            }
136        }
137        if parity != 0 {
138            let blocks = n / 8;
139            for k in 0..blocks {
140                let w = eval_twiddle(cache, 2, blocks, k);
141                let wr = _mm256_set1_pd(w.re);

143                let (ar, ai) = load4(&a[k * 2]);
144                let (br, bi) = load4(&a[k * 2 + 1]);

147                store4(
148                    &mut a[k * 2 + 1],

159    let n = a.len() * 4;
160    RotateCache::ensure(n / 2);
161    RotateCache::with(|cache| {
162        let parity = n.trailing_zeros() & 1;
163        if parity != 0 {
164            let blocks = n / 8;
165            for k in 0..blocks {
166                let w = eval_twiddle(cache, 2, blocks, k).conjugate();
167                let wr = _mm256_set1_pd(w.re);
168                let wi = _mm256_set1_pd(w.im);
169                let (ar, ai) = load4(&a[k * 2]);
170                let (br, bi) = load4(&a[k * 2 + 1]);
171                store4(&mut a[k * 2], _mm256_add_pd(ar, br), _mm256_add_pd(ai, bi));
172                let (br, bi) = mul4(_mm256_sub_pd(ar, br), _mm256_sub_pd(ai, bi), wr, wi);
173                store4(&mut a[k * 2 + 1], br, bi);
174            }
175        }
176        for leaf in (12..n).step_by(16) {
177            let max_level = (leaf + 3).trailing_ones();
178            let mut level = 4 + parity;
179            while level <= max_level {
180                let len = 1usize << level;
181                let q = leaf >> level;
182                let width = len / 16;
183                let start = q * width * 4;
184                let (a, rest) = a[start..start + width * 4].split_at_mut(width);
185                let (b, rest) = rest.split_at_mut(width);
186                let (c, d) = rest.split_at_mut(width);
187                let w1 = eval_twiddle(cache, 4, n >> level, q).conjugate();

190                let (w1r, w1i) = (_mm256_set1_pd(w1.re), _mm256_set1_pd(w1.im));

215                    store4(&mut c[i], cr, ci);

224                level += 2;
225            }

235fn split_coefficients<M>(
236    values: Vec<MInt<M>>,
237    n: usize,
238    modulus: i64,
239    split: i64,
240) -> (Vec<Complex4>, Vec<Complex4>)
241where
242    M: MIntConvert + MIntConvert<u32>,
243{
244    let mut low = vec![Complex4::default(); n / 4];
245    let mut high = low.clone();
246    for (i, x) in values.into_iter().enumerate() {
247        let mut x = <M as MIntConvert<u32>>::into(x.inner()) as i64;
248        if x * 2 > modulus {
249            x -= modulus;
250        }
251        let upper = (x as f64 / split as f64).round() as i64;
252        let (i, imag) = if i < n { (i, false) } else { (i - n, true) };
253        let lane = i & 3;
254        if imag {
255            low[i >> 2].im[lane] = (x - upper * split) as f64;
256            high[i >> 2].im[lane] = upper as f64;
257        } else {
258            low[i >> 2].re[lane] = (x - upper * split) as f64;
259            high[i >> 2].re[lane] = upper as f64;
260        }
261    }
262    (low, high)
263}

266unsafe fn dot_soa(a0: &mut [Complex4], a1: &mut [Complex4], b0: &mut [Complex4], b1: &[Complex4]) {
267    let n = a0.len() * 4;
268    RotateCache::ensure(n / 2);
269    RotateCache::with(|cache| {
270        for i in 0..a0.len() {
271            let (mut cr, mut ci) = load4(&b0[i]);
272            let (mut dr, mut di) = load4(&b1[i]);
273            let mut c0r = _mm256_setzero_pd();
274            let mut c0i = _mm256_setzero_pd();
275            let mut c1r = _mm256_setzero_pd();
276            let mut c1i = _mm256_setzero_pd();
277            let mut c2r = _mm256_setzero_pd();
278            let mut c2i = _mm256_setzero_pd();
279            let w = eval_twiddle(cache, 1, a0.len(), i);
280            let wr = _mm256_setr_pd(w.re, 1.0, 1.0, 1.0);
281            let wi = _mm256_setr_pd(w.im, 0.0, 0.0, 0.0);
282            for lane in 0..4 {
283                let ar = _mm256_set1_pd(a0[i].re[lane]);
284                let ai = _mm256_set1_pd(a0[i].im[lane]);
285                let br = _mm256_set1_pd(a1[i].re[lane]);
286                let bi = _mm256_set1_pd(a1[i].im[lane]);
287                multiply_accumulate4(&mut c0r, &mut c0i, ar, ai, cr, ci);
288                multiply_accumulate4(&mut c1r, &mut c1i, ar, ai, dr, di);
289                multiply_accumulate4(&mut c1r, &mut c1i, br, bi, cr, ci);
290                multiply_accumulate4(&mut c2r, &mut c2i, br, bi, dr, di);
291                if lane != 3 {
292                    cr = _mm256_permute4x64_pd::<0x93>(cr);
293                    ci = _mm256_permute4x64_pd::<0x93>(ci);
294                    dr = _mm256_permute4x64_pd::<0x93>(dr);
295                    di = _mm256_permute4x64_pd::<0x93>(di);
296                    (cr, ci) = mul4(cr, ci, wr, wi);
297                    (dr, di) = mul4(dr, di, wr, wi);
298                }
299            }
300            store4(&mut a0[i], c0r, c0i);
301            store4(&mut a1[i], c1r, c1i);
302            store4(&mut b0[i], c2r, c2i);
303        }
304    });
