110        RotateCache::with(|cache| {
111            let parity = n.trailing_zeros() & 1;
112            for leaf in (0..n).step_by(16) {
113                let mut level = (n + leaf).trailing_zeros();
114                level -= u32::from(level & 1 != parity);
115                while level >= 4 {
116                    let len = 1usize << level;
117                    let q = leaf >> level;
118                    let width = len / 16;
119                    let start = q * width * 4;
120                    let (a, rest) = a[start..start + width * 4].split_at_mut(width);
121                    let (b, rest) = rest.split_at_mut(width);
122                    let (c, d) = rest.split_at_mut(width);

125                    let w3 = w1 * w2;

192                    let wi = _mm256_set1_pd(w.im);

327fn bit_reverse<T>(f: &mut [T]) {
328    let mut ip = vec![0u32];
329    let mut k = f.len();
330    let mut m = 1;
331    while 2 * m < k {
332        k /= 2;
333        for j in 0..m {
334            ip.push(ip[j] + k as u32);
335        }
336        m *= 2;
337    }
338    if m == k {
339        for i in 1..m {
340            for j in 0..i {
341                let ji = j + ip[i] as usize;
342                let ij = i + ip[j] as usize;
343                f.swap(ji, ij);
344            }
345        }
346    } else {
347        for i in 1..m {
348            for j in 0..i {
349                let ji = j + ip[i] as usize;
350                let ij = i + ip[j] as usize;
351                f.swap(ji, ij);
352                f.swap(ji + m, ij + m);
353            }
354        }
355    }
356}

358fn real_twiddles(n: usize, inverse: bool, mut f: impl FnMut(usize, Complex<f64>)) {
359    const BLOCK: usize = 256;
360    let sign = if inverse { 1.0 } else { -1.0 };
361    let step = Complex::primitive_nth_root_of_unity(sign * n as f64);
362    for start in (1..n / 4).step_by(BLOCK) {
363        let mut w = Complex::polar(1.0, sign * std::f64::consts::TAU * start as f64 / n as f64);
364        for k in start..(start + BLOCK).min(n / 4) {
365            f(k, w);
366            w *= step;
367        }
368    }
369}

371pub fn transform_real(t: impl IntoIterator<Item = f64>, len: usize) -> Vec<Complex<f64>> {
372    let n = len.max(4).next_power_of_two();
373    let mut f = vec![Complex::zero(); n / 2];
374    for (i, t) in t.into_iter().enumerate() {
375        if i & 1 == 0 {
376            f[i / 2].re = t;
377        } else {
378            f[i / 2].im = t;
379        }
380    }
381    fft(&mut f);
382    bit_reverse(&mut f);
383    f[0] = Complex::new(f[0].re + f[0].im, f[0].re - f[0].im);
384    f[n / 4] = f[n / 4].conjugate();
385    real_twiddles(n, false, |k, wk| {
386        let c = wk.conjugate().transpose() + 1.;
387        let d = c * (f[k] - f[n / 2 - k].conjugate()) * 0.5;
388        f[k] -= d;
389        f[n / 2 - k] += d.conjugate();
390    });
391    f
392}

394pub fn inverse_transform_real(mut f: Vec<Complex<f64>>, len: usize) -> Vec<f64> {
395    let n = len.max(4).next_power_of_two();
396    assert_eq!(f.len(), n / 2);
397    f[0] = Complex::new((f[0].re + f[0].im) * 0.5, (f[0].re - f[0].im) * 0.5);
398    f[n / 4] = f[n / 4].conjugate();
399    real_twiddles(n, true, |k, wk| {
400        let c = wk.transpose().conjugate() + 1.;
401        let d = c * (f[k] - f[n / 2 - k].conjugate()) * 0.5;
402        f[k] -= d;
403        f[n / 2 - k] += d.conjugate();
404    });
405    bit_reverse(&mut f);
406    ifft(&mut f);
407    let inv = 1. / (n / 2) as f64;
408    (0..len)
409        .map(|i| inv * if i & 1 == 0 { f[i / 2].re } else { f[i / 2].im })
410        .collect()
411}

466pub fn fft(a: &mut [Complex<f64>]) {
467    let n = a.len();
468    RotateCache::ensure(n / 2);
469    RotateCache::with(|cache| {
470        let mut v = n / 2;
471        while v >= 2 {
472            let l = v / 2;
473            for (q, block) in a.chunks_exact_mut(l * 4).enumerate() {
474                let (a, rest) = block.split_at_mut(l);
475                let (b, rest) = rest.split_at_mut(l);
476                let (c, d) = rest.split_at_mut(l);
477                let w0 = cache[q];
478                let w1 = cache[q << 1];
479                let w2 = cache[q << 1 | 1];
480                for i in 0..l {
481                    let cv = c[i] * w0;
482                    let dv = d[i] * w0;
483                    let ac0 = a[i] + cv;
484                    let ac1 = a[i] - cv;
485                    let bd0 = (b[i] + dv) * w1;
486                    let bd1 = (b[i] - dv) * w2;
487                    a[i] = ac0 + bd0;
488                    b[i] = ac0 - bd0;
489                    c[i] = ac1 + bd1;
490                    d[i] = ac1 - bd1;
491                }
492            }
493            v >>= 2;
494        }
495        if v == 1 {
496            for (a, w) in a.as_chunks_mut::<2>().0.iter_mut().zip(cache) {
497                let y = a[1] * *w;
498                a[1] = a[0] - y;
499                a[0] += y;
500            }
501        }
502    });
503}

505pub fn ifft(a: &mut [Complex<f64>]) {
506    let n = a.len();
507    RotateCache::ensure(n / 2);
508    RotateCache::with(|cache| {
509        let mut v = 1;
510        while v < n {
511            for (a, wj) in a.chunks_exact_mut(v << 1).zip(cache) {
512                let (l, r) = a.split_at_mut(v);
513                let wj = wj.conjugate();
514                for (x, y) in l.iter_mut().zip(r) {
515                    let ajv = *x - *y;
516                    *x += *y;
517                    *y = wj * ajv;
518                }
519            }
520            v <<= 1;
521        }
522    });
