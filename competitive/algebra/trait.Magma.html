    pub fn from_slice(slice: &[M::T]) -> Self {
        let n = slice.len();
        let mut bit = vec![M::unit(); n + 1];
        for (i, x) in slice.iter().enumerate() {
            let k = i + 1;
            M::operate_assign(&mut bit[k], x);
            let j = k + (k & (!k + 1));
            if j <= n {
                bit[j] = M::operate(&bit[j], &bit[k]);
            }
        }
        Self { n, bit }
    }
