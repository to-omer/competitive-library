235            f: &mut impl FnMut([usize; K]) -> T,
236            mut index: [usize; K],
237            pos: usize,
238        ) {
239            if pos < K {
240                for i in 0..dim[pos] {
241                    index[pos] = i;
242                    fill(dim, offset, data, f, index, pos + 1);
243                }
244            } else {

248        }

251        for d in (1..K).rev() {
252            offset[d - 1] = offset[d] * (dim[d] + 1);
253        }

255        let mut data = vec![M::unit(); size];
