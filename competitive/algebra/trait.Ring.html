250        let (n, m) = self.shape;
251        let mut c = 0;
252        let mut pivots = vec![];
253        let mut row_id: Vec<usize> = (0..n).collect();
254        for r in 0..n {
255            loop {
256                if c >= m {
257                    return pivots;
258                }
259                if let Some(pivot) = (r..n).find(|&p| !R::is_zero(&self[p][c])) {
260                    self.data.swap(r, pivot);
261                    row_id.swap(r, pivot);
262                    pivots.push((row_id[r], c));
