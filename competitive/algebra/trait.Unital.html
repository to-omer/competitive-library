5pub fn baby_step_giant_step<M>(x: M::T, y: M::T, n: usize) -> Option<usize>
6where
7    M: Monoid,
8    M::T: Eq + Hash,
9{
10    if M::is_unit(&y) {
11        return Some(0);
12    }
13    let block_size = 1usize.max((n as f64).sqrt() as _);
14    let mut baby = HashSet::new();
15    let mut t = y.clone();
16    for _ in 0..block_size {
17        t = M::operate(&x, &t);
18        baby.insert(t.clone());
19    }
20    let g = M::pow(x.clone(), block_size);
21    let mut t = M::unit();
22    let mut fail = 0usize;
23    for k in (0..n).step_by(block_size) {
24        let nt = M::operate(&g, &t);
25        if baby.contains(&nt) {
26            for m in k..n.min(k + block_size) {
27                if t == y {
28                    return Some(m);
29                }
30                t = M::operate(&x, &t);
31            }
32            fail += 1;
33            if fail >= 2 {

35            }
36        }
