    F: FnMut(&[T]), 
Expand description

choose r elements from n in sorted order

Example

let n = vec![1, 2, 3, 4];
let mut p = Vec::new();
let mut q = Vec::new();
combinations_with_replacement(n.as_slice(), 2, |v| p.push(v.to_vec()));
for (i, x) in n.iter().cloned().enumerate() {
    for y in n[i..].iter().cloned() {
        q.push(vec![x, y]);
    }
}
assert_eq!(p, q);
