    pub fn eval(&mut self, s: &G::State) -> u64 {
        if let Some(g) = self.grundy.get(s).cloned() {
            g
        } else {
            let next: HashSet<_> = self.game.next_state(s).map(|ns| self.eval(&ns)).collect();
            let mut g = 0u64;
            while next.contains(&g) {
                g += 1;
            }
            self.grundy.insert(s.clone(), g);
            g
        }
    }
