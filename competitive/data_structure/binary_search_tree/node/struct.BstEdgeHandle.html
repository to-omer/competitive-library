329                            break Ordering::Less;
330                        };
331                        Spec::top_down(grandchild.borrow_datamut());
332                        let child_left = unsafe { child.borrow_mut().left_mut().take() };
333                        if let Some(child_left) = child_left {
334                            unsafe { root.borrow_mut().right_mut().set(child_left) };
335                        }
336                        Spec::bottom_up(root.borrow_datamut());
337                        unsafe { child.borrow_mut().left_mut().set(root) };
338                        add!(@left child);
339                        root = grandchild;
340                    }
341                }
342            }
343        }
344    };
345
346    *left_entry = unsafe { root.borrow_mut().left_mut().take() }.map(|node| node.node);
347    *right_entry = unsafe { root.borrow_mut().right_mut().take() }.map(|node| node.node);
348    unsafe {
349        root.node.as_mut().child[0] = left_subtree;
350        root.node.as_mut().child[1] = right_subtree;
351        while let Some(node) = overflow_stack.pop() {
352            Spec::bottom_up(BstRoot::new(node).borrow_datamut());
353        }
354        while inline_len > 0 {
355            inline_len -= 1;
356            let node = inline_stack[inline_len].unwrap_unchecked();
357            Spec::bottom_up(BstRoot::new(node).borrow_datamut());
358        }
359    }
360    Spec::bottom_up(root.borrow_datamut());
361    (root_ordering, root)
362}

373        (None, None) => None,

377            unsafe { right.borrow_mut().left_mut().set(left) };
378            Spec::bottom_up(right.borrow_datamut());
379            Some(right)
380        }
381        (Some(left), Some(right)) => {

384            Spec::bottom_up(root.borrow_datamut());
