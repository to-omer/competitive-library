289    pub fn root_path(self) -> (Self, Vec<bool>) {
290        let mut node = self;
291        let mut nn = node.node;
292        let mut stack = vec![];
293        let root = loop {
294            match node.ascend() {
295                Ok(parent) => {
296                    node = parent;
297                    stack.push(
298                        node.reborrow()
299                            .left()
300                            .descend()

303                    nn = node.node;
304                }

307                }
308            }
