    /// index of k-th 1
    fn select1(&self, k: usize) -> Option<usize> {
        let n = self.bit_length();
        if self.rank1(n) <= k {
            return None;
        }
        let (mut l, mut r) = (0, n);
        while r - l > 1 {
            let m = (l + r) / 2;
            if self.rank1(m) <= k {
                l = m;
            } else {
                r = m;
            }
        }
        Some(l)
    }

    pub fn new<T>(mut v: Vec<T>, bit_length: usize) -> Self
    where
        T: Clone + RankSelectDictionaries,
    {
        let len = v.len();
        let mut table = Vec::new();
        for d in (0..bit_length).rev() {
            let b: BitVector = v.iter().map(|x| x.access(d)).collect();
            table.push((b.rank0(len), b));
            v = v
                .iter()
                .filter(|&x| !x.access(d))
                .chain(v.iter().filter(|&x| x.access(d)))
                .cloned()
                .collect();
        }
        Self { len, table }
    }

    fn select1(&self, mut k: usize) -> Option<usize> {
        let (mut l, mut r) = (0, self.data.len());
        if self.sum <= k {
            return None;
        }
        while r - l > 1 {
            let m = (l + r) / 2;
            if self.data[m].1 <= k {
                l = m;
            } else {
                r = m;
            }
        }
        let (bit, sum) = self.data[l];
        k -= sum;
        Some(l * Self::WORD_SIZE + bit.select1(k).unwrap())
    }
