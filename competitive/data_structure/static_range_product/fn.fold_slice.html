176        let between = Box::new(Self::new(blocks.products, level - 1));

188        match self {
189            Self::Direct { data } => fold_slice::<S>(data, l, r),
190            Self::Disjoint { table } => table.fold(l, r),
191            Self::Recursive {
192                data,
193                block_shift,
194                prefix,
195                suffix,
196                between,
197            } => {

200                let br = (r - 1) >> block_shift;
201                if bl == br {
