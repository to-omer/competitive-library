impl<T> Line<T>
where
    T: Ccwable + Float,
{
    pub fn projection(&self, p: Complex<T>) -> Complex<T> {
        let e = self.dir().unit();
        self.p1 + e * (p - self.p1).dot(e)
    }
    pub fn reflection(&self, p: Complex<T>) -> Complex<T> {
        let d = self.projection(p) - p;
        p + d + d
    }
    pub fn distance_point(&self, p: Complex<T>) -> T {
        (p / self.dir().unit()).re
    }
}

pub fn convex_hull<T>(mut ps: Vec<Complex<T>>) -> Vec<Complex<T>>
where
    T: PartialOrd + Ccwable,
{
    ps.sort_by(|p1, p2| ((p1.re, p1.im).partial_cmp(&(p2.re, p2.im)).unwrap()));
    let mut qs = Vec::new();
    for &p in ps.iter().chain(ps.iter().rev().skip(1)) {
        while {
            let k = qs.len();
            k > 1 && matches!(Ccw::ccw(qs[k - 2], qs[k - 1], p), Ccw::Clockwise)
        } {
            qs.pop();
        }
        qs.push(p);
    }
    qs.pop();
    qs
}
