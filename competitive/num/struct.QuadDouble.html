    pub fn sqrt(self) -> Self {
        if self.is_zero() {
            return Self::from(0.);
        }
        let x = Self::from(1. / self.0.sqrt());
        let x = x + x * (Self::from(1.) - self * x * x).div2(2.);
        let x = x + x * (Self::from(1.) - self * x * x).div2(2.);
        let x = x + x * (Self::from(1.) - self * x * x).div2(2.);
        x * self
    }
