56#[target_feature(enable = "avx512f,avx512dq,avx512cd,avx512bw,avx512vl")]
57pub unsafe fn pointwise_multiply_avx512<M>(f: &mut [MInt<M>], g: &[MInt<M>])
58where
59    M: Montgomery32NttModulus,
60{
61    let r_vec = _mm512_set1_epi32(M::R as i32);
62    let mod_vec = _mm512_set1_epi32(M::MOD as i32);
63    let mut i = 0;
64    while i + 16 <= f.len() {
65        let a = _mm512_loadu_si512(f.as_ptr().add(i) as *const __m512i);
66        let b = _mm512_loadu_si512(g.as_ptr().add(i) as *const __m512i);
67        let x = simd32::montgomery_mul_512_canon(a, b, r_vec, mod_vec);
68        _mm512_storeu_si512(f.as_mut_ptr().add(i) as *mut __m512i, x);
69        i += 16;
70    }
71    while i < f.len() {
72        f[i] *= g[i];
73        i += 1;
74    }
75}

95    }
96    while i < sum.len() {
97        sum[i] += f[i] * g[i];

102#[target_feature(enable = "avx512f,avx512dq,avx512cd,avx512bw,avx512vl")]
103pub unsafe fn ntt_avx512<M>(a: &mut [MInt<M>])
104where
105    M: Montgomery32NttModulus,
106{
107    let n = a.len();
108    if n <= 1 {
109        return;
110    }
111    let ptr = a.as_mut_ptr() as *mut u32;
112    let a = std::slice::from_raw_parts_mut(ptr, n);
113    let mod_vec = _mm512_set1_epi32(M::MOD as i32);
114    let mod2_vec = _mm512_set1_epi32(M::MOD.wrapping_add(M::MOD) as i32);
115    let r_vec = _mm512_set1_epi32(M::R as i32);
116    let imag = M::INFO.root[2];
117    let imag_vec = _mm512_set1_epi32(imag as i32);
118
119    let mut v = n / 2;
120    if n.trailing_zeros() & 1 == 1 {
121        let mut i = 0;
122        while i + 16 <= v {
123            let x0 = _mm512_loadu_si512(a.as_ptr().add(i) as *const __m512i);
124            let x1 = _mm512_loadu_si512(a.as_ptr().add(v + i) as *const __m512i);
125            let y0 = add_vec_avx512::<M>(x0, x1, mod_vec, mod2_vec);
126            let y1 = sub_vec_avx512::<M>(x0, x1, mod_vec, mod2_vec);
127            _mm512_storeu_si512(a.as_mut_ptr().add(i) as *mut __m512i, y0);
128            _mm512_storeu_si512(a.as_mut_ptr().add(v + i) as *mut __m512i, y1);
129            i += 16;
130        }
131        while i < v {
132            let x0 = a[i];
133            let x1 = a[v + i];
134            a[i] = M::mod_add(x0, x1);
135            a[v + i] = M::mod_sub(x0, x1);
136            i += 1;
137        }
138        v >>= 1;
139    }
140    while v > 1 {
141        let half = v >> 1;
142        let mut w1 = M::N1;
143        for (s, block) in a.chunks_exact_mut(v << 1).enumerate() {
144            let base = block.as_mut_ptr();
145            let ll = base;
146            let lr = base.add(half);
147            let rl = base.add(v);
148            let rr = base.add(v + half);
149            let w2 = M::mod_mul(w1, w1);
150            let w3 = M::mod_mul(w2, w1);
151            let w1v = _mm512_set1_epi32(w1 as i32);
152            let w2v = _mm512_set1_epi32(w2 as i32);
153            let w3v = _mm512_set1_epi32(w3 as i32);
154
155            let mut i = 0;
156            while i + 16 <= half {
157                let x0 = _mm512_loadu_si512(ll.add(i) as *const __m512i);
158                let x1 = _mm512_loadu_si512(lr.add(i) as *const __m512i);
159                let x2 = _mm512_loadu_si512(rl.add(i) as *const __m512i);
160                let x3 = _mm512_loadu_si512(rr.add(i) as *const __m512i);
161
162                let (a1, a2, a3) = if s == 0 {
163                    (x1, x2, x3)
164                } else {
165                    (
166                        mul_vec_avx512::<M>(x1, w1v, r_vec, mod_vec),
167                        mul_vec_avx512::<M>(x2, w2v, r_vec, mod_vec),
168                        mul_vec_avx512::<M>(x3, w3v, r_vec, mod_vec),
169                    )
170                };
171
172                let a0pa2 = add_vec_avx512::<M>(x0, a2, mod_vec, mod2_vec);
173                let a0na2 = sub_vec_avx512::<M>(x0, a2, mod_vec, mod2_vec);
174                let a1pa3 = add_vec_avx512::<M>(a1, a3, mod_vec, mod2_vec);
175                let a1na3 = sub_vec_avx512::<M>(a1, a3, mod_vec, mod2_vec);
176                let a1na3imag = mul_vec_avx512::<M>(a1na3, imag_vec, r_vec, mod_vec);
177
178                let y0 = add_vec_avx512::<M>(a0pa2, a1pa3, mod_vec, mod2_vec);
179                let y1 = sub_vec_avx512::<M>(a0pa2, a1pa3, mod_vec, mod2_vec);
180                let y2 = add_vec_avx512::<M>(a0na2, a1na3imag, mod_vec, mod2_vec);
181                let y3 = sub_vec_avx512::<M>(a0na2, a1na3imag, mod_vec, mod2_vec);
182
183                _mm512_storeu_si512(ll.add(i) as *mut __m512i, y0);
184                _mm512_storeu_si512(lr.add(i) as *mut __m512i, y1);
185                _mm512_storeu_si512(rl.add(i) as *mut __m512i, y2);
186                _mm512_storeu_si512(rr.add(i) as *mut __m512i, y3);
187                i += 16;
188            }
189            while i < half {
190                let a0 = *ll.add(i);
191                let a1 = M::mod_mul(*lr.add(i), w1);
192                let a2 = M::mod_mul(*rl.add(i), w2);
193                let a3 = M::mod_mul(*rr.add(i), w3);
194                let a0pa2 = M::mod_add(a0, a2);
195                let a0na2 = M::mod_sub(a0, a2);
196                let a1pa3 = M::mod_add(a1, a3);
197                let a1na3 = M::mod_sub(a1, a3);
198                let a1na3imag = M::mod_mul(a1na3, imag);
199                *ll.add(i) = M::mod_add(a0pa2, a1pa3);
200                *lr.add(i) = M::mod_sub(a0pa2, a1pa3);
201                *rl.add(i) = M::mod_add(a0na2, a1na3imag);
202                *rr.add(i) = M::mod_sub(a0na2, a1na3imag);
203                i += 1;
204            }
205            w1 = M::mod_mul(w1, M::INFO.rate3[s.trailing_ones() as usize]);
206        }
207        v >>= 2;
208    }
209    normalize_avx512::<M>(a);
210}

213pub unsafe fn intt_avx512<M>(a: &mut [MInt<M>])
214where
215    M: Montgomery32NttModulus,
216{
217    let n = a.len();
218    if n <= 1 {
219        return;
220    }
221    let ptr = a.as_mut_ptr() as *mut u32;
222    let a = std::slice::from_raw_parts_mut(ptr, n);
223    let mod_vec = _mm512_set1_epi32(M::MOD as i32);
224    let mod2_vec = _mm512_set1_epi32(M::MOD.wrapping_add(M::MOD) as i32);
225    let r_vec = _mm512_set1_epi32(M::R as i32);
226    let iimag = M::INFO.inv_root[2];
227    let iimag_vec = _mm512_set1_epi32(iimag as i32);
228
229    let mut v = 1;
230    let limit = if n.trailing_zeros() & 1 == 1 {
231        n / 2
232    } else {
233        n
234    };
235    while v < limit {
236        let mut w1 = M::N1;
237        for (s, block) in a.chunks_exact_mut(v << 2).enumerate() {
238            let base = block.as_mut_ptr();
239            let ll = base;
240            let lr = base.add(v);
241            let rl = base.add(v << 1);
242            let rr = base.add(v * 3);
243            let w2 = M::mod_mul(w1, w1);
244            let w3 = M::mod_mul(w2, w1);
245            let w1v = _mm512_set1_epi32(w1 as i32);
246            let w2v = _mm512_set1_epi32(w2 as i32);
247            let w3v = _mm512_set1_epi32(w3 as i32);
248
249            let mut i = 0;
250            while i + 16 <= v {
251                let x0 = _mm512_loadu_si512(ll.add(i) as *const __m512i);
252                let x1 = _mm512_loadu_si512(lr.add(i) as *const __m512i);
253                let x2 = _mm512_loadu_si512(rl.add(i) as *const __m512i);
254                let x3 = _mm512_loadu_si512(rr.add(i) as *const __m512i);
255
256                let a0pa1 = add_vec_avx512::<M>(x0, x1, mod_vec, mod2_vec);
257                let a0na1 = sub_vec_avx512::<M>(x0, x1, mod_vec, mod2_vec);
258                let a2pa3 = add_vec_avx512::<M>(x2, x3, mod_vec, mod2_vec);
259                let a2na3 = sub_vec_avx512::<M>(x2, x3, mod_vec, mod2_vec);
260                let a2na3iimag = mul_vec_avx512::<M>(a2na3, iimag_vec, r_vec, mod_vec);
261
262                let y0 = add_vec_avx512::<M>(a0pa1, a2pa3, mod_vec, mod2_vec);
263                let y1 = add_vec_avx512::<M>(a0na1, a2na3iimag, mod_vec, mod2_vec);
264                let y2 = sub_vec_avx512::<M>(a0pa1, a2pa3, mod_vec, mod2_vec);
265                let y3 = sub_vec_avx512::<M>(a0na1, a2na3iimag, mod_vec, mod2_vec);
266
267                let (y1, y2, y3) = if s == 0 {
268                    (y1, y2, y3)
269                } else {
270                    (
271                        mul_vec_avx512::<M>(y1, w1v, r_vec, mod_vec),
272                        mul_vec_avx512::<M>(y2, w2v, r_vec, mod_vec),
273                        mul_vec_avx512::<M>(y3, w3v, r_vec, mod_vec),
274                    )
275                };
276
277                _mm512_storeu_si512(ll.add(i) as *mut __m512i, y0);
278                _mm512_storeu_si512(lr.add(i) as *mut __m512i, y1);
279                _mm512_storeu_si512(rl.add(i) as *mut __m512i, y2);
280                _mm512_storeu_si512(rr.add(i) as *mut __m512i, y3);
281                i += 16;
282            }
283            while i < v {
284                let a0 = *ll.add(i);
285                let a1 = *lr.add(i);
286                let a2 = *rl.add(i);
287                let a3 = *rr.add(i);
288                let a0pa1 = M::mod_add(a0, a1);
289                let a0na1 = M::mod_sub(a0, a1);
290                let a2pa3 = M::mod_add(a2, a3);
291                let a2na3iimag = M::mod_mul(M::mod_sub(a2, a3), iimag);
292                *ll.add(i) = M::mod_add(a0pa1, a2pa3);
293                *lr.add(i) = M::mod_mul(M::mod_add(a0na1, a2na3iimag), w1);
294                *rl.add(i) = M::mod_mul(M::mod_sub(a0pa1, a2pa3), w2);
295                *rr.add(i) = M::mod_mul(M::mod_sub(a0na1, a2na3iimag), w3);
296                i += 1;
297            }
298            w1 = M::mod_mul(w1, M::INFO.inv_rate3[s.trailing_ones() as usize]);
299        }
300        v <<= 2;
301    }
302    if n.trailing_zeros() & 1 == 1 {
303        let half = n / 2;
304        let mut i = 0;
305        while i + 16 <= half {
306            let x0 = _mm512_loadu_si512(a.as_ptr().add(i) as *const __m512i);
307            let x1 = _mm512_loadu_si512(a.as_ptr().add(half + i) as *const __m512i);
308            let y0 = add_vec_avx512::<M>(x0, x1, mod_vec, mod2_vec);
309            let y1 = sub_vec_avx512::<M>(x0, x1, mod_vec, mod2_vec);
310            _mm512_storeu_si512(a.as_mut_ptr().add(i) as *mut __m512i, y0);
311            _mm512_storeu_si512(a.as_mut_ptr().add(half + i) as *mut __m512i, y1);
312            i += 16;
313        }
314        while i < half {
315            let x0 = a[i];
316            let x1 = a[half + i];
317            a[i] = M::mod_add(x0, x1);
318            a[half + i] = M::mod_sub(x0, x1);
319            i += 1;
320        }
321    }
322    let inv = M::mod_inv(<M as MIntConvert<u32>>::from(n as u32));
323    let inv_vec = _mm512_set1_epi32(inv as i32);
324    let mut i = 0;
325    while i + 16 <= n {
326        let x = _mm512_loadu_si512(a.as_ptr().add(i) as *const __m512i);
327        let y = simd32::montgomery_mul_512_canon(x, inv_vec, r_vec, mod_vec);
328        _mm512_storeu_si512(a.as_mut_ptr().add(i) as *mut __m512i, y);
329        i += 16;
330    }
331    while i < n {
332        a[i] = M::mod_mul(a[i], inv);
333        i += 1;
334    }
