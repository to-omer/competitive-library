18unsafe fn ntt_block_stage_avx2<M>(
19    a: *mut u32,
20    n: usize,
21    v: usize,
22    mut w: [u32; 3],
23    mod_vec: __m256i,
24    mod2_vec: __m256i,
25    imag_vec: __m256i,
26) where
27    M: Montgomery32NttModulus,
28{
29    let half = v >> 1;
30    let mut base = 0;
31    let mut s = 0usize;
32    while base < n {
33        let w1v = _mm256_set1_epi32(w[0] as i32);
34        let w2v = _mm256_set1_epi32(w[1] as i32);
35        let w3v = _mm256_set1_epi32(w[2] as i32);
36        let w1r = _mm256_set1_epi32(w[0].wrapping_mul(M::R) as i32);
37        let w2r = _mm256_set1_epi32(w[1].wrapping_mul(M::R) as i32);
38        let w3r = _mm256_set1_epi32(w[2].wrapping_mul(M::R) as i32);
39        let mut i = 0;
40        while i < half {
41            let x0 = load_block_avx2(a, base + i);
42            let x1 = load_block_avx2(a, base + half + i);
43            let x2 = load_block_avx2(a, base + v + i);
44            let x3 = load_block_avx2(a, base + v + half + i);
45            let (a1, a2, a3) = if s == 0 {
46                (x1, x2, x3)
47            } else {
48                (
49                    simd32::montgomery_mul_256_fixed(x1, w1v, w1r, mod_vec),
50                    simd32::montgomery_mul_256_fixed(x2, w2v, w2r, mod_vec),
51                    simd32::montgomery_mul_256_fixed(x3, w3v, w3r, mod_vec),
52                )
53            };
54            let a0pa2 = add_vec_avx2::<M>(x0, a2, mod_vec, mod2_vec);
55            let a0na2 = sub_vec_avx2::<M>(x0, a2, mod_vec, mod2_vec);
56            let a1pa3 = add_vec_avx2::<M>(a1, a3, mod_vec, mod2_vec);
57            let a1na3 = sub_vec_avx2::<M>(a1, a3, mod_vec, mod2_vec);
58            let a1na3imag = simd32::montgomery_mul_256_fixed(
59                a1na3,
60                imag_vec,
61                _mm256_set1_epi32(M::INFO.root[2].wrapping_mul(M::R) as i32),
62                mod_vec,
63            );
64            store_block_avx2(
65                a,
66                base + i,
67                add_vec_avx2::<M>(a0pa2, a1pa3, mod_vec, mod2_vec),
68            );
69            store_block_avx2(
70                a,
71                base + half + i,
72                sub_vec_avx2::<M>(a0pa2, a1pa3, mod_vec, mod2_vec),
73            );
74            store_block_avx2(
75                a,
76                base + v + i,
77                add_vec_avx2::<M>(a0na2, a1na3imag, mod_vec, mod2_vec),
78            );
79            store_block_avx2(
80                a,
81                base + v + half + i,
82                sub_vec_avx2::<M>(a0na2, a1na3imag, mod_vec, mod2_vec),
83            );
84            i += 1;
85        }
86        let k = s.trailing_ones() as usize;
87        w[0] = M::mod_mul(w[0], M::INFO.rate3[k]);
88        w[1] = M::mod_mul(w[1], M::INFO.rate3_2[k]);
89        w[2] = M::mod_mul(w[2], M::INFO.rate3_3[k]);
90        s += 1;
91        base += v << 1;
92    }

96unsafe fn ntt_blocks_avx2<M>(a: *mut u32, n: usize)
97where
98    M: Montgomery32NttModulus,
99{
100    let mod_vec = _mm256_set1_epi32(M::MOD as i32);
101    let mod2_vec = _mm256_set1_epi32(M::MOD.wrapping_add(M::MOD) as i32);
102    let imag_vec = _mm256_set1_epi32(M::INFO.root[2] as i32);
103
104    let mut v = n >> 1;
105    if n.trailing_zeros() & 1 == 1 {
106        let mut i = 0;
107        while i < v {
108            let x0 = load_block_avx2(a, i);
109            let x1 = load_block_avx2(a, v + i);
110            store_block_avx2(a, i, add_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec));
111            store_block_avx2(a, v + i, sub_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec));
112            i += 1;
113        }
114        v >>= 1;
115    }
116    while v > 1 {
117        ntt_block_stage_avx2::<M>(a, n, v, [M::N1; 3], mod_vec, mod2_vec, imag_vec);
118        v >>= 2;
119    }
120}

126unsafe fn intt_block_stage_avx2<M>(
127    a: *mut u32,
128    n: usize,
129    v: usize,
130    mut w: [u32; 3],
131    mod_vec: __m256i,
132    mod2_vec: __m256i,
133    iimag_vec: __m256i,
134    scale: u32,
135) where
136    M: Montgomery32NttModulus,
137{
138    let mut base = 0;
139    let mut s = 0usize;
140    while base < n {
141        let w1v = _mm256_set1_epi32(w[0] as i32);
142        let w2v = _mm256_set1_epi32(w[1] as i32);
143        let w3v = _mm256_set1_epi32(w[2] as i32);
144        let w1r = _mm256_set1_epi32(w[0].wrapping_mul(M::R) as i32);
145        let w2r = _mm256_set1_epi32(w[1].wrapping_mul(M::R) as i32);
146        let w3r = _mm256_set1_epi32(w[2].wrapping_mul(M::R) as i32);
147        let mut i = 0;
148        while i < v {
149            let x0 = load_block_avx2(a, base + i);
150            let x1 = load_block_avx2(a, base + v + i);
151            let x2 = load_block_avx2(a, base + (v << 1) + i);
152            let x3 = load_block_avx2(a, base + v * 3 + i);
153            let a0pa1 = add_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
154            let a0na1 = sub_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
155            let a2pa3 = add_vec_avx2::<M>(x2, x3, mod_vec, mod2_vec);
156            let a2na3 = sub_vec_avx2::<M>(x2, x3, mod_vec, mod2_vec);
157            let a2na3iimag = simd32::montgomery_mul_256_fixed(
158                a2na3,
159                iimag_vec,
160                _mm256_set1_epi32(M::INFO.inv_root[2].wrapping_mul(M::R) as i32),
161                mod_vec,
162            );
163            let y0 = add_vec_avx2::<M>(a0pa1, a2pa3, mod_vec, mod2_vec);
164            let y1 = add_vec_avx2::<M>(a0na1, a2na3iimag, mod_vec, mod2_vec);
165            let y2 = sub_vec_avx2::<M>(a0pa1, a2pa3, mod_vec, mod2_vec);
166            let y3 = sub_vec_avx2::<M>(a0na1, a2na3iimag, mod_vec, mod2_vec);
167            let (y0, y1, y2, y3) = if v == 1 {
168                (
169                    simd32::montgomery_mul_256_fixed(
170                        y0,
171                        _mm256_set1_epi32(scale as i32),
172                        _mm256_set1_epi32(scale.wrapping_mul(M::R) as i32),
173                        mod_vec,
174                    ),
175                    simd32::montgomery_mul_256_fixed(y1, w1v, w1r, mod_vec),
176                    simd32::montgomery_mul_256_fixed(y2, w2v, w2r, mod_vec),
177                    simd32::montgomery_mul_256_fixed(y3, w3v, w3r, mod_vec),
178                )
179            } else if s == 0 {
180                (y0, y1, y2, y3)
181            } else {
182                (
183                    y0,
184                    simd32::montgomery_mul_256_fixed(y1, w1v, w1r, mod_vec),
185                    simd32::montgomery_mul_256_fixed(y2, w2v, w2r, mod_vec),
186                    simd32::montgomery_mul_256_fixed(y3, w3v, w3r, mod_vec),
187                )
188            };
189            store_block_avx2(a, base + i, y0);
190            store_block_avx2(a, base + v + i, y1);
191            store_block_avx2(a, base + (v << 1) + i, y2);
192            store_block_avx2(a, base + v * 3 + i, y3);
193            i += 1;
194        }
195        let k = s.trailing_ones() as usize;
196        w[0] = M::mod_mul(w[0], M::INFO.inv_rate3[k]);
197        w[1] = M::mod_mul(w[1], M::INFO.inv_rate3_2[k]);
198        w[2] = M::mod_mul(w[2], M::INFO.inv_rate3_3[k]);
199        s += 1;
200        base += v << 2;
201    }
202}

204#[target_feature(enable = "avx2")]
205unsafe fn intt_blocks_avx2<M>(a: *mut u32, n: usize)
206where
207    M: Montgomery32NttModulus,
208{
209    let mod_vec = _mm256_set1_epi32(M::MOD as i32);
210    let mod2_vec = _mm256_set1_epi32(M::MOD.wrapping_add(M::MOD) as i32);
211    let iimag_vec = _mm256_set1_epi32(M::INFO.inv_root[2] as i32);
212    let limit = if n.trailing_zeros() & 1 == 1 {
213        n >> 1
214    } else {
215        n
216    };
217    let inv = M::mod_inv(<M as MIntConvert<u32>>::from(n as u32));
218
219    let mut v = 1;
220    while v < limit {
221        intt_block_stage_avx2::<M>(
222            a,
223            n,
224            v,
225            if v == 1 { [inv; 3] } else { [M::N1; 3] },
226            mod_vec,
227            mod2_vec,
228            iimag_vec,
229            inv,
230        );
231        v <<= 2;
232    }
233    if n.trailing_zeros() & 1 == 1 {
234        let half = n >> 1;
235        let mut i = 0;
236        while i < half {
237            let x0 = load_block_avx2(a, i);
238            let x1 = load_block_avx2(a, half + i);
239            let y0 = add_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
240            let y1 = sub_vec_avx2::<M>(x0, x1, mod_vec, mod2_vec);
241            store_block_avx2(a, i, y0);
242            store_block_avx2(a, half + i, y1);
243            i += 1;
244        }
245    }
246    let mut i = 0;
247    while i < n {
248        let x = load_block_avx2(a, i);
249        store_block_avx2(a, i, _mm256_min_epu32(x, _mm256_sub_epi32(x, mod_vec)));
250        i += 1;
251    }
252}

255#[target_feature(enable = "avx2")]
256unsafe fn reduce_sum_avx2(
257    even: __m256i,
258    odd: __m256i,
259    r_vec: __m256i,
260    mod_vec: __m256i,
261) -> __m256i {
262    let even_m = _mm256_mul_epu32(even, r_vec);
263    let odd_m = _mm256_mul_epu32(odd, r_vec);
264    let even = _mm256_add_epi64(even, _mm256_mul_epu32(even_m, mod_vec));
265    let odd = _mm256_add_epi64(odd, _mm256_mul_epu32(odd_m, mod_vec));
266    _mm256_or_si256(_mm256_bsrli_epi128::<4>(even), odd)
267}

269#[target_feature(enable = "avx2")]
270unsafe fn convolve_8_avx2<M>(f: *mut u32, g: *mut u32, n: usize)
271where
272    M: Montgomery32NttModulus,
273{
274    let mod_vec = _mm256_set1_epi32(M::MOD as i32);
275    let mod2_vec = _mm256_set1_epi32(M::MOD.wrapping_add(M::MOD) as i32);
276    let r_vec = _mm256_set1_epi32(M::R as i32);
277    let mut rr = M::N1;
278    let mut i = 0;
279    while i < n {
280        let rr_i = M::mod_mul(rr, M::INFO.root[2]);
281        let mut a = [[0u32; 16]; 4];
282        let mut b = [[0u32; 8]; 4];
283        for (j, ww) in [rr, M::MOD - rr, rr_i, M::MOD - rr_i]
284            .into_iter()
285            .enumerate()
286        {
287            let k = i + j;
288            let ff = load_block_avx2(f, k);
289            let gg = load_block_avx2(g, k);
290            let ff = _mm256_min_epu32(ff, _mm256_sub_epi32(ff, mod_vec));
291            let gg = _mm256_min_epu32(gg, _mm256_sub_epi32(gg, mod_vec));
292            // fw < 5 * MOD / 4, so the eight-product sum still reduces below 7 * MOD / 2.
293            let fw = simd32::montgomery_mul_256_fixed(
294                ff,
295                _mm256_set1_epi32(ww as i32),
296                _mm256_set1_epi32(ww.wrapping_mul(M::R) as i32),
297                mod_vec,
298            );
299            _mm256_storeu_si256(a[j].as_mut_ptr() as *mut __m256i, fw);
300            _mm256_storeu_si256(a[j].as_mut_ptr().add(8) as *mut __m256i, ff);
301            _mm256_storeu_si256(b[j].as_mut_ptr() as *mut __m256i, gg);
302        }
303        let mut even = [_mm256_setzero_si256(); 4];
304        let mut odd = [_mm256_setzero_si256(); 4];
305        let mut l = 0;
306        while l < 8 {
307            let mut j = 0;
308            while j < 4 {
309                let x = _mm256_loadu_si256(a[j].as_ptr().add(8 - l) as *const __m256i);
310                let y = _mm256_set1_epi32(b[j][l] as i32);
311                even[j] = _mm256_add_epi64(even[j], _mm256_mul_epu32(x, y));
312                odd[j] = _mm256_add_epi64(odd[j], _mm256_mul_epu32(_mm256_bsrli_epi128::<4>(x), y));
313                j += 1;
314            }
315            l += 1;
316        }
317        let mut j = 0;
318        while j < 4 {
319            let x = reduce_sum_avx2(even[j], odd[j], r_vec, mod_vec);
320            store_block_avx2(f, i + j, _mm256_min_epu32(x, _mm256_sub_epi32(x, mod2_vec)));
321            j += 1;
322        }
323        i += 4;
324        rr = M::mod_mul(rr, M::INFO.rate3[(i >> 2).trailing_zeros() as usize]);
325    }
