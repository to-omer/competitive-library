247    pub fn inv(&self, deg: usize) -> Self {
248        if deg == 0 {
249            return Self::zero();
250        }
251        debug_assert!(!self[0].is_zero());
252        let sparse_limit = deg.next_power_of_two().trailing_zeros() as usize * 6;
253        if self
254            .data
255            .iter()
256            .take(deg)
257            .filter(|x| !x.is_zero())
258            .nth(sparse_limit)
259            .is_none()
260        {
261            let pos: Vec<_> = self
262                .data
263                .iter()
264                .take(deg)
265                .enumerate()
266                .skip(1)
267                .filter_map(|(i, x)| if x.is_zero() { None } else { Some(i) })
268                .collect();

270            f[0] = T::one() / self[0].clone();
271            for i in 1..deg {
272                let mut tot = T::zero();
273                for &j in &pos {
274                    if j > i {
275                        break;
276                    }
277                    tot += self[j].clone() * &f[i - j];
278                }
279                f[i] = -tot * &f[0];
280            }
281            return f;
282        }
283        let mut f = Self::from(T::one() / self[0].clone());
284        f.data.reserve(deg.saturating_sub(1));
285        let mut error = Vec::with_capacity(deg.next_power_of_two());
286        let mut i = 1;
287        while i < deg {
288            error.clear();
289            error.extend(
290                self.data[..(i * 2).min(deg).min(self.length())]
291                    .iter()
292                    .cloned(),
293            );
294            let factor = C::transform(f.data.clone(), 2 * i);
295            let mut error_fft = C::transform(error, 2 * i);
296            C::multiply(&mut error_fft, &factor);
297            error = C::inverse_transform(error_fft, 2 * i);
298            error.drain(..i);
299            let mut error_fft = C::transform(error, 2 * i);
300            C::multiply(&mut error_fft, &factor);
301            error = C::inverse_transform(error_fft, 2 * i);
302            error.truncate(i.min(deg - i));
303            f.data.extend(error.drain(..).map(Neg::neg));
304            i *= 2;
305        }
306        f

308    pub fn exp(&self, deg: usize) -> Self
309    where
310        C: NttReuse<T = Vec<T>>,
311        C::F: Clone,
312    {
313        if deg == 0 {
314            return Self::zero();
315        }
316        debug_assert!(self[0].is_zero());
317        let sparse_limit = deg.next_power_of_two().trailing_zeros() as usize * 16;
318        if self
319            .data
320            .iter()
321            .take(deg)
322            .filter(|x| !x.is_zero())
323            .nth(sparse_limit)
324            .is_none()
325        {
326            let diff = self.prefix_ref(deg).clone().diff();
327            let pos: Vec<_> = diff
328                .data
329                .iter()
330                .enumerate()
331                .filter_map(|(i, x)| if x.is_zero() { None } else { Some(i) })
332                .collect();
333            let mf = T::memorized_factorial(deg);
334            let mut f = Self::zeros(deg);
335            f[0] = T::one();
336            for i in 1..deg {
337                let mut tot = T::zero();
338                for &j in &pos {
339                    if j > i - 1 {
340                        break;
341                    }
342                    tot += f[i - 1 - j].clone() * &diff[j];
343                }
344                f[i] = tot * T::memorized_inv(&mf, i);
345            }
346            return f;
347        }
348        let indices: Vec<_> = (0..=deg).map(T::from).collect();
349        let modulus = <T::Base as MIntConvert<usize>>::mod_into();
350        let mut inv = vec![T::zero(); deg + 1];
351        inv[1] = T::one();
352        for i in 2..=deg {
353            inv[i] = -T::from(modulus / i) * &inv[modulus % i];
354        }
355        let block = deg.next_power_of_two() / 32;
356        let (kernel, mut kernel_inverse, previous_inverse_fft) =
357            self.prefix_ref(block).exp_newton(block, &indices, &inv);
358        let mut error_fft = C::transform_ntt(kernel.data.clone(), block);
359        C::multiply_prefix(&mut error_fft, &previous_inverse_fft);
360        let error = C::inverse_transform_ntt(error_fft, block);
361        let mut error_fft = C::transform_ntt(error.into_iter().skip(block / 2).collect(), block);
362        C::multiply_prefix(&mut error_fft, &previous_inverse_fft);
363        let error = C::inverse_transform_ntt(error_fft, block / 2);
364        kernel_inverse
365            .data
366            .extend(error.into_iter().take(block / 2).map(Neg::neg));
367        let kernel_data = kernel.data;
368        let kernel_inverse_data = kernel_inverse.data;
369        let kernel_inverse = C::transform(kernel_inverse_data, block * 2);
370        let kernel = C::transform(kernel_data.clone(), block * 2);
371        let blocks = deg.div_ceil(block);
372        let mut derivative_ffts = Vec::with_capacity(blocks);
373        for q in 0..blocks {
374            let start = q * block;
375            let mut values = Self::zeros(block * 2);
376            for i in 0..block
377                .min(deg.saturating_sub(start))
378                .min(self.length().saturating_sub(start))
379            {
380                values[i] = self[start + i].clone() * &indices[start + i];
381            }
382            if q > 0 {
383                let start = start - block;
384                for i in 0..block.min(self.length().saturating_sub(start)) {
385                    values[block + i] = self[start + i].clone() * &indices[start + i];
386                }
387            }
388            derivative_ffts.push(C::transform_ntt(values.data, block * 2));
389        }
390
391        let mut result = kernel_data.clone();
392        result.reserve(deg - block);
393        let mut result_ffts = Vec::with_capacity(blocks - 1);
394        for q in 1..blocks {
395            result_ffts.push(C::transform_ntt(
396                result[(q - 1) * block..q * block].to_vec(),
397                block * 2,
398            ));
399            let mut sum_fft = derivative_ffts[q].clone();
400            C::multiply_prefix(&mut sum_fft, &result_ffts[0]);
401            for j in 1..q {
402                C::multiply_add(&mut sum_fft, &derivative_ffts[q - j], &result_ffts[j]);
403            }
404            let values = C::inverse_transform_ntt(sum_fft, block);
405            let mut values = C::transform(values, block * 2);
406            C::multiply(&mut values, &kernel_inverse);
407            let mut values = C::inverse_transform(values, block * 2);
408            values.truncate(block);
409            let len = block.min(deg - q * block);
410            for (i, value) in values.iter_mut().take(len).enumerate() {
411                *value *= &inv[q * block + i];
412            }
413            values[len..].fill(T::zero());
414            let mut values = C::transform(values, block * 2);
415            C::multiply(&mut values, &kernel);
416            let mut values = C::inverse_transform(values, block * 2);
417            values.truncate(len);
418            result.extend(values);
419        }
420        Self::from_vec(result)
421    }

423    fn exp_newton(&self, deg: usize, indices: &[T], inv: &[T]) -> (Self, Self, C::F)
424    where
425        C: NttReuse<T = Vec<T>>,
426        C::F: Clone,
427    {
428        if deg == 1 {
429            let one = Self::one();
430            return (one.clone(), one.clone(), C::transform_ntt(one.data, 1));
431        }
432        let mut f = Self::from_vec(vec![T::one(), self.coeff(1)]);
433        let mut inverse = Self::one();
434        let mut inverse_fft = C::transform_ntt(inverse.data.clone(), 2);
435        let mut m = 2;
436        while m < deg {
437            let f_fft = C::transform_ntt(f.data.clone(), 2 * m);
438
439            let previous_inverse_fft = inverse_fft;
440            let mut error_fft = previous_inverse_fft.clone();
441            C::multiply_prefix(&mut error_fft, &f_fft);
442            let mut error = C::inverse_transform_ntt(error_fft, m);
443            error[..m / 2].fill(T::zero());
444            let mut error_fft = C::transform_ntt(error, m);
445            C::multiply_prefix(&mut error_fft, &previous_inverse_fft);
446            let error = C::inverse_transform_ntt(error_fft, m);
447            inverse
448                .data
449                .extend(error.into_iter().skip(m / 2).map(Neg::neg));
450            inverse_fft = C::transform_ntt(inverse.data.clone(), 2 * m);
451
452            let mut delta = Self::from_vec(
453                self.data
454                    .iter()
455                    .take(m)
456                    .enumerate()
457                    .skip(1)
458                    .map(|(i, value)| value.clone() * &indices[i])
459                    .collect(),
460            );
461            delta.resize(m);
462            let mut delta_fft = C::transform_ntt(delta.data, m);
463            C::multiply_prefix(&mut delta_fft, &f_fft);
464            let mut delta = Self::from_vec(C::inverse_transform_ntt(delta_fft, m));
465            for i in 1..f.length() {
466                delta[i - 1] -= f[i].clone() * &indices[i];
467            }
468            delta.resize(2 * m);
469            for i in (0..m - 1).rev() {
470                delta.data[m + i] = delta.data[i].clone();
471            }
472            delta.data[..m - 1].fill(T::zero());
473            let mut delta_fft = C::transform_ntt(delta.data, 2 * m);
474            C::multiply_prefix(&mut delta_fft, &inverse_fft);
475            let mut delta = C::inverse_transform_ntt(delta_fft, 2 * m);
476            delta.pop();
477            delta.push(T::zero());
478            let target = (2 * m).min(deg);
479            for i in (1..target).rev() {
480                delta[i] = delta[i - 1].clone() * &inv[i];
481            }
482            delta[0] = T::zero();
483            delta[target..].fill(T::zero());
484            for i in m..(2 * m).min(self.length()) {
485                delta[i] += self[i].clone();
486            }

488            let mut delta_fft = C::transform_ntt(delta, 2 * m);
489            C::multiply_prefix(&mut delta_fft, &f_fft);
490            let delta = C::inverse_transform_ntt(delta_fft, 2 * m);
491            f.data
492                .extend(delta.into_iter().skip(m).take((deg - m).min(m)));
493            m *= 2;
494        }

500    pub fn pow(&self, rhs: usize, deg: usize) -> Self
501    where
502        C: NttReuse<T = Vec<T>>,
503        C::F: Clone,
504    {
505        if rhs == 0 {
506            return Self::from_vec(

509                    .take(deg)
510                    .collect(),
511            );

522                {

524                } else {
525                    f = (f.log(deg) * &T::from(rhs)).exp(deg);
526                }
527                f *= x0.pow(rhs);
528                f <<= k * rhs;
529                f

532            Self::zeros(deg)
533        }

568    pub fn solve_sparse_differential2(
569        p: &Self,
570        q: &Self,
571        x: &Self,
572        alpha: T,
573        beta: T,
574        deg: usize,
575    ) -> Self {
576        if deg == 0 {
577            return Self::zero();
578        }
579        let collect_sparse = |p: &Self| -> Vec<(usize, T)> {
580            p.iter()
581                .enumerate()
582                .filter(|&(_, x)| !x.is_zero())
583                .map(|(i, x)| (i, x.clone()))
584                .collect()
585        };
586        assert!(q.coeff(0).is_one());
587        assert!(x.coeff(0).is_one());
588        let p = collect_sparse(p);
589        let q = collect_sparse(q);
590        let x = collect_sparse(x);
591        let diff = |p: &[(usize, T)]| -> Vec<(usize, T)> {
592            p.iter()
593                .filter(|&&(i, _)| i > 0)
594                .map(|&(i, ref x)| (i - 1, x.clone() * T::from(i)))
595                .collect()
596        };
597        let dp = diff(&p);
598        let dq = diff(&q);
599
600        let mf = T::memorized_factorial(deg);
601        let mut f = Self::zeros(deg);
602        let mut qf = Self::zeros(deg);
603        let mut dq_f = Self::zeros(deg);
604        let mut d_qf = Self::zeros(deg);
605        f[0] = T::one();
606        for i in 0..deg - 1 {
607            qf[i] = f.sparse_fold(q.iter().cloned(), i);
608            dq_f[i] = f.sparse_fold(dq.iter().cloned(), i);
609            let dp_qf_i = qf.sparse_fold(dp.iter().cloned(), i);
610            let p_dq_f_i = dq_f.sparse_fold(p.iter().cloned(), i);
611            let x_d_qf_i = d_qf.sparse_fold(
612                x.iter()
613                    .map(|&(i, ref x)| (i, x.clone() - T::from((i == 0) as usize))),
614                i,
615            );
616            d_qf[i] = alpha.clone() * dp_qf_i + beta.clone() * p_dq_f_i - x_d_qf_i;

669            .saturating_add(exp_q.saturating_mul(sq as _)) as usize;

736                f = f.pow_sparse1(T::one() / T::from(2usize), deg);

738                return Some(f);

743            let mut i = 1;

745                f = (&f + &(self.prefix_ref(i * 2) * f.inv(i * 2))).prefix(i * 2) * &inv2;

747            }
748            f.truncate(deg);
749            return Some(f);
