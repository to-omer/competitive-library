288fn convolve_naive<T>(a: &[T], b: &[T]) -> Vec<T>
289where
290    T: Copy + Zero + AddAssign<T> + Mul<Output = T>,
291{
292    if a.is_empty() && b.is_empty() {
293        return Vec::new();
294    }
295    let len = a.len() + b.len() - 1;
296    let mut c = vec![T::zero(); len];
297    if a.len() < b.len() {
298        for (i, &b) in b.iter().enumerate() {
299            for (a, c) in a.iter().zip(&mut c[i..]) {
300                *c += *a * b;
301            }
302        }
303    } else {
304        for (i, &a) in a.iter().enumerate() {
305            for (b, c) in b.iter().zip(&mut c[i..]) {
306                *c += *b * a;
307            }
308        }
309    }
310    c
311}

313fn convolve_karatsuba<T>(a: &[T], b: &[T]) -> Vec<T>
314where
315    T: Copy + Zero + AddAssign<T> + SubAssign<T> + Mul<Output = T>,
316{
317    if a.len().min(b.len()) <= 30 {
318        return convolve_naive(a, b);
319    }
320    let m = a.len().max(b.len()).div_ceil(2);
321    let (a0, a1) = if a.len() <= m {
322        (a, &[][..])
323    } else {
324        a.split_at(m)
325    };
326    let (b0, b1) = if b.len() <= m {
327        (b, &[][..])
328    } else {
329        b.split_at(m)
330    };
331    let f00 = convolve_karatsuba(a0, b0);
332    let f11 = convolve_karatsuba(a1, b1);
333    let mut a0a1 = a0.to_vec();
334    for (a0a1, &a1) in a0a1.iter_mut().zip(a1) {
335        *a0a1 += a1;
336    }
337    let mut b0b1 = b0.to_vec();
338    for (b0b1, &b1) in b0b1.iter_mut().zip(b1) {
339        *b0b1 += b1;
340    }
341    let mut f01 = convolve_karatsuba(&a0a1, &b0b1);
342    for (f01, &f00) in f01.iter_mut().zip(&f00) {
343        *f01 -= f00;
344    }
345    for (f01, &f11) in f01.iter_mut().zip(&f11) {
346        *f01 -= f11;
347    }
348    let mut c = vec![T::zero(); a.len() + b.len() - 1];
349    for (c, &f00) in c.iter_mut().zip(&f00) {
350        *c += f00;
351    }
352    for (c, &f01) in c[m..].iter_mut().zip(&f01) {
353        *c += f01;
354    }
355    for (c, &f11) in c[m << 1..].iter_mut().zip(&f11) {
356        *c += f11;
357    }
358    c
359}

361impl<M> ConvolveSteps for Convolve<M>
362where
363    M: Montgomery32NttModulus,
364{
365    type T = Vec<MInt<M>>;
366    type F = Vec<MInt<M>>;
367    fn length(t: &Self::T) -> usize {
368        t.len()
369    }
370    fn transform(mut t: Self::T, len: usize) -> Self::F {
371        t.resize_with(len.max(1).next_power_of_two(), Zero::zero);
372        ntt(&mut t);
373        t
374    }
375    fn inverse_transform(mut f: Self::F, len: usize) -> Self::T {
376        intt(&mut f);
377        f.truncate(len);
378        f
379    }
380    fn multiply(f: &mut Self::F, g: &Self::F) {
381        assert_eq!(f.len(), g.len());
382        #[cfg(target_arch = "x86_64")]
383        match simd_backend() {
384            SimdBackend::Avx512 => unsafe { ntt_simd::pointwise_multiply_avx512(f, g) },
385            SimdBackend::Avx2 => unsafe { ntt_simd::pointwise_multiply_avx2(f, g) },
386            SimdBackend::Scalar => {
387                for (f, g) in f.iter_mut().zip(g.iter()) {
388                    *f *= *g;
389                }
390            }
391        }
392        #[cfg(not(target_arch = "x86_64"))]
393        for (f, g) in f.iter_mut().zip(g.iter()) {
394            *f *= *g;
395        }
396    }
397    fn convolve(mut a: Self::T, mut b: Self::T) -> Self::T {
398        if Self::length(&a).max(Self::length(&b)) <= 100 {
399            return convolve_karatsuba(&a, &b);
400        }
401        if Self::length(&a).min(Self::length(&b)) <= 60 {
402            return convolve_naive(&a, &b);
403        }
404        let len = (Self::length(&a) + Self::length(&b)).saturating_sub(1);
405        let size = len.max(1).next_power_of_two();
406        if len <= size / 2 + 2 {
407            let xa = a.pop().unwrap();
408            let xb = b.pop().unwrap();
409            let mut c = vec![MInt::<M>::zero(); len];
410            *c.last_mut().unwrap() = xa * xb;
411            for (a, c) in a.iter().zip(&mut c[b.len()..]) {
412                *c += *a * xb;
413            }
414            for (b, c) in b.iter().zip(&mut c[a.len()..]) {
415                *c += *b * xa;
416            }
417            let d = Self::convolve(a, b);
418            for (d, c) in d.into_iter().zip(&mut c) {
419                *c += d;
420            }
421            return c;
422        }
423        let same = a == b;
424        #[cfg(target_arch = "x86_64")]
425        if M::MOD < 1 << 30 && simd_backend() == SimdBackend::Avx2 {
426            a.resize_with(size, Zero::zero);
427            b.resize_with(size, Zero::zero);
428            unsafe { ntt_simd::convolve_blocks_avx2(&mut a, &mut b, same) };
429            a.truncate(len);
430            return a;
431        }
432        let mut a = Self::transform(a, len);
433        if same {
434            for a in a.iter_mut() {
435                *a *= *a;
436            }
437        } else {
438            let b = Self::transform(b, len);
439            Self::multiply(&mut a, &b);
440        }
441        Self::inverse_transform(a, len)
442    }
