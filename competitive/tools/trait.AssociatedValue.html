145pub fn fft(a: &mut [Complex<f64>]) {
146    let n = a.len();
147    RotateCache::ensure(n / 2);
148    RotateCache::with(|cache| {
149        let mut v = n / 2;
150        while v >= 2 {
151            let l = v / 2;
152            for (q, block) in a.chunks_exact_mut(l * 4).enumerate() {
153                let (a, rest) = block.split_at_mut(l);
154                let (b, rest) = rest.split_at_mut(l);
155                let (c, d) = rest.split_at_mut(l);
156                let w0 = cache[q];
157                let w1 = cache[q << 1];
158                let w2 = cache[q << 1 | 1];
159                for i in 0..l {
160                    let cv = c[i] * w0;
161                    let dv = d[i] * w0;
162                    let ac0 = a[i] + cv;
163                    let ac1 = a[i] - cv;
164                    let bd0 = (b[i] + dv) * w1;
165                    let bd1 = (b[i] - dv) * w2;
166                    a[i] = ac0 + bd0;
167                    b[i] = ac0 - bd0;
168                    c[i] = ac1 + bd1;
169                    d[i] = ac1 - bd1;
170                }
171            }
172            v >>= 2;
173        }
174        if v == 1 {
175            for (a, w) in a.chunks_exact_mut(2).zip(cache) {
176                let y = a[1] * *w;
177                a[1] = a[0] - y;
178                a[0] += y;
179            }
180        }
181    });

304unsafe fn bit_reverse_soa(a: &mut Vec<Complex4>) {
305    let quarter = a.len() / 4;
306    let shift = usize::BITS - quarter.trailing_zeros();
307    let mut result = vec![Complex4::default(); a.len()];
308    for i in 0..quarter {
309        let j = i.reverse_bits() >> shift;
310        let (r0, i0) = load4(&a[j]);
311        let (r1, i1) = load4(&a[j + quarter * 2]);
312        let (r2, i2) = load4(&a[j + quarter]);
313        let (r3, i3) = load4(&a[j + quarter * 3]);
314        let re = transpose4(r0, r1, r2, r3);
315        let im = transpose4(i0, i1, i2, i3);
316        store4(&mut result[i], re[0], im[0]);
317        store4(&mut result[i + quarter], re[2], im[2]);
318        store4(&mut result[i + quarter * 2], re[1], im[1]);
319        store4(&mut result[i + quarter * 3], re[3], im[3]);
320    }
321    *a = result;
322}

333    a[middle >> 2].set(middle & 3, z);
