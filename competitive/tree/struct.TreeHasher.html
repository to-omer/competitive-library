    pub fn hash(&mut self, g: &UndirectedSparseGraph) -> u64 {
        match g.tree_center() {
            TreeCenter::One(u) => self.hash_rec(g, u, !0, 0),
            TreeCenter::Two(u, v) => {
                Self::mersenne_mul_mod(self.hash_rooted(g, u, v), self.hash_rooted(g, v, u))
            }
        }
    }
