Expand description

A self-adjusting top tree, also called a strong link-cut tree.

This is not a classical worst-case-balanced top tree. Circular order and select are not supported.

462    fn access_node(&mut self, node: TopPtr<S, A>) {
463        unsafe {
464            let mut previous: Option<TopPtr<S, A>> = None;
465            let mut current = Some(node);
466            while let Some(mut cursor) = current {
467                Self::splay_top(cursor);
468                let next = cursor.as_ref().parent.parent;
469                if let Some(right) = cursor.as_mut().child[1].take() {
470                    let point = S::add_edge(&right.as_ref().data.sum);
471                    let (light, entry) = self.rake_insert(cursor.as_ref().data.light, point);
472                    cursor.as_mut().data.light = Some(light);
473                    (*right.as_ptr()).data.belong = Some(entry);
474                }
475                if let Some(previous) = previous {
476                    let entry = (*previous.as_ptr())
477                        .data
478                        .belong
479                        .take()
480                        .expect("a virtual path must have a rake-tree entry");
481                    let (light, action) = self.rake_remove(entry);
482                    cursor.as_mut().data.light = light;
483                    TopBstSpec::<S, A>::apply_all(previous, &action);
484                    cursor.as_mut().child[1] = Some(previous);
485                    (*previous.as_ptr()).parent.parent = Some(cursor);
486                }
487                Self::pull_top(cursor);
488                previous = Some(cursor);
489                current = next;
490            }
491            Self::splay_top(node);
492        }
