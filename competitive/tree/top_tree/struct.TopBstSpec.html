377    unsafe fn rake_insert(
378        &mut self,

402    unsafe fn rake_remove(
403        &mut self,
404        mut node: RakePtr<S, A>,
405    ) -> (Option<RakePtr<S, A>>, A::Action) {
406        unsafe {
407            Self::splay_rake(node);
408            RakeBstSpec::<S, A>::top_down(BstDataMutRef::new_unchecked(node));
409        }
410        let left = unsafe { node.as_mut().child[0].take() };
411        let right = unsafe { node.as_mut().child[1].take() };
412        for mut child in [left, right].into_iter().flatten() {
413            unsafe { child.as_mut().parent.parent = None };
414        }
415        let root = match (left, right) {
416            (None, right) => right,
417            (left, None) => left,
418            (Some(left), Some(right)) => {
419                let mut root = unsafe { Self::rake_rightmost(left) };
420                unsafe {
421                    Self::splay_rake(root);
422                    root.as_mut().child[1] = Some(right);
423                    (*right.as_ptr()).parent.parent = Some(root);
424                    Self::pull_rake(root);
425                }
426                Some(root)
427            }
428        };
429        let node = self.rake_allocator.deallocate(node);
430        (root, node.data.buffer)
431    }

433    fn access_node(&mut self, node: TopPtr<S, A>) {
434        unsafe {
435            let mut previous: Option<TopPtr<S, A>> = None;
436            let mut current = Some(node);
437            while let Some(mut cursor) = current {
438                Self::splay_top(cursor);
439                let next = cursor.as_ref().parent.parent;
440                if let Some(right) = cursor.as_mut().child[1].take() {
441                    let point = S::add_edge(&right.as_ref().data.sum);
442                    let (light, entry) = self.rake_insert(cursor.as_ref().data.light, point);
443                    cursor.as_mut().data.light = Some(light);
444                    (*right.as_ptr()).data.belong = Some(entry);
445                }
446                if let Some(previous) = previous {
447                    let entry = (*previous.as_ptr())
448                        .data
449                        .belong
450                        .take()
451                        .expect("a virtual path must have a rake-tree entry");
452                    let (light, action) = self.rake_remove(entry);
453                    cursor.as_mut().data.light = light;
454                    TopBstSpec::<S, A>::apply_all(previous, &action);
455                    cursor.as_mut().child[1] = Some(previous);
456                    (*previous.as_ptr()).parent.parent = Some(cursor);
457                }
458                Self::pull_top(cursor);
459                previous = Some(cursor);
460                current = next;
461            }
462            Self::splay_top(node);
463        }
464    }
