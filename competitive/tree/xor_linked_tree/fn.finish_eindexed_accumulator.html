563impl Iterator for Children<'_> {
564    type Item = usize;
565    fn next(&mut self) -> Option<usize> {
566        if self.next == self.end {
567            None
568        } else {
569            let v = self.dfs.order[self.next];
570            self.next = self.dfs.subtree_end[v];
571            Some(v)
572        }
573    }
