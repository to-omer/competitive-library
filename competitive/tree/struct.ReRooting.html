dynamic programming on all-rooted trees

caluculate all subtrees (hanging on the edge) in specific ordering, each subtree calculated in the order of merge and rooting

Fields§

§dp: Vec<M::T>

dp[v]: result of v-rooted tree

§ep: Vec<M::T>

ep[e]: result of e-subtree, if e >= n then reversed-e-subtree

pub fn grl_5_b(reader: impl Read, mut writer: impl Write) {
    let s = read_all_unchecked(reader);
    let mut scanner = Scanner::new(&s);
    scan!(scanner, n, (graph, w): @TreeGraphScanner::<usize, u64>::new(n));
    let re = ReRooting::<MaxOperation<u64>, _>::new(&graph, |d, _vid, eid_opt| {
        d + eid_opt.map_or(0, |eid| w[eid])
    });
    iter_print!(writer, @sep '\n', @it re.dp);
}
